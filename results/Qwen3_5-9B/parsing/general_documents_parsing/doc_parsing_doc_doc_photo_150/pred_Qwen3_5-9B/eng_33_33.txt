\section*{NCERT Solutions for class 10 Maths}
\subsection*{Chapter 1 - Real Number Ex-1.1}

\begin{enumerate}
    \item Express each number as a product of its prime factors:
    \begin{enumerate}
        \item[(i)] 140
        \begin{itemize}
            \item Prime factors of 140 = 2, 2, 5, 7
            \item = $2^2 \times 5 \times 7$
        \end{itemize}
        \item[(ii)] 156
        \begin{itemize}
            \item Prime factors of 156 = $2 \times 2 \times 3 \times 13$
            \item = $2^2 \times 3 \times 13$
        \end{itemize}
        \item[(iii)] 3825
        \begin{itemize}
            \item Prime factors of 3825 = $3 \times 3 \times 5 \times 5 \times 17$
            \item = $3^2 \times 5^2 \times 17$
        \end{itemize}
        \item[(iv)] 5005
        \begin{itemize}
            \item Prime factors of 5005 = $5 \times 7 \times 11 \times 13$
        \end{itemize}
        \item[(v)] 7429
        \begin{itemize}
            \item Prime factors of 7429 = $17 \times 19 \times 23$
        \end{itemize}
    \end{enumerate}

    \item Find the LCM and HCF of the following pairs of integers and verify that LCM $\times$ HCF = product of the two numbers
    \begin{enumerate}
        \item[(i)] 26 and 91
        \begin{itemize}
            \item Prime factors of 26 = $2 \times 13$
            \item Prime factors of 91 = $7 \times 13$
            \item HCF of 26 and 91 = 13
            \item LCM of 26 and 91 = $2 \times 7 \times 13$
            \item = 182
            \item Product of two numbers = $26 \times 91 = 2366$
            \item LCM $\times$ HCF = $182 \times 13 = 2366$
            \item So, Product of two numbers = LCM $\times$ HCF
        \end{itemize}
    \end{enumerate}
\end{enumerate}
