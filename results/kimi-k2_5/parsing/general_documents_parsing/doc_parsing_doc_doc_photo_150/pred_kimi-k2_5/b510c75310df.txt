\section*{2.5\quad Finance formulas -- Practice exercises}

\noindent\rule{\textwidth}{0.4pt}

\medskip

\noindent\textbf{\textsc{Compound Interest Formula:}}\quad $\displaystyle a = p\left(1 + \frac{r}{12}\right)^{12y}$

\medskip

\noindent\textbf{\textsc{Equivalent APR Formula:}}\quad $\displaystyle \text{APR} = \left(1 + \frac{r}{12}\right)^{12} - 1$

\medskip

\noindent\textbf{\textsc{Future Value Annuity Formula:}}\quad $\displaystyle a = p * \frac{\left(1 + \frac{r}{12}\right)^{12y} - 1}{\frac{r}{12}}$

\medskip

\noindent\textbf{\textsc{Loan Payment Formula:}}\quad $\displaystyle p = \frac{a * \frac{r}{12}}{1 - \left(1 + \frac{r}{12}\right)^{-12y}}$

\medskip

\noindent where

\begin{tabular}{@{}l@{\ }l@{}}
$a$ & = account balance or loan amount (\$) \\
$p$ & = initial deposit (principal), regular deposit, or regular payment (\$) \\
$y$ & = time invested (years) \\
$r$ & = interest rate compounded monthly (as a decimal)
\end{tabular}
