    \item \textbf{(4)}
    \begin{align*}
        2x + 3y &= 7 \\
        2x &= 7 - 3y \\
        x &= \frac{7}{2} - \frac{3}{2}y \\
        x &= -\frac{3}{2}y + \frac{7}{2}
    \end{align*}

    \item \textbf{(2)}
    \begin{align*}
        3x + 4y &= 15 \\
        6x + 8y &= 2(3x + 4y) \\
        &= 30 \\
        15x + 20y &= 5(3x + 4y) \\
        &= 75
    \end{align*}

    \item \textbf{(3)}
    \begin{align*}
        4x &\geqslant \frac{1}{2} \\
        \frac{4x}{4} &\geqslant \frac{\left(\frac{1}{2}\right)}{4} \\
        x &\geqslant \frac{1}{8}
    \end{align*}

    \item \textbf{(3)}
    \begin{align*}
        6 - 8x &< -2x \\
        6 - 8x + 8x &< -2x + 8x \\
        6 &< 6x \\
        6x &> 6 \\
        x &> 1
    \end{align*}

    \item \textbf{(2)}
    \begin{align*}
        \text{For rectangle } P, \text{ breadth} &= \frac{120}{15} \\
        &= 8 \text{ cm} \\
        \text{Perimeter of } Q &= \text{Perimeter of } P \\
        &= 2(8 + 15) \\
        &= 46 \text{ cm}
    \end{align*}

    \item \textbf{(4)}
    \begin{align*}
        \text{Circumference of } Y &= \text{Circumference of } X \\
        &= 10\pi \\
        2\pi r &= 10\pi \\
        r &= 5 \text{ cm} \\
        \text{Area of circle } Y &= \pi(5^2) \\
        &= 25\pi \text{ cm}^2
    \end{align*}

    \item \textbf{(3)}
    \begin{align*}
        \text{ratio of sides} &= 1 : 10 \\
        \text{ratio of area} &= 1 : 100
    \end{align*}

    \item \textbf{(3)}
    \begin{align*}

        \text{ratio of perimeter} &= 1 : 5
    \end{align*}

    \item \textbf{(3)}
    \begin{align*}
        \text{Let } H \text{ and } h \text{ be the heights of the 2 candles.} \\
        \text{Burnt amount at any time:} \\
        H: \frac{120}{180} \times H &= \frac{2}{3}H \\
        h: \frac{120}{300} \times h &= \frac{2}{5}h \\
        \text{Remaining:} \\
        H: H - \frac{2}{3}H &= \frac{1}{3}H \\
        h: h - \frac{2}{5}h &= \frac{3}{5}h
    \end{align*}
    The ratio after 2 h is $1 : 1$.
    \begin{align*}
        \frac{\frac{1}{3}H}{\frac{3}{5}h} &= \frac{1}{1} \\
        \text{Cross multiply:} \\
        \frac{1}{3}H &= \frac{3}{5}h \\
        \text{Cross multiply:} \\
        5H &= 9h \\
        \frac{H}{h} &= \frac{9}{5} \\
        H : h &= 9 : 5
    \end{align*}

    \item \textbf{(1)}
    \begin{align*}
        \text{Height after } n \text{ min:} \\
        1^{\text{st}} &: \frac{n}{180} \\
        2^{\text{nd}} &: \frac{n}{240} \\
        \text{Height remaining:} \\
        1^{\text{st}} &: 1 - \frac{n}{180} \\
        2^{\text{nd}} &: 1 - \frac{n}{240} \\
        \text{The ratio became } 1 : 2. \\
        \frac{1 - \frac{n}{180}}{1 - \frac{n}{240}} &= \frac{1}{2} \\
        \text{Cross multiply:} \\
        2 - \frac{n}{90} &= 1 - \frac{n}{240} \\
        \frac{n}{90} - \frac{n}{240} &= 2 - 1 \\
        \frac{8n - 3n}{720} &= 1 \\
        5n &= 720 \\
        n &= 144 \text{ min} = 2 \text{ h } 24 \text{ min} \\
        \text{Working backwards: } 14 \text{ h } 36 \text{ min}
    \end{align*}

    \item \textbf{(2)}
    \begin{align*}
        \text{When } B \text{ meets } C, \text{ the distance between } A \text{ and } C \text{ is} \\
        (50 + 70) \times 2 &= 240 \text{ km} \\
        \text{Difference in speed} &\rightarrow 60 - 50 = 10 \text{ km/h} \\
        240 \div 10 &= 24 \text{ h} \\
        (60 + 70) \times 24 &= 3120 \text{ km}
    \end{align*}

    \item \textbf{(4)}
    \begin{align*}
        15 - 5 &= 10 \text{ h} \\

        \text{Ans: } 09 \text{ h } 00 \text{ min}
    \end{align*}

        4 \text{ cubes} &\rightarrow 6 \times 4 = 24 \text{ faces} \\
        24 - 18 &= 6 \text{ faces} \\
        6 \text{ faces} &\rightarrow 216 \text{ cm}^2
    \end{align*}
\end{enumerate}
