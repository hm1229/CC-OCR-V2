\section*{Mechanics}

\[
x = x_0 + v_{x0}t + \frac{1}{2}\vec{a}_x t^2
\]

\[
v = v_0 + at
\]

\[
v_x^2 - v_{x0}^2 = 2a(x - x_0)
\]

\[
\vec{a} = \frac{\sum \vec{F}}{m} = \frac{\vec{F}_{\text{net}}}{m}
\]

\[
|\vec{F}_{\text{friction}}| \le \mu \, |\vec{F}_{\text{normal}}|
\]

\[
\vec{p} = m\vec{v}
\]

\[
\Delta \vec{p} = \vec{F}\Delta t
\]

\[
\mathrm{KE} = \frac{1}{2}mv^2
\]

\[
\Delta \mathrm{PE} = mg\Delta y
\]

\[
\Delta E = W = Fd\cos\theta
\]

\[
a_c = \frac{v^2}{r}
\]

\[
\theta = \theta_0 + \omega_0 t + \frac{1}{2}\alpha t^2
\]

\[
\omega = \omega_0 + \alpha t
\]

\[
T = \frac{2\pi}{\omega} = \frac{1}{f}
\]

\[
v = f\lambda
\]

\[
x = A\cos(2\pi ft)
\]

\[
\vec{\alpha} = \frac{\sum \vec{\tau}}{I} = \frac{\vec{\tau}_{\text{net}}}{I}
\]

\[
\vec{\tau} = r \times F
\]

\[
L = I\omega
\]

\[
\Delta L = \tau \Delta t
\]

\[
|\vec{F}_{\text{spring}}| = k|\vec{x}|
\]

\[
\mathrm{PE}_{\text{spring}} = \frac{1}{2}kx^2
\]

\[
T_{\text{spring}} = 2\pi \sqrt{\frac{m}{k}}
\]

\[
T_{\text{pendulum}} = 2\pi \sqrt{\frac{\ell}{g}}
\]

\[
|\vec{F}_{\text{gravity}}| = G\frac{m_1m_2}{r^2}
\]

\[
|\vec{F}_{\text{gravity}}| = m\vec{g}
\]

\[
\mathrm{PE}_{\text{gravity}} = -G\frac{m_1m_2}{r}
\]

\[
\rho = \frac{m}{V}
\]

\[
\mathrm{KE} = \frac{1}{2}I\omega^2
\]

\section*{Electricity}

\[
|\vec{F}_e| = k\left|\frac{q_1q_2}{r^2}\right|
\]

\[
I = \frac{\Delta q}{\Delta t}
\]

\[
R_{\text{series}} = R_1 + R_2 + \ldots + R_n
\]

\[
\Delta V = IR
\]

\[
P = I\Delta V
\]

\[
\frac{1}{R_{\text{parallel}}} = \frac{1}{R_1} + \frac{1}{R_2} + \ldots + \frac{1}{R_n}
\]

\[
R = \rho \frac{\ell}{A}
\]

\]

\[

\]

\[
\text{Rectangular Solid} \qquad V = \ell wh
\]

\[
\text{Cylinder} \qquad V = \pi r^2\ell
\]

\[
S = 2\pi r\ell + 2\pi r^2
\]

\[
\text{Triangle} \qquad A = \frac{1}{2}bh
\]

\[
\text{Sphere} \qquad V = \frac{4}{3}\pi r^3
\]

\[
S = 4\pi r^2
\]

\]

\[

\qquad

\qquad

\]
