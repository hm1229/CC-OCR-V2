\begin{enumerate}

\item[c)] $\left[-1, \dfrac{1}{3}\right]$
\item[d)] $(-\infty, -1] \cup \left[\dfrac{1}{3}, \infty\right)$
\end{enumerate}

\item The function $f : [0, 2\pi] \to [-1, 1]$ defined by $f(x) = \sin x$ is

\begin{enumerate}
\item[a)] one-to-one \hspace{0.5cm} b) onto \hspace{0.5cm} c) bijection \hspace{0.5cm} d) cannot be defined
\end{enumerate}

\item The solution set of the following inequality $|x - 1| \geq |x - 3|$ is

\begin{enumerate}
\item[a)] $[0, 2]$ \hspace{1cm} b) $[2, \infty)$ \hspace{1cm} c) $(0, 2)$ \hspace{1cm} d) $(-\infty, 2)$
\end{enumerate}

\item If 3 is the logarithm of 343, then the base is

\begin{enumerate}
\item[a)] 5 \hspace{1.5cm} b) 7 \hspace{1.5cm} c) 6 \hspace{1.5cm} d) 9
\end{enumerate}

\item In a $\triangle ABC$, $\tan\left(\dfrac{A}{2}\right) =$

\begin{enumerate}
\item[a)] $\sqrt{\dfrac{(s-b)(s-c)}{bc}}$
\item[b)] $\sqrt{\dfrac{s(s-a)}{bc}}$
\item[c)] $\sqrt{\dfrac{(s-b)(s-c)}{s(s-a)}}$
\item[d)] $\sqrt{s(s-a)(s-b)(s-c)}$
\end{enumerate}

\item There are 10 points in a plane and 4 of them are collinear. The number of straight lines joining any two points is

\begin{enumerate}
\item[a)] 45 \hspace{1.5cm} b) 40 \hspace{1.5cm} c) 39 \hspace{1.5cm} d) 38
\end{enumerate}

\item The $n^{\text{th}}$ term of the sequence 1, 2, 4, 7, 11, \ldots\ldots is

\begin{enumerate}
\item[a)] $n^3 + 3n^2 + 2n$
\item[b)] $n^3 - 3n^2 + 3n$
\item[c)] $\dfrac{n(n+1)(n+2)}{3}$
\item[d)] $\dfrac{n^2 - n + 2}{2}$
\end{enumerate}

\item The slopes of the line which makes an angle $45^\circ$ with the line $3x - y = -5$ are

\begin{enumerate}
\item[a)] $1, -1$
\item[b)] $\dfrac{1}{2}, -2$
\item[c)] $1, \dfrac{1}{2}$
