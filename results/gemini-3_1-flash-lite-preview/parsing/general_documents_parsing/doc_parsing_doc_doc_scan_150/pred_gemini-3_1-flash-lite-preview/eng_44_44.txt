\section*{Secondary 3 Mathematics Set 1}

\subsection*{PART 1: Multiple-choice questions}

\begin{enumerate}
    \item \textbf{(4)} \\
    $2x + 3y = 7$ \\
    $2x = 7 - 3y$ \\
    $x = \frac{7}{2} - \frac{3}{2}y$ \\
    $x = -\frac{3}{2}y + \frac{7}{2}$

    \item \textbf{(2)} \\
    $3x + 4y = 15$ \\
    $6x + 8y = 2(3x + 4y)$ \\
    $= 30$ \\
    $15x + 20y = 5(3x + 4y)$ \\
    $= 75$

    \item \textbf{(3)} \\
    $4x \geqslant \frac{1}{2}$ \\
    $\frac{4x}{4} \geqslant \frac{(\frac{1}{2})}{4}$ \\
    $x \geqslant \frac{1}{8}$

    \item \textbf{(3)} \\
    $6 - 8x < -2x$ \\
    $6 - 8x + 8x < -2x + 8x$ \\
    $6 < 6x$ \\
    $6x > 6$ \\
    $x > 1$

    \item \textbf{(2)} \\
    For rectangle $P$, breadth $= \frac{120}{15}$ \\
    $= 8 \text{ cm}$ \\
    Perimeter of $Q = \text{Perimeter of } P$ \\
    $= 2(8 + 15)$ \\
    $= 46 \text{ cm}$

    \item \textbf{(4)} \\
    Circumference of $Y = \text{Circumference of } X$ \\
    $= 10\pi$ \\
    $2\pi r = 10\pi$ \\
    $r = 5 \text{ cm}$ \\
    Area of circle $Y = \pi(5^2)$ \\
    $= 25\pi \text{ cm}^2$

    \item \textbf{(3)} \\
    ratio of sides $1 : 10$ \\
    ratio of area $1 : 100$

    \item \textbf{(3)} \\
    ratio of sides $1 : 5$ \\
    ratio of perimeter $1 : 5$

    \item \textbf{(3)} \\
    Let $H$ and $h$ be the heights of the 2 candles. \\
    Burnt amount at any time: \\
    $H: \frac{120}{180} \times H = \frac{2}{3}H$ \\
    $h: \frac{120}{300} \times h = \frac{2}{5}h$ \\
    Remaining: \\
    $H: H - \frac{2}{3}H = \frac{1}{3}H$ \\
    $h: h - \frac{2}{5}h = \frac{3}{5}h$ \\
    The ratio after $2 \text{ h}$ is $1 : 1$. \\
    $\frac{\frac{1}{3}H}{\frac{3}{5}h} = \frac{1}{1}$ \\
    Cross multiply: \\
    $\frac{1}{3}H = \frac{3}{5}h$ \\
    Cross multiply: \\
    $5H = 9h$ \\
    $\frac{H}{h} = \frac{9}{5}$ \\
    $H : h = 9 : 5$

    \item \textbf{(1)} \\
    Height after $n \text{ min}$: \\
    $1^{\text{st}} : \frac{n}{180}$ \\
    $2^{\text{nd}} : \frac{n}{240}$ \\
    Height remaining: \\
    $1^{\text{st}} : 1 - \frac{n}{180}$ \\
    $2^{\text{nd}} : 1 - \frac{n}{240}$ \\
    The ratio became $1 : 2$. \\
    $\frac{1 - \frac{n}{180}}{1 - \frac{n}{240}} = \frac{1}{2}$ \\
    Cross multiply: \\
    $2 - \frac{n}{90} = 1 - \frac{n}{240}$ \\
    $\frac{n}{90} - \frac{n}{240} = 2 - 1$ \\
    $\frac{8n - 3n}{720} = 1$ \\
    $5n = 720$ \\
    $n = 144 \text{ min} = 2 \text{ h } 24 \text{ min}$ \\
    Working backwards: $14 \text{ } 36 \text{ h}$

    \item \textbf{(2)} \\
    Town $X \xrightarrow{\hspace{2cm} d \hspace{2cm}} \text{Town } Y$ \\
    When $B$ meets $C$, the distance between $A$ and $C$ is \\
    $(50 + 70) \times 2 = 240 \text{ km}$ \\
    Difference in speed $\rightarrow 60 - 50 = 10 \text{ km/h}$ \\
    $240 \div 10 = 24 \text{ : h}$ \\
    $(60 + 70) \times 24 = 3120 \text{ km}$

    \item \textbf{(4)} \\
    $15 - 5 = 10 \text{ h}$ \\
    Consider station $S$ and Town $Q$ \\
    $P \xrightarrow{05 \text{ } 00 \text{ h}} S \xrightarrow{09 \text{ } 00 \text{ h}} Q \xrightarrow{05 \text{ } 00 \text{ h}}$ \\
    $3 : 2 = 6 : 4$ \\
    $5 + 4 = 9$ \\
    Ans: $09 \text{ } 00 \text{ h}$

    \item \textbf{(1)} \\
    Total number of faces: \\
    Original cuboid $\rightarrow 18 \text{ faces}$ \\
    $4 \text{ cubes} \rightarrow 6 \times 4 = 24 \text{ faces}$ \\
    $24 - 18 = 6 \text{ faces}$ \\
    $6 \text{ faces} \rightarrow 216 \text{ cm}^2$
\end{enumerate}
