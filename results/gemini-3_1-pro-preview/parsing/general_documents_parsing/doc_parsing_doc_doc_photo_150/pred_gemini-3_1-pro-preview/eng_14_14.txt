\subsection*{5) Compound Angles :}
i) $\sin(A+B) = \sin A \cos B + \cos A \sin B$ \\
ii) $\sin(A-B) = \sin A \cos B - \cos A \sin B$ \\
iii) $\cos(A+B) = \cos A \cos B - \sin A \sin B$ \\
iv) $\cos(A-B) = \cos A \cos B + \sin A \sin B$ \\
v) $\tan(A+B) = \frac{\tan A + \tan B}{1 - \tan A \tan B}$ \\
vi) $\tan(A-B) = \frac{\tan A - \tan B}{1 + \tan A \tan B}$

\subsection*{6) Multiple Angle and Sub Multiple Angle :}
$\sin 2A = 2 \sin A \cos A$ \quad i) $\sin A = 2 \sin(\frac{A}{2}) \cos(\frac{A}{2})$ \\
$\cos 2A = \cos^2 A - \sin^2 A$ \quad ii) $\cos A = \cos^2(\frac{A}{2}) - \sin^2(\frac{A}{2})$ \\
\quad \quad \quad $= 1 - 2 \sin^2 A$ \quad iii) $\cos A = 1 - 2 \sin^2(\frac{A}{2})$ \\
\quad \quad \quad $= 2 \cos^2 A - 1$ \quad iv) $\cos A = 2 \cos^2(\frac{A}{2}) - 1$ \\
\quad \quad \quad $= \frac{1 - \tan^2 A}{1 + \tan^2 A}$ \quad v) $\cos A = \frac{1 - \tan^2(\frac{A}{2})}{1 + \tan^2(\frac{A}{2})}$ \\
$\sin 2A = \frac{2 \tan A}{1 + \tan^2 A}$ \quad vi) $\sin A = \frac{2 \tan(\frac{A}{2})}{1 + \tan^2(\frac{A}{2})}$ \\
$\tan 2A = \frac{2 \tan A}{1 - \tan^2 A}$ \quad vii) $\tan A = \frac{2 \tan(\frac{A}{2})}{1 - \tan^2(\frac{A}{2})}$ \\
$\cos^2(\frac{A}{2}) = \frac{1 + \cos A}{2}$ \quad viii) $1 - \cos A = 2 \sin^2(\frac{A}{2})$

\subsection*{7) Triple Angle Formulae :-}
i) $\sin 3\theta = 3 \sin \theta - 4 \sin^3 \theta$ \quad ii) $\cos 3\theta = 4 \cos^3 \theta - 3 \cos \theta$ \\
iii) $\tan 3\theta = \frac{3 \tan \theta - \tan^3 \theta}{1 - 3 \tan^2 \theta}$ \\
iv) $\cos^3 \theta = \frac{\cos 3\theta + 3 \cos \theta}{4}$ \quad v) $\sin^3 \theta = \frac{3 \sin \theta - \sin 3\theta}{4}$

\subsection*{8) Factorization Formulae :-}
i. $\sin C + \sin D = 2 \sin \left(\frac{C+D}{2}\right) \cos \left(\frac{C-D}{2}\right)$ \\
ii. $\sin C - \sin D = 2 \cos \left(\frac{C+D}{2}\right) \sin \left(\frac{C-D}{2}\right)$ \\
iii. $\cos C + \cos D = 2 \cos \left(\frac{C+D}{2}\right) \cos \left(\frac{C-D}{2}\right)$ \\
iv. $\cos C - \cos D = 2 \sin \left(\frac{C+D}{2}\right) \sin \left(\frac{D-C}{2}\right)$ if $C > D$ \\
\quad \quad \quad \quad \quad \quad $= -2 \sin \left(\frac{C+D}{2}\right) \sin \left(\frac{C-D}{2}\right)$ if $D > C$

\subsection*{9) Defactorization formulae :-}
i. $2 \sin A \cos B = \sin(A+B) + \sin(A-B)$ \\
ii. $2 \cos A \sin B = \sin(A+B) - \sin(A-B)$ \\
iii. $2 \cos A \cos B = \cos(A+B) + \cos(A-B)$ \\
iv. $2 \sin A \sin B = \cos(A-B) - \cos(A+B)$

\section*{Co - Ordinate - Geometry (2D)}
1. Distance Formula : $\sqrt{(x_2-x_1)^2 + (y_2-y_1)^2}$ \\
2. Section Formula : $P(x,y) = \left( \frac{mx_2+nx_1}{m+n}, \frac{my_2+ny_1}{m+n} \right)$ Internally \\
\quad \quad \quad \quad \quad \quad $Q(x,y) = \left( \frac{mx_2-nx_1}{m-n}, \frac{my_2-ny_1}{m-n} \right)$ Externally \\
3. Mid point Formula : $\left( \frac{x_1+x_2}{2}, \frac{y_1+y_2}{2} \right)$ \\
4. Centroid of triangle : $G \left( \frac{x_1+x_2+x_3}{3}, \frac{y_1+y_2+y_3}{3} \right)$ \\
5. Area of triangle $\Delta = \frac{1}{2} |x_1(y_2-y_3) + x_2(y_3-y_1) + x_3(y_1-y_2)|$ \\
6. Slope of straight line : $m = \tan \theta$ \\
7. One point Form : $y - y_1 = m(x - x_1)$ \\
8. Slope Intercept from : $y = mx + C$ \\
9. Two point Form : $\frac{y-y_1}{y_2-y_1} = \frac{x-x_1}{x_2-x_1}$ \\
10. Intercept Form : $\frac{x}{a} + \frac{y}{b} = 1$ \\
11. Normal Form : $x \cos \alpha + y \sin \alpha = p$ \\
12. Slope of line $ax + by + c = 0$ is $-\frac{a}{b}$ \\
13. Intersection of two lines : $L_1 + \lambda L_2 = 0$ \\
14. Angle between two s.t lines : $\tan \theta = \left| \frac{m_1-m_2}{1+m_1m_2} \right|$ \\
15. If the lines are parallel : $m_1 = m_2$ \\
\quad If the lines are $\perp$ : $m_1 m_2 = -1$ \\
16. Length of $\perp$ form : $\frac{|ax_1+by_1+c|}{\sqrt{a^2+b^2}}$ \\
17. Bisector Equations : $\frac{a_1x+b_1y+c_1}{\sqrt{a_1^2+b_1^2}} = \pm \frac{a_2x+b_2y+c_2}{\sqrt{a_2^2+b_2^2}}$ \\
18. Distance bet$^n$ parallel lines : $\left| \frac{c_1-c_2}{\sqrt{a^2+b^2}} \right|$ \\
19. Homogeneous Equation : $ax^2 + 2hxy + by^2 = 0$ \\
\quad \quad \quad \quad \quad \quad $\tan \theta = \left| \frac{2\sqrt{h^2-ab}}{a+b} \right|$ \\
20. General Equation : $ax^2 + 2hxy + by^2 + 2gx + 2fy + c = 0$ represents a pair of lines \\
\quad If $abc + 2fgh - af^2 - bg^2 - ch^2 = 0$ \\
\quad Point of intersection $(x,y) = \left( \frac{hf-bg}{ab-h^2}, \frac{hg-af}{ab-h^2} \right)$
