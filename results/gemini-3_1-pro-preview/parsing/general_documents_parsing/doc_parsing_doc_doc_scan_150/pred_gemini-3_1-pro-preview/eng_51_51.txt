\section*{formulas for discrete-time LTI signals and systems}

\begin{tabular}{ll}
\hline
name & formula \\
\hline
area under impulse & $\sum_{n} \delta(n) = 1$ \\
multiplication by impulse & $f(n) \delta(n) = f(0) \delta(n)$ \\
... by shifted impulse & $f(n) \delta(n - n_o) = f(n_o) \delta(n - n_o)$ \\
convolution & $f(n) * g(n) = \sum_{k} f(k) g(n - k)$ \\
... with an impulse & $f(n) * \delta(n) = f(n)$ \\
... with a shifted impulse & $f(n) * \delta(n - n_o) = f(n - n_o)$ \\
transfer function & $H(z) = \sum_{n} h(n) z^{-n}$ \\
frequency response & $H^f(\omega) = \sum_{n} h(n) e^{-j \omega n}$ \\
... their connection & $H^f(\omega) = H(e^{j \omega})$ \\
& provided unit circle $\subset$ ROC \\
\hline
\end{tabular}

\section*{formulas for continuous-time LTI signals and systems}

\begin{tabular}{ll}
\hline
name & formula \\
\hline
area under impulse & $\int \delta(t) dt = 1$ \\
multiplication by impulse & $f(t) \delta(t) = f(0) \delta(t)$ \\
... by shifted impulse & $f(t) \delta(t - t_o) = f(t_o) \delta(t - t_o)$ \\
convolution & $f(t) * g(t) = \int f(\tau) g(t - \tau) d\tau$ \\
... with an impulse & $f(t) * \delta(t) = f(t)$ \\
... with a shifted impulse & $f(t) * \delta(t - t_o) = f(t - t_o)$ \\
transfer function & $H(s) = \int h(t) e^{-st} dt$ \\
frequency response & $H^f(\omega) = \int h(t) e^{-j \omega t} dt$ \\
... their connection & $H^f(\omega) = H(j \omega)$ \\
& provided $j\omega$-axis $\subset$ ROC \\
\hline
\end{tabular}

\section*{useful formulas}

\begin{tabular}{ll}

name & formula \\
\hline

... for cosine & $\cos(\theta) = \frac{e^{j \theta} + e^{-j \theta}}{2}$ \\
... for sine & $\sin(\theta) = \frac{e^{j \theta} - e^{-j \theta}}{2j}$ \\
Sinc function & $\text{sinc}(\theta) := \frac{\sin(\pi \theta)}{\pi \theta}$ \\
\hline
\end{tabular}
