\paragraph{Mean}
\[
\bar{x} = \frac{1}{n}\sum_{i=1}^n x_i
\]

\paragraph{Standard Deviation*}
\[
S = \sqrt{\frac{\sum (x_i - \bar{x})^2}{n-1}}
\]

\paragraph{Standard Error of the Mean*}
\[
SE_{\bar{x}} = \frac{s}{\sqrt{n}}
\]

\paragraph{Chi-Square}
\[
\chi^2 = \sum \frac{(o-e)^2}{e}
\]

\subsubsection*{Laws of Probability}
If A and B are mutually exclusive, then:
\[
P(A \text{ or } B) = P(A) + P(B)
\]
If A and B are independent, then:
\[
P(A \text{ and } B) = P(A) \times P(B)
\]

\subsubsection*{Hardy-Weinberg Equations}
\begin{align*}
p^2 + 2pq + q^2 &= 1 \quad &p &= \text{frequency of the dominant allele} \\
& & &\text{in a population} \\
p + q &= 1 \quad &q &= \text{frequency of the recessive allele} \\
& & &\text{in a population}
\end{align*}

\text{Median} &= \text{middle value that separates the greater and lesser halves of a data set} \\
\text{Mean} &= \text{sum of all data points divided by number of data points} \\
\text{Range} &= \text{value obtained by subtracting the smallest observation (sample minimum) from the greatest (sample maximum)}
