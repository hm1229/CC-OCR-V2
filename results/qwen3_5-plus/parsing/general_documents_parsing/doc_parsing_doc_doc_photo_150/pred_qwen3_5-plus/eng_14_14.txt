        \item Area of sector = $\frac{1}{2} r \theta$
    \end{enumerate}

\subsection*{4) Allied - Angles :-}
\begin{multicols}{2}
    \begin{enumerate}
        \item $\text{Sin }(-\theta) = -\text{Sin}$
        \item $\text{Cos }(-\theta) = \text{Cos}$
        \item $\text{tan }(-\theta) = -\text{tan}$
        \item $\text{Cot }(-\theta) = -\text{Cot}$
        \item $\text{Sec }(-\theta) = \text{Sec}$
        \item $\text{Cosec }(-\theta) = -\text{Cosec}$
    \end{enumerate}
    \begin{enumerate}
        \item[2)] $\text{Sin }(90^\circ-\theta) = \text{Cos}$
        \item $\text{Cos }(90^\circ-\theta) = \text{Sine}$
        \item $\text{tan }(90^\circ-\theta) = \text{Cot}$
        \item $\text{Cot }(90^\circ-\theta) = \text{tan}$
        \item $\text{Sec }(90^\circ-\theta) = \text{Cosec}$
        \item $\text{Cosec }(90^\circ-\theta) = \text{sec}$
    \end{enumerate}
\end{multicols}

\subsection*{5) Compound Angles :-}
\begin{enumerate}[i)]
    \item $\text{Sin }(A+B) = \text{Sin A Cos B + Cos A Sin B}$
    \item $\text{Sin }(A-B) = \text{Sin A Cos B - Cos A Sin B}$
    \item $\text{Cos }(A+B) = \text{Cos A Cos B - Sin A Sin B}$
    \item $\text{Cos }(A-B) = \text{Cos A Cos B + Sin A Sin B}$
    \item $\text{tan }(A+B) = \frac{\text{tan A + tan B}}{1 - \text{tan A tan B}}$
    \item $\text{tan }(A-B) = \frac{\text{tan A - tan B}}{1 + \text{tan A tan B}}$
\end{enumerate}

\subsection*{6) Multiple Angle and Sub Multiple Angle :}
\begin{multicols}{2}
    $\text{Sin } 2A = 2 \text{Sin A Cos A}$
    
    $\text{Cos } 2A = \text{Cos}^2 A - \text{Sin}^2 A$
    \begin{enumerate}[i)]
        \item $\text{Sin A} = 2 \text{Sin}\left(\frac{A}{2}\right) \text{Cos}\left(\frac{A}{2}\right)$
        \item $\text{Cos A} = \text{Cos}^2\left(\frac{A}{2}\right) - \text{Sin}^2\left(\frac{A}{2}\right)$
        \item $\text{Cos A} = 1 - 2\text{Sin}^2\left(\frac{A}{2}\right)$
        \item $\text{Cos A} = 2\text{Cos}^2\left(\frac{A}{2}\right) - 1$
        \item $\text{Cos A} = \frac{1 - \text{tan}^2\left(\frac{A}{2}\right)}{1 + \text{tan}^2\left(\frac{A}{2}\right)}$
        \item $\text{Sin A} = \frac{2 \text{tan}\left(\frac{A}{2}\right)}{1 + \text{tan}^2\left(\frac{A}{2}\right)}$
        \item $\text{tan A} = \frac{2 \text{tan}\left(\frac{A}{2}\right)}{1 - \text{tan}^2\left(\frac{A}{2}\right)}$
        \item $1 + \text{Cos A} = 2\text{Sin}^2\left(\frac{A}{2}\right)$
    \end{enumerate}
\end{multicols}

\columnbreak

\subsection*{7) Triple Angle Formulae :-}
\begin{enumerate}[i)]
    \item $\text{Sin } 3\theta = 3 \text{ sin } \theta - 4 \text{ sin}^3 \theta$
    \item $\text{cos } 3\theta = 4 \text{ cos}^3 \theta - 3 \text{ cos } \theta$
    \item $\text{tan } 3\theta = \frac{3 \text{tan } \theta - \text{tan}^3 \theta}{1 - 3 \text{tan}^2 \theta}$
    \item $\text{cos}^3 \theta = \frac{\text{Cos } 3\theta + 3 \text{Cos } \theta}{4}$
    \item $\text{sin}^3 \theta = \frac{3 \text{sin } \theta - \text{sin } 3\theta}{4}$
\end{enumerate}

\subsection*{8) Factorization Formulae :-}
\begin{enumerate}[i)]
    \item $\text{sin C + sin D} = 2 \text{ sin}\left(\frac{C+D}{2}\right) \text{cos}\left(\frac{C-D}{2}\right)$
    \item $\text{sin C - sin D} = 2 \text{ sin}\left(\frac{C-D}{2}\right) \text{sin}\left(\frac{C-D}{2}\right)$
    \item $\text{cos C + cos D} = 2 \text{ cos}\left(\frac{C+D}{2}\right) \text{cos}\left(\frac{C-D}{2}\right)$
    \item $\text{cos C - cos D} = 2 \text{ sin}\left(\frac{C+D}{2}\right) \text{sin}\left(\frac{C-D}{2}\right)$ if C>D
    \item[] $= 2 \text{ sin}\left(\frac{C+D}{2}\right) \text{sin}\left(\frac{D-C}{2}\right)$ if D>C
\end{enumerate}

\subsection*{9) Defactorization formulae :-}
\begin{enumerate}[i)]
    \item $2 \text{ sin A cos B} = \text{sin (A+B) + sin (A - B)}$
    \item $2 \text{ cos A sin B} = \text{sin (A+B) - sin (A - B)}$
    \item $2 \text{ cos A cos B} = \text{cos (A+B) + cos (A - B)}$
    \item $2 \text{ sin A sin B} = \text{cos (A - B) - cos (A + B)}$
\end{enumerate}

\section*{Co - Ordinate - Geometry (2D)}
\begin{enumerate}
    \item \textbf{Distance Formula :} $\sqrt{(x_2-x_1)^2 + (y_2-y_1)^2}$
    \item \textbf{Section Formula :} P $(x, y) = \left( \frac{mx_2+nx_1}{m+n}, \frac{my_2+ny_1}{m+n} \right)$ Internally \\
    Q $(x, y) = \left( \frac{mx_2-nx_1}{m-n}, \frac{my_2-ny_1}{m-n} \right)$ Externally
    \item \textbf{Mid point Formula :} $\left( \frac{x_1+x_2}{2}, \frac{y_1+y_2}{2} \right)$
    \item \textbf{Centroid of triangle :} G $\left( \frac{x_1+x_2+x_3}{3}, \frac{y_1+y_2+y_3}{2} \right)$
    \item \textbf{Area of triangle} $\Delta = \frac{1}{2} |x_1(y_2-y_3) + x_2(y_3-y_1) + x_3(y_1-y_2)|$
    \item \textbf{Slope of straight line :} m = tan $\theta$
    \item \textbf{One point Form :} $y-y_1 = m(x-x_1)$
    \item \textbf{Slope Intercept from :} $y = mx + C$
    \item \textbf{Two point Form :} $\frac{y-y_1}{y_2-y_1} = \frac{x-x_1}{x_2-x_1}$
    \item \textbf{Intercept Form :} $\frac{x}{a} + \frac{y}{b} = 1$
    \item \textbf{Normal Form :} $x \text{Cos } \alpha + y \text{Sin } \alpha = p$
    \item \textbf{Slope of line} $ax + by + c = 0$ is $\frac{-a}{b}$
    \item \textbf{Intersection to two lines :} $L_1 + \lambda L_2 = 0$
    \item \textbf{Angle between two s.t lines :} $\text{tan } \theta = \left| \frac{m_1 - m_2}{1 + m_1 m_2} \right|$
    \item If the lines are parallel : $m_1 = m_2$ \\
    If the lines are $L' : m_1 m_2 = -1$
    \item \textbf{Length of $\perp$ form :} $\left| \frac{ax_1+by_1+c}{\sqrt{a^2+b^2}} \right|$
    \item \textbf{Bisector Equations :} $\frac{ax+by+c}{\sqrt{a^2+b^2}} = \pm \frac{a_1x+b_1y+c_1}{\sqrt{a_1^2+b_1^2}}$
    \item \textbf{Distance bet" parallel lines :} $\left| \frac{c_1 - c_2}{\sqrt{a^2+b^2}} \right|$
    \item \textbf{Homogeneous Equation :} $ax^2 + 2hxy + by^2 = 0$ \\
    $\text{tan } \theta = \frac{2\sqrt{h^2-ab}}{a+b}$
    \item \textbf{General Equation :} $ax^2 + 2hxy + by^2 + 2gx + 2fy + c = 0$ represents a pair of lines \\
    If $abc + 2fgh - af^2 - bg^2 - ch^2 = 0$ \\
    Point of intersection $(x,y) = \left( \frac{hf-bg}{ab-h^2}, \frac{hg-af}{ab-h^2} \right)$
\end{enumerate}
