\section*{III. CALCULATIONS:}

\subsection*{Calculating the volume of the given examined bodies based on the formula:}

$$V_{b} = a \cdot b \cdot c$$

$V_{b}$ - volume of examined bodies $[cm^{3}]$, $a$ - width of examined bodies $[cm]$, $b$ - length of examined bodies $[cm]$, $c$ - height of examined bodies $[cm]$.

$$V_{b} = 2,5 \, [cm] \cdot 4,2 \, [cm] \cdot 1,5 \, [cm] = 15,75 \, [cm^{3}]$$

\subsection*{Calculating the material density of the given examined bodies based on the formula:}

$$\varrho_{b} = \frac{m_{b}}{V_{b}}$$

$\varrho_{b}$ - material density of given body $[g/cm^{3}]$, $m_{b}$ - mass of given body $[g]$, $V_{b}$ - volume of examined body.

\begin{align*}
\varrho_{b} \text{ (ebonite)} &= \frac{22,7}{15,75} \, [g/cm^{3}] = 1,44121 \, [g/cm^{3}] = 1441,21 \, [kg/m^{3}] \\
\varrho_{b} \text{ (bakelite)} &= \frac{19,8}{15,75} \, [g/cm^{3}] = 1,25714 \, [g/cm^{3}] = 1257,14 \, [kg/m^{3}] \\
\varrho_{b} \text{ (aluminium (10))} &= \frac{42,6}{15,75} \, [g/cm^{3}] = 2,70476 \, [g/cm^{3}] = 2704,76 \, [kg/m^{3}]
\end{align*}

\subsection*{Calculating the material density using aerometer based on the formula:}

$$\varrho_{b}' = \frac{m_{1} - m_{2}}{m_{3} - m_{2}} \cdot \varrho_{w}$$

$\varrho_{b}'$ - material density of given body using aerometer $[g/cm^{3}]$, $m_{1}$ - values of mass from experiment $[g]$, $\varrho_{w}$ - density of distilled water $= 1 \, [g/cm^{3}]$.

\begin{align*}
\varrho_{b}' \text{ (ebonite)} &= \frac{10,8 - 8,7}{10,3 - 8,7} \, [g/g] \cdot 1 \, [g/cm^{3}] = \frac{23}{16} \, [g/cm^{3}] = 1,4375 \, [g/cm^{3}] = 1437,5 \, [kg/m^{3}] \\
\varrho_{b}' \text{ (bakelite)} &= \frac{10,5 - 9,0}{10,5 - 9,0} \, [g/g] \cdot 1 \, [g/cm^{3}] = \frac{20}{15} \, [g/cm^{3}] = 1,33333 \, [g/cm^{3}] = 1333,33 \, [kg/m^{3}] \\
\varrho_{b}' \text{ (aluminium (10))} &= \frac{11,0 - 6,7}{8,3 - 6,7} \, [g/g] \cdot 1 \, [g/cm^{3}] = \frac{43}{16} \, [g/cm^{3}] = 2,6875 \, [g/cm^{3}] = 2687,5 \, [kg/m^{3}] \\
\varrho_{b}' \text{ (aluminium (11))} &= \frac{10 - 8,6}{74 - 8,6} \, [g/g] \cdot 1 \, [g/cm^{3}] = \frac{24}{8} \, [g/cm^{3}] = 3 \, [g/cm^{3}] = 3000 \, [kg/m^{3}]
\end{align*}

\subsection*{Comparing calculated density values with table density values:}

\begin{align*}
\varrho_{b} \text{ (ebonite)} - \varrho_{tab} \text{ (ebonite)} &= 1441,21 \, [kg/m^{3}] - 1150 \, [kg/m^{3}] = 291,21 \, [kg/m^{3}] = 22,48 \, [\%] \\
\varrho_{b}' \text{ (ebonite)} - \varrho_{tab} \text{ (ebonite)} &= 1437,5 \, [kg/m^{3}] - 1150 \, [kg/m^{3}] = 287,5 \, [kg/m^{3}] = 22,22 \, [\%] \\
\varrho_{b} \text{ (bakelite)} - \varrho_{tab} \text{ (bakelite)} &= 1257,14 \, [kg/m^{3}] - 1300 \, [kg/m^{3}] = 42,86 \, [kg/m^{3}] = 3,35 \, [\%] \\
\varrho_{b}' \text{ (bakelite)} - \varrho_{tab} \text{ (bakelite)} &= 1333,33 \, [kg/m^{3}] - 1300 \, [kg/m^{3}] = 33,33 \, [kg/m^{3}] = 2,53 \, [\%] \\
\varrho_{b} \text{ (aluminium (10))} - \varrho_{tab} \text{ (aluminium)} &= 2704,76 \, [kg/m^{3}] - 2700 \, [kg/m^{3}] = 4,76 \, [kg/m^{3}] = 0,17 \, [\%] \\
\varrho_{b}' \text{ (aluminium (10))} - \varrho_{tab} \text{ (aluminium)} &= 2687,5 \, [kg/m^{3}] - 2700 \, [kg/m^{3}] = 12,5 \, [kg/m^{3}] = 0,46 \, [\%] \\
\varrho_{b}' \text{ (aluminium (11))} - \varrho_{tab} \text{ (aluminium)} &= 3000 \, [kg/m^{3}] - 2700 \, [kg/m^{3}] = 300 \, [kg/m^{3}] = 10,53 \, [\%]
\end{align*}
