\underline{\textbf{TRIGONOMETRY}}

\begin{itemize}
\item $\sin(A+B) = \sin A \cdot \cos B + \cos A \cdot \sin B$
\item $\sin(A-B) = \sin A \cdot \cos B - \cos A \cdot \sin B$
\item $\cos(A+B) = \cos A \cdot \cos B - \sin A \cdot \sin B$
\item $\cos(A-B) = \cos A \cdot \cos B + \sin A \cdot \sin B$
\item $\tan(A+B) = \dfrac{\tan A + \tan B}{1 - \tan A \tan B}$
\item $\tan(A-B) = \dfrac{\tan A - \tan B}{1 + \tan A \cdot \tan B}$
\item $\cot(A+B) = \dfrac{\cot B \cdot \cot A - 1}{\cot B + \cot A}$
\item $\cot(A-B) = \dfrac{\cot B \cdot \cot A + 1}{\cot B - \cot A}$
\end{itemize}

\begin{itemize}
\item $\sin C + \sin D = 2 \cdot \sin\left(\dfrac{C+D}{2}\right) \cdot \cos\left(\dfrac{C-D}{2}\right)$
\item $\sin C - \sin D = 2 \cos\left(\dfrac{C+D}{2}\right) \cdot \sin\left(\dfrac{C-D}{2}\right)$
\item $\cos C + \cos D = 2 \cos\left(\dfrac{C+D}{2}\right) \cdot \cos\left(\dfrac{C-D}{2}\right)$
\item $\cos C - \cos D = -2 \sin\left(\dfrac{C+D}{2}\right) \cdot \sin\left(\dfrac{C-D}{2}\right)$\\
$= 2 \sin\left(\dfrac{C+D}{2}\right) \cdot \sin\left(\dfrac{D-C}{2}\right)$
\end{itemize}

\begin{itemize}
\item $\sin 2x = 2\sin x \cdot \cos x = \dfrac{2\tan x}{1+\tan^2 x}$
\item $\cos 2x = \cos^2 x - \sin^2 x = 2\cos^2 x - 1$\\
$= 1 - 2\sin^2 x = \dfrac{1-\tan^2 x}{1+\tan^2 x}$
\item $\tan 2x = \dfrac{2\tan x}{1-\tan^2 x}$
\item $\sin 3x = 3\sin x - 4\sin^3 x$
\item $\cos 3x = 4\cos^3 x - 3\cos x$
\item $\tan 3x = \dfrac{3\tan x - \tan^3 x}{1-3\tan^2 x}$
\item $a\sin x + b\cos x \xrightarrow{\hspace{3cm}}$
\end{itemize}

\underline{*General Solution of trigonometry}

\begin{itemize}
\item $\sin x = 0 \Rightarrow x = n\pi$
\item $\cos x = 0 \Rightarrow x = (2n+1)\dfrac{\pi}{2}$
\item $\tan x = 0 \Rightarrow x = n\pi$
\item $\sin x = \sin y \Rightarrow x = n\pi + (-1)^n \cdot y$
\item $\cos x = \cos y \Rightarrow x = 2n\pi \pm y$
\item $\tan x = \tan y \Rightarrow x = n\pi + y$
\end{itemize}

\[
\left.
\begin{array}{l}
\sin^2 x = \sin^2 y \\
\cos^2 x = \cos^2 y \\
\tan^2 x = \tan^2 y
\end{array}
\right\}
\Rightarrow x = n\pi \pm y
\]

\[
\boxed{n \in \mathbb{Z}}
