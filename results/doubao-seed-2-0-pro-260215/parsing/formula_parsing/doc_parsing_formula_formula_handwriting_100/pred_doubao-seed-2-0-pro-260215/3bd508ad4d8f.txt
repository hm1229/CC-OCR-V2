\[
\begin{aligned}
\int_{\mathbb{R}^2} \Lambda^{\frac{3}{2}} (u \cdot \nabla \theta) \Lambda^{\frac{3}{2}} \theta \, dx &= \int_{\mathbb{R}^2} \Lambda^{\frac{1}{2}} (u_i \partial_i \theta) \Lambda^{\frac{3}{2}} \theta \, dx \\
&= \int_{\mathbb{R}^2} \Lambda (u_i \partial_i \theta) \Lambda^2 \theta \, dx \\
&= \int_{\mathbb{R}^2} [\Lambda, u_i] \partial_i \theta \Lambda^2 \theta \, dx + \int_{\mathbb{R}^2} u_i \Lambda \partial_i \theta \Lambda^2 \theta \, dx \\
&= \int_{\mathbb{R}^2} [\Lambda, u_i] \partial_i \theta \Lambda^2 \theta \, dx + \int_{\mathbb{R}^2} \Lambda^{\frac{1}{2}} (u_i \Lambda \partial_i \theta) \Lambda^{\frac{3}{2}} \theta \, dx \\
&\le \int_{\mathbb{R}^2} [\Lambda, u_i] \partial_i \theta \Lambda^2 \theta \, dx + \int_{\mathbb{R}^2} [\Lambda^{\frac{1}{2}}, u_i] \Lambda \partial_i \theta \Lambda^{\frac{3}{2}} \theta \, dx + \int_{\mathbb{R}^2} u_i \Lambda^{\frac{3}{2}} \partial_i \theta \Lambda^{\frac{3}{2}} \theta \, dx \\
&= \int_{\mathbb{R}^2} [\Lambda, u_i] \partial_i \theta \Lambda^2 \theta \, dx + \int_{\mathbb{R}^2} [\Lambda^{\frac{1}{2}}, u_i] \Lambda \partial_i \theta \Lambda^{\frac{3}{2}} \theta \, dx + \int_{\mathbb{R}^2} u_i \partial_i \Lambda^{\frac{3}{2}} \theta \Lambda^{\frac{3}{2}} \theta \, dx \\
&= \int_{\mathbb{R}^2} [\Lambda, u_i] \partial_i \theta \Lambda^2 \theta \, dx + \int_{\mathbb{R}^2} [\Lambda^{\frac{1}{2}}, u_i] \Lambda \partial_i \theta \Lambda^{\frac{3}{2}} \theta \, dx \\
&\triangleq \tilde{\Gamma}_1 + \tilde{\Gamma}_2
\end{aligned}
\]
注：推导中最后一项积分在不可压条件$\nabla\cdot u=0$、全空间边界积分为0的假设下取值为0，故消去。
