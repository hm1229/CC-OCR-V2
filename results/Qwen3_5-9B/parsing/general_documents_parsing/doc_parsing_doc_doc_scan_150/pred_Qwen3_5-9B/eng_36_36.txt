\begin{equation*}
\sin C = \frac{\text{side opposite to angle } C}{\text{hypotenuse}} = \frac{p}{h}
\end{equation*}

\begin{equation*}
\cos C = \frac{\text{side adjacent to angle } C}{\text{hypotenuse}} = \frac{b}{h}
\end{equation*}

\begin{equation*}
\tan C = \frac{\text{side opposite to angle } C}{\text{side adjacent to angle } C} = \frac{p}{b}
\end{equation*}

\begin{equation*}
\cosec \theta = \frac{1}{\sin \theta}; \sec \theta = \frac{1}{\cos \theta}; \cot \theta = \frac{1}{\tan \theta}; \tan \theta = \frac{\sin \theta}{\cos \theta}
\end{equation*}
