\section*{\centering Trigonometry}

\subsection*{1) Trigonometry Ratios:}

\[
\sin\theta = \frac{\text{Opp}}{\text{Hyp}}, \quad \cos\theta = \frac{\text{Adj}}{\text{Hyp}}
\]
\[
\tan\theta = \frac{\text{Opp}}{\text{Adj}}, \quad \cot\theta = \frac{\text{Adj}}{\text{Opp}}
\]
\[
\sec\theta = \frac{\text{Hyp}}{\text{Adj}}, \quad \csc\theta = \frac{\text{Hyp}}{\text{Opp}}
\]

\subsection*{2) Imp - Results:}

\begin{itemize}
    \item[i)] $\sin\theta \cdot \csc\theta = 1$
    \item[ii)] $\cos\theta \cdot \sec\theta = 1$
    \item[iii)] $\tan\theta \cdot \cot\theta = 1$
    \item[iv)] $\tan\theta = \dfrac{\sin\theta}{\cos\theta}$
    \item[v)] $\cot\theta = \dfrac{\cos\theta}{\sin\theta}$
    \item[vi)] $\sin^2\theta + \cos^2\theta = 1$
    \item[vii)] $\sec^2\theta = 1 + \tan^2\theta$
    \item[viii)] $\csc^2\theta = 1 + \cot^2\theta$
    \item[ix)] $\pi = 180°$
    \item[x)] length of arc $s = r\theta$
    \item[xi)] Area of sector $= \dfrac{1}{2}r^2\theta$
\end{itemize}

\subsection*{4) Allied - Angles:}

\begin{itemize}
    \item[1)] $\sin(-\theta) = -\sin\theta$
    \item[2)] $\sin(90°-\theta) = \cos\theta$
    \item[] $\cos(-\theta) = -\cos\theta$ \quad $\cos(90°-\theta) = \sin\theta$
    \item[] $\tan(-\theta) = -\tan\theta$ \quad $\tan(90°-\theta) = \cot\theta$
    \item[] $\cot(-\theta) = -\cot\theta$ \quad $\cot(90°-\theta) = \tan\theta$
    \item[] $\sec(-\theta) = -\sec\theta$ \quad $\sec(90°-\theta) = \csc\theta$
    \item[] $\csc(-\theta) = -\csc\theta$ \quad $\csc(90°-\theta) = \sec\theta$
\end{itemize}

\subsection*{5) Compound Angles:}

\begin{itemize}
    \item[i)] $\sin(A+B) = \sin A \cos B + \cos A \sin B$
    \item[ii)] $\sin(A-B) = \sin A \cos B - \cos A \sin B$
    \item[iii)] $\cos(A+B) = \cos A \cos B - \sin A \sin B$
    \item[iv)] $\cos(A-B) = \cos A \cos B + \sin A \sin B$
    \item[v)] $\tan(A+B) = \dfrac{\tan A + \tan B}{1 - \tan A \tan B}$
    \item[vi)] $\tan(A-B) = \dfrac{\tan A - \tan B}{1 + \tan A \tan B}$
\end{itemize}

\subsection*{5) Multiple Angle and Sub Multiple Angle:}

\[
\sin 2A = 2\sin A \cos A \qquad \text{i)} \quad \sin A = 2\sin\!\left(\frac{A}{2}\right)\cos\!\left(\frac{A}{2}\right)
\]
\[
2A = \cos^2 A - \sin^2 A \qquad \text{ii)} \quad \cos A = \cos^2\!\left(\frac{A}{2}\right) - \sin^2\!\left(\frac{A}{2}\right)
\]
\[
1 \cdot \sin^2 A \qquad \text{iii)} \quad \cos A = 1 - 2\sin^2\!\left(\frac{A}{2}\right)
\]
\[
\text{iv)} \quad \cos A = 2\cos^2\!\left(\frac{A}{2}\right) - 1
\]
\[
\text{v)} \quad \cos A = \frac{1 - \tan^2\!\left(\dfrac{A}{2}\right)}{1 + \tan^2\!\left(\dfrac{A}{2}\right)}
\]
\[
\text{vi)} \quad \sin A = \frac{2\tan\!\left(\dfrac{A}{2}\right)}{1 + \tan^2\!\left(\dfrac{A}{2}\right)}
\]
\[
\text{vii)} \quad \tan A = \frac{2\tan\!\left(\dfrac{A}{2}\right)}{1 - \tan^2\!\left(\dfrac{A}{2}\right)}
\]
\[
\cos^2\!\left(\frac{A}{2}\right) \qquad \text{viii)} \quad 1 - \cos A = 2\sin^2\!\left(\frac{A}{2}\right)
\]

\section*{7) Triple Angle Formulae:}

\begin{itemize}
    \item[i)] $\sin 3\theta = 3\sin\theta - 4\sin^3\theta$
    \item[ii)] $\cos 3\theta = 4\cos^3\theta - 3\cos\theta$
    \item[iii)] $\tan 3\theta = \dfrac{3\tan\theta - \tan^3\theta}{1 - 3\tan^2\theta}$
    \item[iv)] $\cos^n\theta = \dfrac{\cos 3\theta + 3\cos\theta}{4}$, \quad $v)\ \sin^3\theta = \dfrac{3\sin\theta - \sin 3\theta}{4}$
\end{itemize}

\section*{8) Factorization Formulae:}

\begin{itemize}
    \item[i.] $\sin C + \sin D = 2\sin\!\left(\dfrac{C+D}{2}\right)\cos\!\left(\dfrac{C-D}{2}\right)$
    \item[ii.] $\sin C - \sin D = 2\cos\!\left(\dfrac{C+D}{2}\right)\sin\!\left(\dfrac{C-D}{2}\right)$
    \item[iii.] $\cos C + \cos D = 2\cos\!\left(\dfrac{C+D}{2}\right)\cos\!\left(\dfrac{C-D}{2}\right)$
    \item[iv.] $\cos C - \cos D = 2\sin\!\left(\dfrac{C+D}{2}\right)\sin\!\left(\dfrac{C-D}{2}\right)$ if $C > D$
    \[
    = 2\sin\!\left(\dfrac{C+D}{2}\right)\sin\!\left(\dfrac{D-C}{2}\right) \text{ if } D > C
    \]
\end{itemize}

\section*{9) Defactorization Formulae:}

\begin{itemize}
    \item[i.] $2\sin A \cos B = \sin(A+B) + \sin(A-B)$
    \item[ii.] $2\cos A \sin B = \sin(A+B) - \sin(A-B)$
    \item[iii.] $2\cos A \cos B = \cos(A+B) + \cos(A-B)$
    \item[iv.] $2\sin A \sin B = \cos(A-B) - \cos(A+B)$
\end{itemize}

\section*{Co-Ordinate Geometry (2D)}

\begin{enumerate}
    \item Distance Formula: $\sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}$

    \item Section Formula:
    \[
    P(x,y) = \left(\frac{mx_2 + nx_1}{m+n},\ \frac{my_2 + ny_1}{m+n}\right) \quad \text{Internally}
    \]
    \[
    Q(x,y) = \left(\frac{mx_2 - nx_1}{m-n},\ \frac{my_2 - ny_1}{m-n}\right) \quad \text{Externally}
    \]

    \item Mid point Formula: $\left(\dfrac{x_1+x_2}{2},\ \dfrac{y_1+y_2}{2}\right)$

    \item Centroid of triangle: $G\!\left(\dfrac{x_1+x_2+x_3}{3},\ \dfrac{y_1+y_2+y_3}{3}\right)$

    \item Area of triangle:
    \[
    \Delta = \frac{1}{2}\left|x_1(y_2 - y_3) + x_2(y_3 - y_1) + x_3(y_1 - y_2)\right|
    \]

    \item Slope of straight line: $m = \tan\theta$

    \item One point Form: $y - y_1 = m(x - x_1)$

    \item Slope Intercept from: $y = mx + C$

    \item Two point Form: $\dfrac{y-y_1}{x-x_1} = \dfrac{y_2-y_1}{x_2-x_1}$

    \item Intercept Form: $\dfrac{x}{a} + \dfrac{y}{b} = 1$

    \item Normal Form: $x\cos\alpha + y\sin\alpha = p$

    \item Slope of line $ax + by + c = 0$ is $-\dfrac{a}{b}$

    \item Intersection of two lines: $L_1 + \lambda L_2 = 0$

    \item Angle between two straight lines: $\tan\theta = \left|\dfrac{m_1 - m_2}{1 + m_1 m_2}\right|$

    \item If the lines are parallel: $m_1 = m_2$ \\
          If the lines are $\perp$: $m_1 \cdot m_2 = -1$

    \item Length of $\perp$ form: $\left|\dfrac{ax_1 + by_1 + c}{\sqrt{a^2 + b^2}}\right|$

    \item Bisector Equations:
    \[
    \frac{ax_1 + by_1 + c}{\sqrt{a_1^2 + b_1^2}} = \pm \frac{a_2x + b_2y + c_2}{\sqrt{a_2^2 + b_2^2}}
    \]

    \item Distance between parallel lines:
    \[
    \left|\frac{c_1 - c_2}{\sqrt{a^2 + b^2}}\right|
    \]

    \item Homogeneous Equation: $ax^2 + 2hxy + by^2 = 0$
    \[
    \tan\theta = \left|\frac{2\sqrt{h^2 - ab}}{a + b}\right|
    \]

    \item General Equation: $ax^2 + 2hxy + by^2 + 2gx + 2fy + c = 0$ represents a pair of lines

    If $abc + 2fgh - af^2 - bg^2 - ch^2 = 0$

    Point of intersection $(x,y) = \left(\dfrac{hf - bg}{ab - h^2},\ \dfrac{gh - af}{ab - h^2}\right)$
\end{enumerate}
