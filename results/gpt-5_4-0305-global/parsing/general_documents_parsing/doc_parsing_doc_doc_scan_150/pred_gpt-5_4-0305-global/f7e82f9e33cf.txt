\textbf{formulas for discrete-time LTI signals and systems}

\begin{tabular}{ll}
\textbf{name} & \textbf{formula} \\
\hline
area under impulse & $\sum_n \delta(n) = 1$ \\[4pt]
multiplication by impulse & $f(n)\,\delta(n) = f(0)\,\delta(n)$ \\[4pt]
\ldots by shifted impulse & $f(n)\,\delta(n-n_0) = f(n_0)\,\delta(n-n_0)$ \\[4pt]
convolution & $f(n) * g(n) = \sum_k f(k)\,g(n-k)$ \\[4pt]
\ldots with an impulse & $f(n) * \delta(n) = f(n)$ \\[4pt]
\ldots with a shifted impulse & $f(n) * \delta(n-n_0) = f(n-n_0)$ \\[4pt]
transfer function & $H(z) = \sum_n h(n)\,z^{-n}$ \\[4pt]
frequency response & $H^f(\omega) = \sum_n h(n)\,e^{-j\omega n}$ \\[4pt]
\ldots their connection & $H^f(\omega) = H(e^{j\omega})$ \\
& provided unit circle $\subset$ ROC
\end{tabular}

\textbf{formulas for continuous-time LTI signals and systems}

\begin{tabular}{ll}
\textbf{name} & \textbf{formula} \\
\hline
area under impulse & $\int \delta(t)\,dt = 1$ \\[4pt]
multiplication by impulse & $f(t)\,\delta(t) = f(0)\,\delta(t)$ \\[4pt]
\ldots by shifted impulse & $f(t)\,\delta(t-t_0) = f(t_0)\,\delta(t-t_0)$ \\[4pt]
convolution & $f(t) * g(t) = \int f(\tau)\,g(t-\tau)\,d\tau$ \\[4pt]
\ldots with an impulse & $f(t) * \delta(t) = f(t)$ \\[4pt]
\ldots with a shifted impulse & $f(t) * \delta(t-t_0) = f(t-t_0)$ \\[4pt]
transfer function & $H(s) = \int h(t)\,e^{-st}\,dt$ \\[4pt]
frequency response & $H^f(\omega) = \int h(t)\,e^{-j\omega t}\,dt$ \\[4pt]
\ldots their connection & $H^f(\omega) = H(j\omega)$ \\
& provided $j\omega$-axis $\subset$ ROC
\end{tabular}

\end{tabular}
