\[
\sin C = \frac{\text{side opposite to angle } C}{\text{hypotenuse}} = \frac{p}{h}
\]

\[
\cos C = \frac{\text{side adjacent to angle } C}{\text{hypotenuse}} = \frac{b}{h}
\]

\[
\tan C = \frac{\text{side opposite to angle } C}{\text{side adjacent to angle } C} = \frac{p}{b}
\]

\[
\operatorname{cosec} \theta = \frac{1}{\sin \theta}; \sec \theta = \frac{1}{\cos \theta}; \cot \theta = \frac{1}{\tan \theta}; \tan \theta = \frac{\sin \theta}{\cos \theta}
\]
