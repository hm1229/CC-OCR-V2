\begin{table}
\caption{Formulas for discrete-time LTI signals and systems}
\begin{tabular}{l l}
\hline
name & formula \\
\hline
area under impulse & $\displaystyle\sum_{n}\delta(n)=1$ \\
multiplication by impulse & $f(n)\,\delta(n)=f(0)\,\delta(n)$ \\
$\ldots$by shifted impulse & $f(n)\,\delta(n-n_{o})=f(n_{o})\,\delta(n-n_{o})$ \\
convolution & $f(n)*g(n)=\displaystyle\sum_{k}f(k)\,g(n-k)$ \\
$\ldots$with an impulse & $f(n)*\delta(n)=f(n)$ \\
$\ldots$with a shifted impulse & $f(n)*\delta(n-n_{o})=f(n-n_{o})$ \\
transfer function & $H(z)=\displaystyle\sum_{n}h(n)\,z^{-n}$ \\
frequency response & $H^{f}(\omega)=\displaystyle\sum_{n}h(n)\,e^{-j\omega n}$ \\
$\ldots$their connection & $H^{f}(\omega)=H(e^{j\omega})$ \\
 & provided unit circle $\subset$ ROC \\
\hline
\end{tabular}
\end{table}

\begin{table}

\begin{tabular}{l l}
\hline
name & formula \\
\hline
area under impulse & $\displaystyle\int\delta(t)\,dt=1$ \\
multiplication by impulse & $f(t)\,\delta(t)=f(0)\,\delta(t)$ \\

convolution & $f(t)*g(t)=\displaystyle\int f(\tau)\,g(t-\tau)\,d\tau$ \\
$\ldots$with an impulse & $f(t)*\delta(t)=f(t)$ \\
$\ldots$with a shifted impulse & $f(t)*\delta(t-t_{o})=f(t-t_{o})$ \\
transfer function & $H(s)=\displaystyle\int h(t)\,e^{-s\,t}\,dt$ \\
frequency response & $H^{f}(\omega)=\displaystyle\int h(t)\,e^{-j\omega t}\,dt$ \\
$\ldots$their connection & $H^{f}(\omega)=H(j\omega)$ \\
 & provided $j\omega$-axis $\subset$ ROC \\
\hline
\end{tabular}
