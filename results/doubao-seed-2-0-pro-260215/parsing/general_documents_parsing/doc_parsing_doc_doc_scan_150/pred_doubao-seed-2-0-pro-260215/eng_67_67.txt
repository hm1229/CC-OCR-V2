\textit{(Visuals: Four 3D shapes constructed from 1 cm unit cubes, labeled P, Q, R, S left to right)}

    \item P and Q
    \item P and R
    \item P and S

    \item R and S
\end{enumerate}

    \item P and S only
    \item R and T only
    \item Q, R and T
    \item P, Q and S
    \item Only S
\end{enumerate}
\end{enumerate}
