\vspace{1.5em}

\noindent \textsc{Compound Interest Formula:} \quad $a = p \left( 1 + \frac{r}{12} \right)^{12y}$

\vspace{1.5em}

\noindent \textsc{Equivalent APR Formula:} \quad $\text{APR} = \left( 1 + \frac{r}{12} \right)^{12} - 1$

\vspace{1.5em}

\noindent \textsc{Future Value Annuity Formula:} \quad $a = p * \frac{\left( 1 + \frac{r}{12} \right)^{12y} - 1}{\frac{r}{12}}$

\vspace{1.5em}

\noindent \textsc{Loan Payment Formula:} \quad $p = \frac{a * \frac{r}{12}}{1 - \left( 1 + \frac{r}{12} \right)^{-12y}}$

\vspace{1.5em}

\noindent where
\begin{itemize}
    \item[] $a = \text{account balance or loan amount (\$)}$
    \item[] $p = \text{initial deposit (principal), regular deposit, or regular payment (\$)}$
    \item[] $y = \text{time invested (years)}$
    \item[] $r = \text{interest rate compounded monthly (as a decimal)}$
\end{itemize}
