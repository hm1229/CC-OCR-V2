\Sigma_1 &= \frac{(N-N_{de})!}{N!} \sum_{\alpha_1} \sum_{\alpha_2 \neq \alpha_1} \dots \sum_{\alpha_{N_{de}} \neq \alpha_1, \dots, \alpha_{N_{de}-1}} \sum_{i \in D} \sum_{j \neq i} \Gamma_{ij} \Gamma^{ji} \\
&= \frac{(N-N_{de})!}{N!} \sum_{\alpha_1} \sum_{\alpha_2 \neq \alpha_1} \dots \sum_{\alpha_{N_{de}} \neq \alpha_1, \dots, \alpha_{N_{de}-1}} \left( \sum_{j \neq \alpha_1} \Gamma_{\alpha_1 j} \Gamma^{j \alpha_1} + \dots + \sum_{j \neq \alpha_{N_{de}}} \Gamma_{\alpha_{N_{de}} j} \Gamma^{j \alpha_{N_{de}}} \right) \\
