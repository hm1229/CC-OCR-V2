```latex
\documentclass{article}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{geometry}
\geometry{a4paper, margin=1in}

\begin{document}

\begin{center}
    \textbf{\Large Important Algebraic Formulas}
\end{center}

\vspace{0.5cm}

$(a+b)^2 = a^2 + b^2 + 2ab$

\vspace{0.4cm}

$(a-b)^2 = a^2 + b^2 - 2ab$

\vspace{0.4cm}

$a^2 - b^2 = (a+b)(a-b)$

\vspace{0.4cm}

$a^2 + b^2 = (a+b)^2 - 2ab \qquad or \qquad a^2 + b^2 = (a-b)^2 + 2ab$

\vspace{0.4cm}

$a^3 + b^3 = (a+b)(a^2 - ab + b^2) = (a+b)^3 - 3ab(a+b)$

\vspace{0.4cm}

$a^3 - b^3 = (a-b)(a^2 + ab + b^2) = (a-b)^3 + 3ab(a-b)$

\vspace{0.4cm}

$2(a^2 + b^2) = (a+b)^2 + (a-b)^2$

\vspace{0.4cm}

$(a+b)^2 - (a-b)^2 = 4ab$

\vspace{0.4cm}

$a^4 + b^4 = (a+b)(a-b)[(a+b)^2 - 2ab]$

\vspace{0.4cm}

$(a-b)^2 = (a+b)^2 - 4ab$

\vspace{0.4cm}

$(a+b)^2 = (a-b)^2 + 4ab$

\vspace{0.4cm}

$a^4 + b^4 = [(a+b)^2 - 2ab]^2 - 2(ab)^2$

\vspace{0.4cm}

$(a+b+c)^2 = a^2 + b^2 + c^2 + 2ab + 2bc + 2ca$

\vspace{0.4cm}

$(a+b-c)^2 = a^2 + b^2 + c^2 + 2ab - 2bc - 2ca$

\vspace{0.4cm}

$(a-b-c)^2 = a^2 + b^2 + c^2 - 2ab + 2bc - 2ca$

\vspace{0.4cm}

$a^3 + b^3 + c^3 - 3abc = (a+b+c)(a^2 + b^2 + c^2 - ab - bc - ca)$

\vspace{0.4cm}

$a^4 + a^2b^2 + b^4 = (a^2 + ab + b^2)(a^2 - ab + b^2)$

\vspace{0.4cm}

$a^4 + a^2 + 1 = (a^2 + a + 1)(a^2 - a + 1)$

\vspace{0.4cm}

$if\ a + b + c = 0\ then\ \ a^3 + b^3 + c^3 = 3abc$

\vspace{0.4cm}

$a^8 - b^8 = (a^4 + b^4)(a^2 + b^2)(a+b)(a-b)$

\end{document}
```
