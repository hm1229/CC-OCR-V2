```latex
\documentclass{article}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{geometry}
\geometry{a4paper, margin=1in}

\begin{document}

\begin{center}
    \Large\textbf{Formula Sheet for Physics 251}
\end{center}

\begin{multicols}{2}

\section*{Constants}
$R = 8.3145 \quad \frac{\text{J}}{\text{m}\cdot\text{K}}$ \\
$N_A = 6.02 \times 10^{23}$ \\
$k_B = \frac{R}{N_A} = 1.38 \times 10^{-23} \quad \frac{\text{J}}{\text{m}\cdot\text{K}}$ \\
$e = 1.602 \times 10^{-19} \text{ C}$ \\
$k = \frac{1}{4\pi\epsilon_0} = 9 \times 10^9 \quad \frac{\text{N}\cdot\text{m}^2}{\text{C}^2}$ \\
$\mu_0 = 4\pi \times 10^{-7} \quad \frac{\text{T}\cdot\text{m}}{\text{A}}$ \\
$c = 3 \times 10^8 \quad \frac{\text{m}}{\text{s}}$

\section*{Thermal Expansion}
$\Delta L = \alpha L_0 \Delta T$ \\
$\beta = 3\alpha$

\section*{Heat}
$Q = mc\Delta T$ \\
$Q = \pm mL$ \\
$Q = nC_V \Delta T$ \\
$Q = nC_p \Delta T$

\section*{Heat Current}
$H = k\frac{A}{L}(T_H - T_C)$ \\
$H = Ae\sigma T^4$

\section*{Quantities}

\section*{Work}
$W = \int_{V_1}^{V_2} P dV$

\section*{Isochoric Processes}
$W = 0$ \\
$Q = nC_V \Delta T$ \\
$* \Delta U = nC_V \Delta T$

\section*{Ideal gases}
$* PV = nRT$ \\
$* PV = NkT$ \\
$* v_{\text{XMS}} = \sqrt{\frac{3RT}{M}}$ \\
$* C_V = \frac{3}{2}R \text{ (monatomic)}$ \\
$* C_p = \frac{5}{2}R \text{ (diatomic)}$ \\
$* C_p = C_V + R$

\section*{Isobaric Processes}
$W = P\Delta V$ \\
$Q = nC_p \Delta T$ \\
$* \Delta U = nC_V \Delta T$

\section*{Isothermal Processes}
$* W = nRT \ln\left(\frac{V_2}{V_1}\right)$ \\
$* Q = nRT \ln\left(\frac{V_2}{V_1}\right)$ \\
$* \Delta U = 0$

\section*{Adiabatic Processes}
$* W = -nC_V \Delta T$ \\
$Q = 0$ \\
$* \Delta U = nC_V \Delta T$ \\
$* P_1 V_1^\gamma = P_2 V_2^\gamma$ \\
$* T_1 V_1^{\gamma-1} = T_2 V_2^{\gamma-1}$ \\
$\gamma = \frac{C_p}{C_V}$

\section*{Cyclic Processes}
$W = Q$ \\
$\Delta U = 0$

\section*{Heat Engines}
$W = Q_H + Q_C$ \\
$e = \frac{W}{Q_H}$ \\
$e_r = 1 - \frac{T_C}{T_H}$

\section*{Refrigerators}
$W = Q_H + Q_C$ \\
$K = \frac{|Q_C|}{|W|}$ \\
$K_r = \frac{T_C}{T_H - T_C}$

\section*{Force, Field, Energy and Potential}
$\vec{F} = q\vec{E}$ \\
$\Delta U = q\Delta V$ \\
$V_{ab} = \int_a^b \vec{E} \cdot d\vec{l}$ \\
$\vec{E} = -\nabla V$

\section*{1 Point Charge}
$\vec{E} = k\frac{q}{r^2}\hat{r}$ \\
$V = k\frac{q}{r}$

\section*{2 Point Charges}
$\vec{F} = k\frac{q_1 q_2}{r^2}\hat{r}$ \\
$U = k\frac{q_1 q_2}{r}$

\section*{Distributed Charge}
$\vec{E} = \int k\frac{dq}{r^2}\hat{r}$ \\
$V = \int k\frac{dq}{r}$

\section*{Many Sources}
$\vec{E}_T = \sum \vec{E}_i$ \\
$U_T = \sum_{\text{pairs}} U_{ij}$

\section*{``Elementary'' E Fields}
$\vec{E} = k\frac{Q}{r^2}\hat{r}$ sphere \\
$\vec{E} = \frac{\lambda}{2\pi\epsilon_0 r}\hat{r}$ line \\
$\vec{E} = \frac{\sigma}{2\epsilon_0}\hat{n}$ plane \\
$\vec{E} = \frac{\sigma}{\epsilon_0}\hat{n}$ conducting surface

\section*{Capacitors}
$C = \frac{Q}{V}$ \\
$C_p = \sum C_i$ parallel \\
$\frac{1}{C_T} = \sum \frac{1}{C_i}$ series \\
$U = \frac{1}{2}CV^2 = \frac{Q^2}{2C} = \frac{1}{2}QV$ \\
$^\dagger C = \epsilon_0 \frac{A}{d}$ \\
$|\vec{E}| = \frac{V}{d} = \frac{\sigma}{\epsilon_0}$

\section*{Flux and Gauss's Law}
$\Phi = \iint \vec{E} \cdot d\vec{a}$ \\
$\Phi_{\text{closed}} = \frac{Q_{\text{enclosed}}}{\epsilon_0}$

\section*{Current and Current Density}
$I = \frac{dq}{dt}$ \\
$I = \iint \vec{J} \cdot d\vec{a}$ \\
$\vec{J} = nqv_d$

\end{multicols}

\vspace{1cm}
\noindent * indicates formulas that are specific to ideal gases \\
$\dagger$ indicates formulas that are specific to parallel-plate capacitors

\end{document}
```
