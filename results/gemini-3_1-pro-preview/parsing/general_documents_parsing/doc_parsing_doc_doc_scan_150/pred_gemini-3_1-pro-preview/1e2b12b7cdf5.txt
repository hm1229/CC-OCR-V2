\section*{3 PRESENT CONTINUOUS}

\begin{tabular}{|l|p{10cm}|}
\hline
\textbf{Form} & am, is, or are + verb + ing \\ \hline
\textbf{Use} & actions happening now, at the moment of speaking. \\ \hline
\textbf{Signal words} & now, at the moment, still, Look! Listen!... \\ \hline
\end{tabular}

\subsection*{EXERCISE 1 Find the signal words for present continuous.}
\begin{enumerate}
    \item Which is a signal word for present continuous? \\
    last – now – every night
    \item Which is not a signal word for present continuous? \\
    now – already - still
\end{enumerate}

\section*{4 PAST CONTINUOUS}

\begin{tabular}{|l|p{10cm}|}
\hline
\textbf{Form} & was or were + verb + ing \\ \hline
\textbf{Use} & For actions which were in progress in the past. \\ \hline
\textbf{Signal words} & when, while + long action in the past \\ \hline
\end{tabular}

\subsection*{EXERCISE 1 Find the signal words for past continuous.}
\begin{enumerate}
    \item Which is a signal word for past continuous? \\
    while - by next week - tomorrow
    \item Which is not a signal word for past continuous? \\
    next week – when - while
\end{enumerate}

\section*{5 PRESENT PERFECT}

\begin{tabular}{|l|p{10cm}|}
\hline
\textbf{Form} & have or has + Past Participle \\ \hline
\textbf{Use} & For actions which started in the past and continued up to the present. \\ \hline
\textbf{Signal words} & just, already, yet, for, since, so far, recently, ever... \\ \hline
\end{tabular}

\medskip
\noindent \textbf{NB.} We use Present Perfect Continuous (have/has+ been + verb +ing) to give more emphasis on the duration of the action. However, some verbs can't be used in the continuous form such as own, smell, need, belong, cost, agree, contain..., etc.

\subsection*{EXERCISE 1 Find the signal words for present perfect.}
\begin{enumerate}
    \item Which is a signal word for present perfect? \\
    still – usually – since
    \item Which is not a signal word for present perfect? \\
    since – before – for
\end{enumerate}
