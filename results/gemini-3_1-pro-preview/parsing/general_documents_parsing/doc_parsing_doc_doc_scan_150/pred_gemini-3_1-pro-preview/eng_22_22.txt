```latex
\documentclass{article}

\begin{center}
    \huge \textbf{Important Algebraic Formulas}
\end{center}

\begin{gather*}
(a + b)^2 = a^2 + b^2 + 2ab \\[10pt]
(a - b)^2 = a^2 + b^2 - 2ab \\[10pt]
a^2 - b^2 = (a + b)(a - b) \\[10pt]
a^2 + b^2 = (a + b)^2 - 2ab \qquad \text{or} \qquad a^2 + b^2 = (a - b)^2 + 2ab \\[10pt]
a^3 + b^3 = (a + b)(a^2 - ab + b^2) = (a + b)^3 - 3ab(a + b) \\[10pt]
a^3 - b^3 = (a - b)(a^2 + ab + b^2) = (a - b)^3 + 3ab(a - b) \\[10pt]
2(a^2 + b^2) = (a + b)^2 + (a - b)^2 \\[10pt]
(a + b)^2 - (a - b)^2 = 4ab \\[10pt]
a^4 + b^4 = (a + b)(a - b)[(a + b)^2 - 2ab] \\[10pt]
(a - b)^2 = (a + b)^2 - 4ab \\[10pt]
(a + b)^2 = (a - b)^2 + 4ab \\[10pt]
a^4 + b^4 = [(a + b)^2 - 2ab]^2 - 2(ab)^2 \\[10pt]
(a + b + c)^2 = a^2 + b^2 + c^2 + 2ab + 2bc + 2ca \\[10pt]
(a + b - c)^2 = a^2 + b^2 + c^2 + 2ab - 2bc - 2ca \\[10pt]
(a - b - c)^2 = a^2 + b^2 + c^2 - 2ab + 2bc - 2ca \\[10pt]
a^3 + b^3 + c^3 - 3abc = (a + b + c)(a^2 + b^2 + c^2 - ab - bc - ca) \\[10pt]
a^4 + a^2b^2 + b^4 = (a^2 + ab + b^2)(a^2 - ab + b^2) \\[10pt]
a^4 + a^2 + 1 = (a^2 + a + 1)(a^2 - a + 1) \\[10pt]
\text{if } a + b + c = 0 \text{ then } a^3 + b^3 + c^3 = 3abc \\[10pt]
a^8 - b^8 = (a^4 + b^4)(a^2 + b^2)(a + b)(a - b)
\end{gather*}
```
