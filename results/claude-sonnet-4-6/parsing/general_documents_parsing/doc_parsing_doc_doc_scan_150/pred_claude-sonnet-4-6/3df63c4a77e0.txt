\textbf{Mechanics}
\end{center}

\[
x = x_o + v_{xo}t + \frac{1}{2}a_x t^2 \qquad
a_c = \frac{v^2}{r} \qquad
|\vec{F}_{spring}| = k|\vec{x}|
\]

\[
v = v_o + at \qquad
\theta = \theta_o + \omega_o t + \frac{1}{2}\alpha t^2 \qquad
PE_{spring} = \frac{1}{2}kx^2
\]

\[
v_x^2 - v_{xo}^2 = 2a(x - x_o) \qquad
\omega = \omega_o + \alpha t \qquad
T_{spring} = 2\pi\sqrt{\frac{m}{k}}
\]

\[
\vec{a} = \frac{\sum \vec{F}}{m} = \frac{\vec{F}_{net}}{m} \qquad
T = \frac{2\pi}{\omega} = \frac{1}{f} \qquad
T_{pendulum} = 2\pi\sqrt{\frac{\ell}{g}}
\]

\[
|\vec{F}_{friction}| \leq \mu |\vec{F}_{Normal}| \qquad
v = f\lambda
\]

\[
\vec{p} = m\vec{v} \qquad
x = A\cos(2\pi f t) \qquad
|\vec{F}_{gravity}| = G\frac{m_1 m_2}{r^2}
\]

\[
\Delta \vec{p} = \vec{F}\Delta t \qquad
\vec{\alpha} = \frac{\sum \vec{\tau}}{I} = \frac{\vec{\tau}_{net}}{I} \qquad
|\vec{F}_{gravity}| = m\vec{g}
\]

\[
KE = \frac{1}{2}mv^2 \qquad
\vec{\tau} = r \times F \qquad
PE_{gravity} = -G\frac{m_1 m_2}{r}
\]

\[
\Delta PE = mg\Delta y \qquad
L = I\omega \qquad
\rho = \frac{m}{V}
\]

\[
\Delta E = W = Fd\cos\theta \qquad
\Delta L = \tau \Delta t \qquad
KE = \frac{1}{2}I\omega^2
\]

\begin{center}
\textbf{Electricity}
\end{center}

\[
|\vec{F}_E| = k\left|\frac{q_1 q_2}{r^2}\right| \qquad
\Delta V = IR \qquad
R = \frac{\rho \ell}{A}
\]

\[
I = \frac{\Delta q}{\Delta t} \qquad
P = I\Delta V
\]

\[
R_{series} = R_1 + R_2 + \ldots + R_n \qquad
\frac{1}{R_{Parallel}} = \frac{1}{R_1} + \frac{1}{R_2} + \ldots + \frac{1}{R_n}
\]
