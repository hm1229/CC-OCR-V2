\subsection*{5) Compound Angles:}
\begin{align*}
\text{i) Sin }(A+B) &= \text{Sin A Cos B + Cos A Sin B} \\
\text{ii) Sin }(A-B) &= \text{Sin A Cos B - Cos A Sin B} \\
\text{iii) Cos }(A+B) &= \text{Cos A Cos B - Sin A Sin B} \\
\text{iv) Cos }(A-B) &= \text{Cos A Cos B + Sin A Sin B} \\
\text{v) tan }(A+B) &= \frac{\text{tan A + tan B}}{1-\text{tan A tan B}} \\
\text{vi) tan }(A+B) &= \frac{\text{tan A + tan B}}{1+\text{tan A tan B}}
\end{align*}

\subsection*{6) Multiple Angle and Sub Multiple Angle:}
\begin{align*}
\text{i) Sin }2A &= 2\text{Sin A Cos A} & \text{i) Sin }A &= 2\text{Sin}\left(\frac{A}{2}\right)\text{Cos}\left(\frac{A}{2}\right) \\
\text{ii) Cos }2A &= \text{Cos}^2A - \text{Sin}^2A & \text{ii) Cos }A &= \text{Cos}^2\left(\frac{A}{2}\right) - \text{Sin}^2\left(\frac{A}{2}\right) \\
&= 1 - 2\text{Sin}^2A & \text{iii) Cos }A &= 1 - 2\text{Sin}^2\left(\frac{A}{2}\right) \\
&= 2\text{Cos}^2A - 1 & \text{iv) Cos }A &= 2\text{Cos}^2\left(\frac{A}{2}\right) - 1 \\
& & \text{v) Cos }A &= \frac{1-\text{tan}^2\left(\frac{A}{2}\right)}{1+\text{tan}^2\left(\frac{A}{2}\right)} \\
& & \text{vi) Sin }A &= \frac{2\text{-tan}\left(\frac{A}{2}\right)}{1+\text{tan}^2\left(\frac{A}{2}\right)} \\
& & \text{vii) tan }A &= \frac{2\text{-tan}\left(\frac{A}{2}\right)}{1-\text{tan}^2\left(\frac{A}{2}\right)} \\
& & \text{viii) } 1-\text{Cos }A &= 2\text{Sin}^2\left(\frac{A}{2}\right) \\
& & 1+\text{Cos }A &= 2\text{Cos}^2\left(\frac{A}{2}\right)
\end{align*}

\subsection*{7) Triple Angle Formulae:}
\begin{align*}
\text{i) Sin }3\theta &= 3\text{Sin }\theta - 4\text{ Sin}^3\theta & \text{ii) cos }3\theta &= 4\text{ cos}^3\theta - 3\text{ cos }\theta \\
\text{iii) tan }3\theta &= \frac{3\text{tan }\theta - \text{tan}^3\theta}{1-3\text{tan}^2\theta} & & \\
\text{iv) cos }3\theta &= \frac{\text{cos}^3\theta - 3\text{cos }\theta}{4} & \text{v) sin}^3\theta &= \frac{3\text{sin }\theta - \text{sin }3\theta}{4}
\end{align*}

\subsection*{8) Factorization Formulae:}
\begin{align*}
\text{i. sin C + sin D} &= 2\text{ sin}\left(\frac{C+D}{2}\right)\text{cos}\left(\frac{C-D}{2}\right) \\
\text{ii. sin C - sin D} &= 2\text{ sin}\left(\frac{C+D}{2}\right)\text{sin}\left(\frac{C-D}{2}\right) \\
\text{iii. cos C + cos D} &= 2\text{ cos}\left(\frac{C+D}{2}\right)\text{cos}\left(\frac{C-D}{2}\right) \\
\text{iv. cos C - cos D} &= 2\text{ sin}\left(\frac{C+D}{2}\right)\text{sin}\left(\frac{C-D}{2}\right) \text{ if C>D} \\
&= 2\text{ sin}\left(\frac{C+D}{2}\right)\text{sin}\left(\frac{D-C}{2}\right) \text{ if D>C}
\end{align*}

\subsection*{9) Defactorization formulae:}
\begin{align*}
\text{i. } 2\text{ sin A cos B} &= \text{sin }(A+B) + \text{sin }(A-B) \\
\text{ii. } 2\text{ cos A sin B} &= \text{sin }(A+B) - \text{sin }(A-B) \\
\text{iii. } 2\text{ cos A cos B} &= \text{cos }(A+B) + \text{cos }(A-B) \\
\text{iv. } 2\text{ sin A sin B} &= \text{cos }(A-B) - \text{cos }(A+B)
\end{align*}

\item Slope of straight line: $m = \text{tan }\theta$

\item One point form: $y-y_1 = m(x-x_1)$

\item Slope Intercept from: $y = mx + C$

\item Two point Form: $\frac{y-y_1}{x-x_1} = \frac{y_2-y_1}{x_2-x_1}$

\item Intercept Form: $\frac{x}{a} + \frac{y}{b} = 1$

\item Normal Form: $x\text{ Cos }\theta + y\text{ Sin }\theta = p$

\item Intersection fo two lines: $L_1+\lambda L_2 = 0$

\item If the lines are parallel: $m_1m_2$

If the lines are $\perp$: $m_1m_2 = -1$

\item Length of $\perp$ form: $\left|\frac{ax_1+by_1+c}{\sqrt{a^2+b^2}}\right|$

\item Bisector Equations: $\frac{a_1x+b_1y+c_1}{\sqrt{a_1^2+b_1^2}} = \pm\frac{a_2x+b_2y+c_2}{\sqrt{a_2^2+b_2^2}}$

\item Distance bet$^n$ parallel lines: $\left|\frac{c_1-c_2}{\sqrt{a^2+b^2}}\right|$

\item Homogeneous Equation: $ax^2+2hxy+by^2=0$

$\text{tan }\theta = \frac{2\sqrt{h^2-ab}}{a+b}$
