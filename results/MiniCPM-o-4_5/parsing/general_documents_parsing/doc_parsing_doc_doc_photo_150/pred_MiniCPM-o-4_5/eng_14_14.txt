\section*{Trigonometry}

\subsection*{1) Trigonometry Ratios:}

\begin{tabular}{|c|c|c|c|}
\hline
 & Sine & Cosine & Tangent \\
\hline
Opp & $\sin \theta = \frac{\text{Opp}}{\text{Hyp}}$ & $\cos \theta = \frac{\text{Adj}}{\text{Hyp}}$ & $\tan \theta = \frac{\text{Opp}}{\text{Adj}}$ \\
\hline
Adj & & & \\
\hline
Hyp & & & \\
\hline
\end{tabular}

\subsection*{2) Imp - Results:}
\begin{itemize}
    \item $\sin^2 \theta + \cos^2 \theta = 1$
    \item $\sec^2 \theta = 1 + \tan^2 \theta$
    \item $\csc^2 \theta = 1 + \cot^2 \theta$
    \item $\sin 2\theta = 2 \sin \theta \cos \theta$
    \item $\cos 2\theta = \cos^2 \theta - \sin^2 \theta$
    \item $\tan 2\theta = \frac{2 \tan \theta}{1 - \tan^2 \theta}$
\end{itemize}

\subsection*{3) Sign Convention:}

\begin{tabular}{|c|c|c|c|c|}
\hline
 & $\sin$ & $\cos$ & $\tan$ & $\cot$ \\
\hline
I & + & + & + & + \\
\hline
II & + & - & - & + \\
\hline
III & - & - & + & - \\
\hline
IV & - & + & - & - \\
\hline
\end{tabular}

\subsection*{4) Trigonometry Standards of Angled Angles:}

\begin{tabular}{|c|c|c|c|c|c|c|}
\hline
$\theta$ & $0^\circ$ & $30^\circ$ & $45^\circ$ & $60^\circ$ & $90^\circ$ & $180^\circ$ \\
\hline
$\sin \theta$ & 0 & $\frac{1}{2}$ & $\frac{\sqrt{2}}{2}$ & $\frac{\sqrt{3}}{2}$ & 1 & 0 \\
\hline
$\cos \theta$ & 1 & $\frac{\sqrt{3}}{2}$ & $\frac{\sqrt{2}}{2}$ & $\frac{1}{2}$ & 0 & -1 \\
\hline
$\tan \theta$ & 0 & $\frac{1}{\sqrt{3}}$ & 1 & $\sqrt{3}$ & $\infty$ & 0 \\
\hline
\end{tabular}

\subsection*{5) Allied Angles:}
\begin{itemize}
    \item $\sin (A \pm B) = \sin A \cos B \pm \cos A \sin B$
    \item $\cos (A \pm B) = \cos A \cos B \mp \sin A \sin B$
\end{itemize}

\subsection*{6) Compound Angles:}
\begin{itemize}
    \item $\sin A \cos B + \cos A \sin B = \sin (A + B)$
    \item $\sin A \cos B - \cos A \sin B = \sin (A - B)$
\end{itemize}

\subsection*{7) Triple Angle Formulae:}
\begin{enumerate}
    \item $\sin 3\theta = 3 \sin \theta - 4 \sin^3 \theta$
    \item $\cos 3\theta = 4 \cos^3 \theta - 3 \cos \theta$
    \item $\tan 3\theta = \frac{3 \tan \theta - \tan^3 \theta}{1 - 3 \tan^2 \theta}$
\end{enumerate}

\subsection*{8) Factorization Formulae:}
\begin{enumerate}
    \item $\sin C + \sin D = 2 \sin \left( \frac{C+D}{2} \right) \cos \left( \frac{C-D}{2} \right)$
    \item $\cos C + \cos D = 2 \cos \left( \frac{C+D}{2} \right) \cos \left( \frac{C-D}{2} \right)$
    \item $\sin C - \sin D = 2 \cos \left( \frac{C+D}{2} \right) \sin \left( \frac{C-D}{2} \right)$
    \item $\cos C - \cos D = -2 \sin \left( \frac{C+D}{2} \right) \sin \left( \frac{C-D}{2} \right)$
\end{enumerate}

\subsection*{9) Defactorization formulae:}
\begin{enumerate}
    \item $2 \sin A \cos B = \sin (A + B) + \sin (A - B)$
    \item $2 \cos A \sin B = \sin (A + B) - \sin (A - B)$
    \item $2 \cos A \cos B = \cos (A + B) + \cos (A - B)$
    \item $2 \sin A \sin B = \cos (A - B) - \cos (A + B)$
\end{enumerate}

\section*{Co - Ordinate - Geometry (2D)}

\subsection*{1. Distance Formula:}
\[
P(x_1, y_1), Q(x_2, y_2) \implies PQ = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}
\]

\subsection*{2. Section Formula:}
\[
P(x, y) = \left( \frac{mx_2 + nx_1}{m+n}, \frac{my_2 + ny_1}{m+n} \right) \quad \text{(Internally)}
\]

\subsection*{3. Mid point Formula:}
\[
\left( \frac{x_1 + x_2}{2}, \frac{y_1 + y_2}{2} \right)
\]

\subsection*{4. Centroid of triangle:}
\[
\left( \frac{x_1 + x_2 + x_3}{3}, \frac{y_1 + y_2 + y_3}{3} \right)
\]

\subsection*{5. Area of triangle:}
\[
\frac{1}{2} \left| x_1(y_2 - y_3) + x_2(y_3 - y_1) + x_3(y_1 - y_2) \right|
\]

\subsection*{6. Slope of straight line:}
\[
m = \tan \theta
\]

\subsection*{7. One point form:}
\[
y - y_1 = m(x - x_1)
\]

\subsection*{8. Intercept Form:}
\[
\frac{x}{a} + \frac{y}{b} = 1
\]

\subsection*{9. Point Form:}
\[
\frac{xx_1}{a} + \frac{yy_1}{b} = 1
\]

\subsection*{10. Normal Form:}
\[
x \cos \alpha + y \sin \alpha = p
\]

\subsection*{11. Intersection of two lines:}
\[
\begin{vmatrix}
a_1 & b_1 & c_1 \\
a_2 & b_2 & c_2 \\
a_3 & b_3 & c_3 \\
\end{vmatrix} \neq 0
\]

\subsection*{12. Angle between two lines:}
\[
\tan \theta = \left| \frac{m_1 - m_2}{1 + m_1 m_2} \right|
\]

\subsection*{13. If the lines are parallel:}
\[
m_1 = m_2
\]

\subsection*{14. Length of line segment:}
\[
\sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}
\]

\subsection*{15. Bisector Equations:}
\[
\frac{x^2 - y^2}{a^2 - b^2} = \frac{y^2 - x^2}{b^2 - a^2}
\]

\subsection*{16. Distance between parallel lines:}
\[
\frac{|c_1 - c_2|}{\sqrt{a^2 + b^2}}
\]

\subsection*{17. Homogeneous Equation:}
\[
ax^2 + 2hxy + by^2 = 0
\]

\subsection*{18. General Equation:}
\[
ax^2 + 2hxy + by^2 + 2gx + 2fy + c = 0
\]

\subsection*{19. If $h^2 = ab$, represents a pair of lines:}
\[
\text{Point of intersection: } \left( \frac{-g}{a}, \frac{-f}{b} \right)
\]

\section*{Co - Ordinate - Geometry (3D)}

\subsection*{1. Distance Formula:}
\[
P(x_1, y_1, z_1), Q(x_2, y_2, z_2) \implies PQ = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2 + (z_2 - z_1)^2}
\]

\subsection*{2. Section Formula:}
\[
P(x, y, z) = \left( \frac{mx_2 + nx_1}{m+n}, \frac{my_2 + ny_1}{m+n}, \frac{mz_2 + nz_1}{m+n} \right) \quad \text{(Internally)}
\]
