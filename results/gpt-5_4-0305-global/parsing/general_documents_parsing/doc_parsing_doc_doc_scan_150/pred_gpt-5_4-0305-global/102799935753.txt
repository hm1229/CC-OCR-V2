\[
\sin C=\frac{\textit{side opposite to angle }C}{\textit{hypotenuse}}=\frac{p}{h}
\]

\[
\cos C=\frac{\textit{side adjacent to angle }C}{\textit{hypotenuse}}=\frac{b}{h}
\]

\[
\tan C=\frac{\textit{side opposite to angle }C}{\textit{side adjacent to angle }C}=\frac{p}{b}
\]

\[
\cosec \theta=\frac{1}{\sin \theta};\ \sec \theta=\frac{1}{\cos \theta};\ \cot \theta=\frac{1}{\tan \theta};\ \tan \theta=\frac{\sin \theta}{\cos \theta}
\]
