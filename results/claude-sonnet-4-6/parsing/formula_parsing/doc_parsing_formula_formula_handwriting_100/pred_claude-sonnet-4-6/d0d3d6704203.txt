$$= \sum_{t=1}^{T} R_t$$

$$\leq \sum_{t=1}^{T} \frac{\mathbb{E}}{r_{t,i}, i \in [n]} \left[ \underbrace{\left| f_2(\phi(x_t, i^*); \theta_{t-1}^{2,*}) - (\gamma_t, i^* - f_1(x_t, i^*; \theta_{t-1}^{1,*})) \right|}_{I_2} \right]$$

$$+ \sum_{t=1}^{T} \underbrace{\left| f_2(\phi(x_t, i^*); \theta_{t-1}^{2,*}) - f_2(\phi(x_t, i^*); \theta_{t-1}^2); \theta_{t-1}^2) \right|}_{I_3}$$

$$+ \sum_{t=1}^{T} \underbrace{\left| f_1(x_t, i^*; \theta_{t-1}^{1,*}) - f_1(x_t, i^*; \theta_{t-1}^1) \right|}_{I_4}$$

$$+ \sum_{t=1}^{T} \frac{\mathbb{E}}{r_t, \hat{i}, \hat{i} \in [n]} \left[ \underbrace{\left| f_2(\phi(x_t, \hat{i}); \theta_{t-1}^2) - (\gamma_t, \hat{i} - f_1(x_t, \hat{i}; \theta_{t-1}^1)) \right|}_{I_5} \right]$$

$$\leq 2\sum_{t=1}^{T} \left( \frac{\sqrt{\Psi(\theta_0^2, R/m^{1/4})} + O(1)}{\sqrt{T}} + \sqrt{\frac{2\log(O(1)/\delta)}{T}} \right)$$

$$+ 2\sum_{t=1}^{T} \left( O(\sqrt{L} \, R/m^{1/4}) + O(R^{4/3} L^2 \sqrt{\log m}/m^{1/3}) \right)$$

$$\leq \sum_{t=1}^{T} \left( \frac{2\sqrt{\Psi(\theta_0^2, R/m^{1/4})} + O(1)}{\sqrt{T}} + 2\sqrt{\frac{2\log(O(1)/\delta)}{T}} \right)$$

$$+ O(\sqrt{L T R/m^{1/4}}) + O(T R^{4/3} L^2 \sqrt{\log m}/m^{1/3})$$

$$\leq 2\sqrt{\Psi(\theta_0^2, R/m^{1/4}) T} + 2\sqrt{2\log(O(1)/\delta) \cdot T} + O(1)$$
