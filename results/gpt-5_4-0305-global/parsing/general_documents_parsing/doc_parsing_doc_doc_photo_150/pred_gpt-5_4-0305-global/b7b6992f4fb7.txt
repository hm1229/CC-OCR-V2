```latex
\textbf{SEQUENCE AND SERIES} \hfill 8.3

\begin{center}
\textbf{OBJECTIVE QUESTIONS}
\end{center}

\begin{enumerate}
\item If the sum of the first $2n$ terms of the A.P.
$2, 5, 8, \ldots$ is equal to the sum of the first
$n$ terms of the A.P. $57, 59, 61, \ldots$, then $n$ equals

\begin{enumerate}
\item[(a)] $10$
\item[(b)] $12$
\item[(c)] $11$
\item[(d)] $13$
\end{enumerate}

\item If $x, y, z$ are three positive numbers in G.P. then
\[
\frac{1}{1+\ln x},\ \frac{1}{1+\ln y},\ \frac{1}{1+\ln z}
\]
are in

\begin{enumerate}
\item[(a)] A.P.
\item[(b)] G.P.
\item[(c)] H.P.
\item[(d)] none of those
\end{enumerate}

\item Let $a, b, c$ are in A.P. and $a^2, b^2, c^2$ are in G.P. If $a<c$ and $a+b+c=\dfrac{3}{2}$, then value of $a$ is

\begin{enumerate}
\item[(a)] $\dfrac{1}{2\sqrt{2}}$
\item[(b)] $\dfrac{1}{2\sqrt{3}}$
\item[(c)] $\dfrac{1}{2}-\dfrac{1}{\sqrt{3}}$
\item[(d)] $\dfrac{1}{2}-\dfrac{1}{\sqrt{2}}$
\end{enumerate}

\item If $1, \log_y x, \log_z y, -15\log_x z$ are in A.P., then

\begin{enumerate}
\item[(a)] $z^3=x$
\item[(b)] $x=y^{-1}$
\item[(c)] $z^{-1}=y$
\item[(d)] all of these
\end{enumerate}

\item The sum to $n$ terms of the series
\[
\frac{2}{3}+\frac{8}{9}+\frac{26}{27}+\frac{80}{81}+\cdots
\]
is equal to

\begin{enumerate}
\item[(a)] $1-\left(\dfrac{1}{3}\right)^n$
\item[(b)] $2-\dfrac{1}{2}\left(\dfrac{2}{3}\right)^n$
\item[(c)] $n-n\left(\dfrac{1}{3}\right)^n$
\item[(d)] none of these
\end{enumerate}

\item Let $a_n$ be the $n$th term of a G.P. of positive integers.
Let
\[
\sum_{n=1}^{100}\frac{1}{2a_n}=\alpha \quad \text{and} \quad \sum_{n=1}^{100}\frac{1}{2a_{n+1}}=\beta
\]
such that $\alpha\ne\beta$.
Thus common ratio of the G.P. is

\begin{enumerate}
\item[(a)] $\dfrac{\alpha}{\beta}$
\item[(b)] $\dfrac{\beta}{\alpha}$
\item[(c)] $\left(\dfrac{\alpha}{\beta}\right)^{1/2}$
\item[(d)] $\left(\dfrac{\beta}{\alpha}\right)^{1/2}$
\end{enumerate}

\item Let $T_r$ be the $r$th term of an A.P. for
$r=1,2,3\ldots$ If for some positive integers $m,n$
\[
T_m=\frac{1}{n},\ T_n=\frac{1}{m},
\]
then $T_{m+n}$ equals

\begin{enumerate}
\item[(a)] $\dfrac{1}{mn}$
\item[(b)] $\dfrac{1}{m}+\dfrac{1}{n}$
\item[(c)] $1$
\item[(d)] $0$
\end{enumerate}

\item If $a_1,a_2,\ldots,a_n$ are positive real numbers whose
product is a fixed number $c$, then the minimum
value of $a_1+a_2+\cdots+a_{n-1}+2a_n$ is

\begin{enumerate}
\item[(a)] $n(2c)^{1/n}$
\item[(b)] $(n+1)c^{1/n}$
\item[(c)] $2nc^{1/n}$
\item[(d)] $(n+1)(2c)^{1/n}$
\end{enumerate}

\item If the first term of an A.P. of consecutive integers
is $p^2+1$, then sum of $(2p+1)$ terms of this series
can be expressed as

\begin{enumerate}
\item[(a)] $(p+1)^2$
\item[(b)] $(2p+1)(p+1)^2$
\item[(c)] $(p+1)^3$
\item[(d)] $p^3+(p+1)^3$
\end{enumerate}

\item If $x>1$ and
\[
\left(\frac{1}{x}\right)^a,\ \left(\frac{1}{x}\right)^b,\ \left(\frac{1}{x}\right)^c
\]
are in G.P., then
$a,b,c$ are in

\begin{enumerate}
\item[(a)] A.P.
\item[(b)] G.P.
\item[(c)] H.P.
\item[(d)] none of these
\end{enumerate}

\item If the positive numbers $a,b,c,d$ be in A.P, then
$abc, abd, bcd$ are

\begin{enumerate}
\item[(a)] not in A.P., G.P. and H.P.
\item[(b)] in A.P.
\item[(c)] in G.P.
\item[(d)] in H.P.
\end{enumerate}

\item The sixth term of an A.P. is $2$ and its common
difference is greater than one. The value of the
common difference of the progression so that the
product of the first, fourth and fifth terms is
greatest, is

\begin{enumerate}
\item[(a)] $\dfrac{8}{5}$
\item[(b)] $\dfrac{2}{3}$
\item[(c)] $\dfrac{5}{8}$
\item[(d)] $\dfrac{3}{2}$
\end{enumerate}

\end{enumerate}
```
