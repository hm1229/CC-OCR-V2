```latex
\documentclass{article}
\usepackage{amsmath, amssymb}

\begin{document}

\begin{center}
\textbf{OBJECTIVE QUESTIONS}
\end{center}

\textbf{1.} If the sum of the first $2n$ terms of the A.P. $2, 5, 8, \ldots$ is equal to the sum of the first $n$ terms of the A.P. $57, 59, 61, \ldots$, then $n$ equals

(a) $10$ \hfill (b) $12$

(c) $11$ \hfill (d) $13$

\medskip

\textbf{2.} If $x, y, z$ are three positive numbers in G.P. then
$$\frac{1}{1+\ln x}, \frac{1}{1+\ln y}, \frac{1}{1+\ln z}$$
are in

(a) A.P. \hfill (b) G.P.

(c) H.P. \hfill (d) none of these

\medskip

\textbf{3.} Let $a, b, c$ are in A.P. and $a^2, b^2, c^2$ are in G.P. If $a < b < c$ and $a + b + c = \dfrac{3}{2}$, then value of $a$ is

(a) $\dfrac{1}{2\sqrt{2}}$ \hfill (b) $\dfrac{1}{2\sqrt{3}}$

(c) $\dfrac{1}{2} - \dfrac{1}{\sqrt{3}}$ \hfill (d) $\dfrac{1}{2} - \dfrac{1}{\sqrt{2}}$

\medskip

\textbf{4.} If $1, \log_y x, \log_x y, -15\log_x z$ are in A.P., then

(a) $z^3 = x$ \hfill (b) $x = y^{-1}$

(c) $z^{-3} = y$ \hfill (d) all of these

\medskip

\textbf{5.} The sum to $n$ terms of the series
$$\frac{2}{3} + \frac{8}{9} + \frac{26}{27} + \frac{80}{81} + \cdots$$
is equal to

(a) $1 - \left(\dfrac{1}{3}\right)^n$ \hfill (b) $2 - \dfrac{1}{2}\left(\dfrac{2}{3}\right)^n$

(c) $n - n\left(\dfrac{1}{3}\right)^n$ \hfill (d) none of these

\medskip

\textbf{6.} Let $a_n$ be the $n$th term of a G.P. of positive integers.

Let $\displaystyle\sum_{n=1}^{100} a_{2n} = \alpha$ and $\displaystyle\sum_{n=1}^{100} a_{2n+1} = \beta$ such that $\alpha \neq \beta$.

Then common ratio of the G.P. is

(a) $\dfrac{\alpha}{\beta}$ \hfill (b) $\dfrac{\beta}{\alpha}$

(c) $\left(\dfrac{\alpha}{\beta}\right)^{1/2}$ \hfill (d) $\left(\dfrac{\beta}{\alpha}\right)^{1/2}$

\medskip

\textbf{7.} Let $T_r$ be the $r$th term of an A.P. for $r = 1, 2, 3, \ldots$ If for some positive integers $m, n$
$$T_m = \frac{1}{n}, \quad T_n = \frac{1}{m},$$
then $T_{mn}$ equals

(a) $\dfrac{1}{mn}$ \hfill (b) $\dfrac{1}{m} + \dfrac{1}{n}$

(c) $1$ \hfill (d) $0$

\medskip

\textbf{8.} If $a_1, a_2, \ldots, a_n$ are positive real numbers whose product is a fixed number $c$, then the minimum value of $a_1 + a_2 + \cdots + a_{n-1} + 2a_n$ is

(a) $n(2c)^{1/n}$ \hfill (b) $(n+1)c^{1/n}$

(c) $2nc^{1/n}$ \hfill (d) $(n+1)(2c)^{1/n}$

\medskip

\textbf{9.} If the first term of an A.P. of consecutive integers is $p^2 + 1$, then sum of $(2p+1)$ terms of this series can be expressed as

(a) $(p+1)^2$ \hfill (b) $(2p+1)(p+1)^2$

(c) $(p+1)^3$ \hfill (d) $p^3 + (p+1)^3$

\medskip

\textbf{10.} If $x > 1$ and $\left(\dfrac{1}{x}\right)^a, \left(\dfrac{1}{x}\right)^b, \left(\dfrac{1}{x}\right)^c$ are in G.P., then $a, b, c$ are in

(a) A.P. \hfill (b) G.P.

(c) H.P. \hfill (d) none of these

\medskip

\textbf{11.} If the positive numbers $a, b, c, d$ be in A.P., then $abc, abd, bcd$ are

(a) not in A.P., G.P. and H.P. \hfill (b) in A.P.

(c) in G.P. \hfill (d) in H.P.

\medskip

\textbf{12.} The sixth term of an A.P. is $2$ and its common difference is greater than one. The value of the common difference of the progression so that the product of the first, fourth and fifth terms is greatest, is

(a) $\dfrac{8}{5}$ \hfill (b) $\dfrac{2}{3}$

(c) $\dfrac{5}{8}$ \hfill (d) $\dfrac{3}{2}$
```
