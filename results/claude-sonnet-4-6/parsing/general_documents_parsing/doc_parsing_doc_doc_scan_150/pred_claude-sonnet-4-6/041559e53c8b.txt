\begin{center}
\textbf{Important Algebraic Formulas}
\end{center}

\begin{align*}
&(a + b)^2 = a^2 + b^2 + 2ab \\[6pt]
&(a - b)^2 = a^2 + b^2 - 2ab \\[6pt]
&a^2 - b^2 = (a + b)(a - b) \\[6pt]
&a^2 + b^2 = (a + b)^2 - 2ab \qquad \text{or} \qquad a^2 + b^2 = (a - b)^2 + 2ab \\[6pt]
&a^3 + b^3 = (a + b)(a^2 - ab + b^2) = (a + b)^3 - 3ab(a + b) \\[6pt]
&a^3 - b^3 = (a - b)(a^2 + ab + b^2) = (a - b)^3 + 3ab(a - b) \\[6pt]
&2(a^2 + b^2) = (a + b)^2 + (a - b)^2 \\[6pt]
&(a + b)^2 - (a - b)^2 = 4ab \\[6pt]
&a^4 + b^4 = (a + b)(a - b)\left[(a + b)^2 - 2ab\right] \\[6pt]
&(a - b)^2 = (a + b)^2 - 4ab \\[6pt]
&(a + b)^2 = (a - b)^2 + 4ab \\[6pt]
&a^4 + b^4 = \left[(a + b)^2 - 2ab\right]^2 - 2(ab)^2 \\[6pt]
&(a + b + c)^2 = a^2 + b^2 + c^2 + 2ab + 2bc + 2ca \\[6pt]
&(a + b - c)^2 = a^2 + b^2 + c^2 + 2ab - 2bc - 2ca \\[6pt]
&(a - b - c)^2 = a^2 + b^2 + c^2 - 2ab + 2bc - 2ca \\[6pt]
&a^3 + b^3 + c^3 - 3abc = (a + b + c)(a^2 + b^2 + c^2 - ab - bc - ca) \\[6pt]
&a^4 + a^2b^2 + b^4 = (a^2 + ab + b^2)(a^2 - ab + b^2) \\[6pt]
&a^4 + a^2 + 1 = (a^2 + a + 1)(a^2 - a + 1) \\[6pt]
&\textit{if } a + b + c = 0 \textit{ then } a^3 + b^3 + c^3 = 3abc \\[6pt]
&a^8 - b^8 = (a^4 + b^4)(a^2 + b^2)(a + b)(a - b)
\end{align*}
