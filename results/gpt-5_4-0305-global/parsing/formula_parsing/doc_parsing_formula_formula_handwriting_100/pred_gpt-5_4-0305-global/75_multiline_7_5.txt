\[
\Sigma_{1}
=\frac{(N-N_{de})!}{N!}\sum_{\alpha_{1}}\sum_{\alpha_{2}\neq \alpha_{1}}\cdots
\sum_{\alpha_{N_{de}}\neq \alpha_{1},\ldots,\alpha_{N_{de}-1}}
\sum_{j\neq i}\Gamma_{j i}\,[\Gamma_{i j}]
\]

\[
=\frac{(N-N_{de})!}{N!}\sum_{\alpha_{1}}\sum_{\alpha_{2}\neq \alpha_{1}}\cdots
\sum_{\alpha_{N_{de}}\neq \alpha_{1},\ldots,\alpha_{N_{de}-1}}
\left(
\sum_{j\neq \alpha_{1}}\Gamma_{j\alpha_{1}}
+\cdots+
\sum_{j\neq \alpha_{N_{de}}}\Gamma_{j\alpha_{N_{de}}}
\right)
\]

\[
=\frac{(N-N_{de})!}{N!}\sum_{\alpha_{1}}\sum_{\alpha_{2}\neq \alpha_{1}}\cdots
\sum_{\alpha_{N_{de}}\neq \alpha_{1},\ldots,\alpha_{N_{de}-1}}
\]

\[
=\frac{N_{de}}{N!}\sum_{j\neq i}\Gamma_{j i}
\]

\[
\Sigma_{3}
=\frac{(N-N_{de})!}{N!}\sum_{\alpha_{1}}\sum_{\alpha_{2}\neq \alpha_{1}}\cdots
\sum_{\alpha_{N_{de}}\neq \alpha_{1},\ldots,\alpha_{N_{de}-1}}
\sum_{j\neq i}\Gamma_{i j}\,[\Gamma_{j i}]
\]

\[
=\frac{(N-N_{de})!}{N!}\sum_{\alpha_{1}}\sum_{\alpha_{2}\neq \alpha_{1}}\cdots
\sum_{\alpha_{N_{de}}\neq \alpha_{1},\ldots,\alpha_{N_{de}-1}}
\left(
\Gamma_{\alpha_{1}j}\sum_{j\neq \alpha_{1}}
+\Gamma_{\alpha_{2}j}\sum_{j\neq \alpha_{2}}
+\cdots+
\Gamma_{\alpha_{N_{de}}j}\sum_{j\neq \alpha_{N_{de}}}
\right)
\]

\[
= N_{de}(N_{de}-1)\frac{(N-N_{de})!}{N!}
\sum_{\alpha_{1}}\sum_{\alpha_{2}\neq \alpha_{1}}\cdots
\sum_{\alpha_{N_{de}}\neq \alpha_{1},\ldots,\alpha_{N_{de}-1}}
\]

\[
=\frac{N_{de}(N_{de}-1)}{N(N-1)}
\sum_{i,j\neq i}\Gamma_{i j}\,[\Gamma_{j i}]
\]
