\[
\lim_{\varepsilon\to 0}\left[I_{\varepsilon-k,\varepsilon}(\phi)-\frac{\phi^{(k-1)}(0)}{\varepsilon (k-1)!}\right]
\]

\[
=\lim_{\varepsilon\to 0}\left[\frac{(-1)^k I_{\varepsilon,0}(\phi^{(k)})}{\prod_{\ell=0}^{k-1}(\varepsilon-\ell)}
+\sum_{j=0}^{k-1}\frac{(-1)^j\phi^{(j)}(\varepsilon)\varepsilon^{-k+j+1}}{\prod_{\ell=k-j-1}^{k-1}(\varepsilon-\ell)}
-\frac{\phi^{(k-1)}(0)}{\varepsilon (k-1)!}\right]
\]

\[
=\lim_{\varepsilon\to 0}\left[\frac{(-1)^k}{\prod_{\ell=1}^{k-1}(\varepsilon-\ell)}
\int_{r=0}^{\varepsilon}\frac{r^{\varepsilon-1}}{\varepsilon}\,\phi^{(k)}(r)
+\frac{(-1)^k\varepsilon^{\varepsilon}\phi^{(k)}(0)}{\varepsilon\prod_{\ell=1}^{k-1}(\varepsilon-\ell)}
-\frac{\phi^{(k-1)}(0)}{\varepsilon\prod_{\ell=1}^{k-1}(\varepsilon-\ell)}
+\sum_{j=0}^{k-2}\frac{(-1)^j\phi^{(j)}(0)\varepsilon^{-k+j+1}}{\prod_{\ell=k-j-1}^{k-1}(\varepsilon-\ell)}\right]
\]

\[
=\lim_{\varepsilon\to 0}\left[\frac{(-1)^k}{\prod_{\ell=1}^{k-1}(\varepsilon-\ell)}
\int_{r=0}^{\varepsilon}\frac{r^{\varepsilon-1}}{\varepsilon}\,\phi^{(k)}(r)
-\frac{\phi^{(k-1)}(0)}{\varepsilon}\left(\frac{1}{(k-1)!}+\frac{(-1)^k}{\prod_{\ell=1}^{k-1}(\varepsilon-\ell)}\right)
+\sum_{j=0}^{k-2}\frac{(-1)^j\phi^{(j)}(0)\varepsilon^{-k+j+1}}{\prod_{\ell=k-j-1}^{k-1}(\varepsilon-\ell)}\right]
\]

=-\frac{1}{(k-1)!}\int_{r=0}^{\varepsilon}(\ln r)\,\phi^{(k)}(r)
+\frac{\phi^{(k-1)}(0)}{(k-1)!}\sum_{r=1}^{k-1}\frac{1}{r}
-\frac{1}{(k-1)!}\sum_{j=0}^{k-2}\phi^{(j)}(0)\varepsilon^{-k+j+1}(k-j-2)!.
