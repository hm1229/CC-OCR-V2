\begin{center}
\textbf{Formula Sheet for Physics 251}

\begin{multicols}{3}

\underline{Constants}

\[R = 8.3145 \frac{\text{J}}{\text{m} \cdot \text{K}}\]

\[N_A = 6.02 \times 10^{23}\]

\[k_B = \frac{R}{N_A} = 1.38 \times 10^{-23} \frac{\text{J}}{\text{m} \cdot \text{K}}\]

\[e = 1.602 \times 10^{-19} \text{ C}\]

\[k = \frac{1}{4\pi\varepsilon_0} = 9 \times 10^9 \frac{\text{N} \cdot \text{m}^2}{\text{C}^2}\]

\[\mu_0 = 4\pi \times 10^{-7} \frac{\text{T} \cdot \text{m}}{\text{A}}\]

\[c = 3 \times 10^8 \frac{\text{m}}{\text{s}}\]

\underline{Thermal Expansion}

\[\Delta L = \alpha l_0 \Delta T\]

\[\beta = 3\alpha\]

\underline{Heat}

\[Q = mc\Delta T\]

\[Q = \pm mL\]

\[Q = nC_V \Delta T\]

\[Q = nC_P \Delta T\]

\underline{Heat Current}

\[H = kA\frac{\dot{T}}{l}(T_H - T_C)\]

\[H = Ae\sigma T^4\]

\underline{Quantities}

\underline{Work}

\[W = \int_{V_i}^{V_f} P \, dV\]

\underline{Isochoric Processes}

\[W = 0\]

\[Q = nC_V \Delta T\]

\[* \Delta U = nC_V \Delta T\]

\columnbreak

\[m_{tot} = nM\]

\[M = N_A m\]

\[N = N_A n\]

\underline{Ideal gases}

\[* PV = nrT\]

\[* PV = NkT\]

\[* v_{rms} = \sqrt{\frac{3RT}{M}}\]

\[* C_V = \frac{3}{2}R \text{ (monatomic)}\]

\[* C_V = \frac{5}{2}R \text{ (diatomic)}\]

\[* C_P = C_V + R\]

\underline{Isobaric Processes}

\[W = P\Delta V\]

\[Q = nC_P \Delta T\]

\[* \Delta U = nC_V \Delta T\]

\underline{Isothermal Processes}

\[* W = nRT\ln\left(\frac{V_f}{V_i}\right)\]

\[* Q = nRT\ln\left(\frac{V_f}{V_i}\right)\]

\[* \Delta U = 0\]

\underline{Adiabatic Processes}

\[* W = -nC_V \Delta T\]

\[Q = 0\]

\[* \Delta U = nC_V \Delta T\]

\[* P_i V_i^\gamma = P_f V_f^\gamma\]

\[* T_i V_i^{\gamma-1} = T_f V_f^{\gamma-1}\]

\[\gamma = \frac{C_P}{C_V}\]

\underline{Cyclic Processes}

\[W = Q\]

\[\Delta U = 0\]

\columnbreak

\underline{Heat Engines}

\[W = Q_H + Q_C\]

\[e = \frac{W}{Q_H}\]

\[e_c = 1 - \frac{T_C}{T_H}\]

\underline{Refrigerators}

\[W = Q_H + Q_C\]

\[K = \frac{|Q_C|}{|W|}\]

\[K_c = \frac{T_C}{T_H - T_C}\]

\underline{Force, Field, Energy and Potential}

\[\vec{F} = q\vec{E}\]

\[\Delta U = q\Delta V\]

\[V_{ab} = \int_a^b \vec{E} \cdot d\vec{l}\]

\[\vec{E} = -\nabla V\]

\underline{1 Point Charge}

\[\vec{E} = k\frac{q}{r^2}\hat{r}\]

\[V = k\frac{q}{r}\]

\underline{2 Point Charges}

\[\vec{F} = k\frac{q_1 q_2}{r^2}\hat{r}\]

\[U = k\frac{q_1 q_2}{r}\]

\underline{Distributed Charge}

\[\vec{E} = \int k\frac{dq}{r^2}\hat{r}\]

\[V = \int k\frac{dq}{r}\]

\underline{Many Sources}

\[\vec{E}_T = \sum \vec{E}_i\]

\[U_T = \sum_{pairs} U_{ij}\]

\underline{``Elementary'' $E$ Fields}

\[\vec{E} = k\frac{q}{r^2}\hat{r} \text{ sphere}\]

\[\vec{E} = \frac{\lambda}{2\pi\varepsilon_0 r}\hat{r} \text{ line}\]

\[\vec{E} = \frac{\sigma}{2\varepsilon_0}\hat{n} \text{ plane}\]

\[\vec{E} = \frac{\sigma}{\varepsilon_0}\hat{n} \text{ conducting surface}\]

\underline{Capacitors}

\[C = \frac{Q}{V}\]

\[C_T = \sum C_i \text{ parallel}\]

\[\frac{1}{C_T} = \sum \frac{1}{C_i} \text{ series}\]

\[U = \frac{1}{2}CV^2 = \frac{Q^2}{2C} = \frac{1}{2}QV\]

\[\dagger C = \varepsilon_0 \frac{A}{d}\]

\[\dagger |E| = \frac{V}{d} = \frac{\sigma}{\varepsilon_0}\]

\underline{Flux and Gauss's Law}

\[\phi = \iint \vec{E} \cdot d\vec{a}\]

\[\phi_{closed} = \frac{Q_{enclosed}}{\varepsilon_0}\]

\underline{Current and Current Density}

\[I = \frac{dq}{dt}\]

\[I = \iint \vec{J} \cdot d\vec{a}\]

\[\vec{J} = nqv_d\]

\end{multicols}

\vspace{1em}
\noindent * indicates formulas that are specific to ideal gases

\noindent $\dagger$ indicates formulas that are specific to parallel-plate capacitors
