\section*{III. CALCULATIONS:}

Calculating the volume of the given examined bodies based on the formula:
\[ V_b = a \cdot b \cdot c \]
$V_b$ - volume of examined bodies [cm$^3$], $a$ - width of examined bodies [cm], $b$ - length of examined bodies [cm], $c$ - height of examined bodies [cm].
\[ V_b = 2,5 \text{ [cm]} \cdot 4,2 \text{ [cm]} \cdot 1,5 \text{ [cm]} = 15,75 \text{ [cm}^3\text{]} \]

Calculating the material density of the given examined bodies based on the formula:
\[ \varrho_b = \frac{m_b}{V_b} \]
$\varrho_b$ - material density of given body [g/cm$^3$], $m_b$ - mass of given body [g], $V_b$ - volume of examined body.
\begin{align*}
\varrho_b \text{ (ebonite)} &= \frac{22,7}{15,75} \text{ [g/cm}^3\text{]} = 1,44127 \text{ [g/cm}^3\text{]} = 1441,27 \text{ [kg/m}^3\text{]} \\
\varrho_b \text{ (bakelite)} &= \frac{19,8}{15,75} \text{ [g/cm}^3\text{]} = 1,25714 \text{ [g/cm}^3\text{]} = 1257,14 \text{ [kg/m}^3\text{]} \\
\varrho_b \text{ (aluminium (old))} &= \frac{42,6}{15,75} \text{ [g/cm}^3\text{]} = 2,70476 \text{ [g/cm}^3\text{]} = 2704,76 \text{ [kg/m}^3\text{]}
\end{align*}

Calculating the material density using aerometer based on the formula:
\[ \varrho_b' = \frac{m_1 - m_2}{m_3 - m_2} \varrho_w \]
$\varrho_b'$ - material density of given body using aerometer [g/cm$^3$], $m_A$ - values of mass from experiment [g], $\varrho_w$ - density of distilled water = 1 [g/cm$^3$].
\begin{align*}
\varrho_b' \text{ (ebonite)} &= \frac{110 - 87}{103 - 87} \text{ [g/s]} \cdot 1 \text{ [g/cm}^3\text{]} = \frac{23}{16} \text{ [g/cm}^3\text{]} = 1,4375 \text{ [g/cm}^3\text{]} = 1437,5 \text{ [kg/m}^3\text{]}. \\
\varrho_b' \text{ (bakelite)} &= \frac{110 - 90}{105 - 90} \text{ [g/s]} \cdot 1 \text{ [g/cm}^3\text{]} = \frac{20}{15} \text{ [g/cm}^3\text{]} = 1,33333 \text{ [g/cm}^3\text{]} = 1333,33 \text{ [kg/m}^3\text{]} \\
\varrho_b' \text{ (aluminium (old))} &= \frac{110 - 67}{83 - 67} \text{ [g/s]} \cdot 1 \text{ [g/cm}^3\text{]} = \frac{43}{16} \text{ [g/cm}^3\text{]} = 2,6875 \text{ [g/cm}^3\text{]} = 2687,5 \text{ [kg/m}^3\text{]} \\
\varrho_b' \text{ (aluminium (Al))} &= \frac{110 - 86}{94 - 86} \text{ [g/s]} \cdot 1 \text{ [g/cm}^3\text{]} = \frac{24}{8} \text{ [g/cm}^3\text{]} = 3 \text{ [g/cm}^3\text{]} = 3000 \text{ [kg/m}^3\text{]}
\end{align*}

Comparing calculated density values with table density values:
\begin{align*}
\varrho_b \text{ (ebonite)} - \varrho_{\text{tab}} \text{ (ebonite)} &= 1441,27 \text{ [kg/m}^3\text{]} - 1150 \text{ [kg/m}^3\text{]} = 291,27 \text{ [kg/m}^3\text{]} = 22,48 \text{ [\%]} \\
\varrho_b' \text{ (ebonite)} - \varrho_{\text{tab}} \text{ (ebonite)} &= 1437,5 \text{ [kg/m}^3\text{]} - 1150 \text{ [kg/m}^3\text{]} = 287,5 \text{ [kg/m}^3\text{]} = 22,22 \text{ [\%]} \\
\varrho_b \text{ (bakelite)} - \varrho_{\text{tab}} \text{ (bakelite)} &= 1257,14 \text{ [kg/m}^3\text{]} - 1300 \text{ [kg/m}^3\text{]} = 42,86 \text{ [kg/m}^3\text{]} = 3,35 \text{ [\%]} \\
\varrho_b' \text{ (bakelite)} - \varrho_{\text{tab}} \text{ (bakelite)} &= 1333,33 \text{ [kg/m}^3\text{]} - 1300 \text{ [kg/m}^3\text{]} = 33,33 \text{ [kg/m}^3\text{]} = 2,53 \text{ [\%]} \\
\varrho_b \text{ (aluminium (old))} - \varrho_{\text{tab}} \text{ (aluminium)} &= 2704,76 \text{ [kg/m}^3\text{]} - 2700 \text{ [kg/m}^3\text{]} = 4,76 \text{ [kg/m}^3\text{]} = 0,17 \text{ [\%]} \\

\end{align*}
