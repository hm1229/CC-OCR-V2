$$
\begin{aligned}
|T X^R|_{L^r_W L^\infty_{[0, T]} L^2_x} & \le |X_0|_{L^r_W L^2_x} + \left| \int_0^t S_\beta(t, s) \theta_R(X^R)(s) |X^R(s)|^{2\sigma} X^R(s) ds \right|_{L^r_W L^\infty_{[0, T]} L^2_x} \\
& \le |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\bar{\beta}} |\theta_R(X^R) |X^R|^{2\sigma+1}|_{L^r_W L^{p'}_{[0, T]} L^2_x} \\
& = |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\bar{\beta}} \left| \left[ \int_0^T \theta_R(X^R)^{r'}(t) \left( \int_I |X^R(t)|^{(2\sigma+1)p'} dx \right)^{\frac{r'}{p'}} dt \right]^{\frac{1}{r'}} \right|_{L^r_W}
\end{aligned}
$$

Since $p' = \frac{2\sigma+2}{2\sigma+1}$, the right hand side equals
$$
= |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\bar{\beta}} \left| \left( \int_0^T \theta_R(X^R)^{r'}(t) |X^R(t)|_{L^2_x}^{(2\sigma+1)r'} dt \right)^{\frac{1}{r'}} \right|_{L^r_W}
$$

Since $r \ge p = 2\sigma+2$, we can apply Hölder's inequality to get
$$
\le |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\bar{\beta}} \left| \left( \int_0^T [\theta_R(X^R)(t)]^{\frac{r}{2\sigma+1}} |X^R(t)|_{L^2_x}^r dt \right)^{\frac{2\sigma+1}{r}} T^{\frac{r - r'(2\sigma+1)}{r r'}} \right|_{L^r_W}
$$

Letting $\tilde{\tau}(w) := \inf \{ s : |X^R(w)|_{L^r([0, s]; L^2_x)} \ge 2R \}$, from definition of $\theta_R$ in (??), we have
$$
= |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\tilde{\beta}} \left| \left( \int_0^{\tilde{\tau}} [\theta_R(X^R)(t)]^{\frac{r}{2\sigma+1}} |X^R(t)|_{L^2_x}^r dt \right)^{\frac{2\sigma+1}{r}} \right|_{L^r_W}
$$

where $\tilde{\beta} = \bar{\beta} + \frac{r - r'(2\sigma+1)}{r r'} > 0$. Furthermore,
$$
\begin{aligned}
& \le |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\tilde{\beta}} \left| \left( \int_0^{\tilde{\tau}} |X^R(t)|_{L^2_x}^r dt \right)^{\frac{2\sigma+1}{r}} \right|_{L^r_W} \\
& \le |X_0|_{L^r_W L^2_x} + C_{r, p} T^{\tilde{\beta}} (2R)^{2\sigma} |X^R|_{L^r_W L^r_{[0, \tilde{\tau}]} L^2_x},
\end{aligned}
$$
