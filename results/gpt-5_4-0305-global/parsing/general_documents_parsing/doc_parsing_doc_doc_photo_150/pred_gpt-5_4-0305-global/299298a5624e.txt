\subsection*{Trigonometry}

\textbf{1) Trigonometry Ratios :}

\[
\sin \theta=\frac{\text{Opp}}{\text{Hyp}}
\qquad
\cos \theta=\frac{\text{Adj}}{\text{Hyp}}
\]

\[
\tan \theta=\frac{\text{Opp}}{\text{Adj}}
\qquad
\cot \theta=\frac{\text{Adj}}{\text{Opp}}
\]

\[
\sec \theta=\frac{\text{Hyp}}{\text{Adj}}
\qquad
\cosec \theta=\frac{\text{Hyp}}{\text{Opp}}
\]

\textbf{2) Imp.\ Results :}

\[
\text{i)}\ \sin \theta \cdot \cosec \theta=1
\qquad
\text{ii)}\ \cos \theta \cdot \sec \theta=1
\]

\[
\text{iii)}\ \tan \theta \cdot \cot \theta=1
\qquad
\text{iv)}\ \tan \theta=\frac{\sin \theta}{\cos \theta}
\]

\[
\text{v)}\ \cot \theta=\frac{\cos \theta}{\sin \theta}
\qquad
\text{vi)}\ \sin^2\theta+\cos^2\theta=1
\]

\[
\text{vii)}\ \sec^2\theta=1+\tan^2\theta
\qquad
\text{viii)}\ \cosec^2\theta=1+\cot^2\theta
\]

\[
\text{ix)}\ \pi^r=180^\circ
\qquad
\text{x)}\ \text{length of arc }s=r\theta
\]

\[
\text{xi)}\ \text{Area of sector}=\frac12 r^2\theta
\]

\textbf{3) Trigonometry Ratios of standard Angles :}

\[
\begin{array}{|c|c|c|c|c|c|c|c|}
\hline
\text{Ratios} & 0^\circ & 30^\circ & 45^\circ & 60^\circ & 90^\circ & 180^\circ & 270^\circ & 360^\circ \\
\hline
\sin & 0 & \frac12 & \frac{1}{\sqrt2} & \frac{\sqrt3}{2} & 1 & 0 & -1 & 0 \\
\hline
\cos & 1 & \frac{\sqrt3}{2} & \frac{1}{\sqrt2} & \frac12 & 0 & -1 & 0 & 1 \\
\hline
\tan & 0 & \frac{1}{\sqrt3} & 1 & \sqrt3 & \infty & 0 & \infty & 0 \\
\hline
\end{array}
\]

\textbf{4) Allied - Angles :}

\[
\text{i)}\ \sin(-\theta)=-\sin\theta
\qquad
\sin(90^\circ-\theta)=\cos\theta
\]

\[
\cos(-\theta)=\cos\theta
\qquad
\cos(90^\circ-\theta)=\sin\theta
\]

\[
\tan(-\theta)=-\tan\theta
\qquad
\tan(90^\circ-\theta)=\cot\theta
\]

\[
\cot(-\theta)=-\cot\theta
\qquad
\cot(90^\circ-\theta)=\tan\theta
\]

\[
\sec(-\theta)=\sec\theta
\qquad
\sec(90^\circ-\theta)=\cosec\theta
\]

\[
\cosec(-\theta)=-\cosec\theta
\qquad
\cosec(90^\circ-\theta)=\sec\theta
\]

\textbf{5) Compound Angles :}

\[
\text{i)}\ \sin(A+B)=\sin A\cos B+\cos A\sin B
\]

\[
\text{ii)}\ \sin(A-B)=\sin A\cos B-\cos A\sin B
\]

\[
\text{iii)}\ \cos(A+B)=\cos A\cos B-\sin A\sin B
\]

\[
\text{iv)}\ \cos(A-B)=\cos A\cos B+\sin A\sin B
\]

\[
\text{v)}\ \tan(A+B)=\frac{\tan A+\tan B}{1-\tan A\tan B}
\]

\[
\text{vi)}\ \tan(A-B)=\frac{\tan A-\tan B}{1+\tan A\tan B}
\]

\textbf{6) Multiple Angle and Sub Multiple Angle :}

\[
\text{i)}\ \sin 2A=2\sin A\cos A
\qquad
\text{i)}\ \sin A=2\sin\left(\frac A2\right)\cos\left(\frac A2\right)
\]

\[
\text{ii)}\ \cos 2A=\cos^2A-\sin^2A
\qquad
\text{ii)}\ \cos A=\cos^2\left(\frac A2\right)-\sin^2\left(\frac A2\right)
\]

\[
\text{iii)}\ \cos 2A=1-2\sin^2A
\qquad
\text{iii)}\ \cos A=1-2\sin^2\left(\frac A2\right)
\]

\[
\text{iv)}\ \cos 2A=2\cos^2A-1
\qquad
\text{iv)}\ \cos A=2\cos^2\left(\frac A2\right)-1
\]

\[
\text{v)}\ \cos A=\frac{1-\tan^2\left(\frac A2\right)}{1+\tan^2\left(\frac A2\right)}
\]

\[
\text{vi)}\ \sin A=\frac{2\tan\left(\frac A2\right)}{1+\tan^2\left(\frac A2\right)}
\]

\[
\text{vii)}\ \tan A=\frac{2\tan\left(\frac A2\right)}{1-\tan^2\left(\frac A2\right)}
\]

\[
\text{viii)}\ 1-\cos A=2\sin^2\left(\frac A2\right)
\]

\textbf{7) Triple Angle Formulae :}

\[
\text{i)}\ \sin 3\theta=3\sin\theta-4\sin^3\theta
\qquad
\text{ii)}\ \cos 3\theta=4\cos^3\theta-3\cos\theta
\]

\[
\text{iii)}\ \tan 3\theta=\frac{3\tan\theta-\tan^3\theta}{1-3\tan^2\theta}
\]

\[
\text{iv)}\ \cos^3\theta=\frac{\cos 3\theta+3\cos\theta}{4}
\qquad
\text{v)}\ \sin^3\theta=\frac{3\sin\theta-\sin 3\theta}{4}
\]

\textbf{8) Factorization Formulae :}

\[
\text{i.}\ \sin C+\sin D=2\sin\left(\frac{C+D}{2}\right)\cos\left(\frac{C-D}{2}\right)
\]

\[
\text{ii.}\ \sin C-\sin D=2\sin\left(\frac{C-D}{2}\right)\sin\left(\frac{C-D}{2}\right)
\]

\[
\text{iii.}\ \cos C+\cos D=2\cos\left(\frac{C+D}{2}\right)\cos\left(\frac{C-D}{2}\right)
\]

\[
\text{iv.}\ \cos C-\cos D=-2\sin\left(\frac{C+D}{2}\right)\sin\left(\frac{C-D}{2}\right)\ \text{if }C>D
\]

\[
=2\sin\left(\frac{C+D}{2}\right)\sin\left(\frac{D-C}{2}\right)\ \text{if }D>C
\]

\textbf{9) Defactorization formulae :}

\[
\text{i.}\ 2\sin A\cos B=\sin(A+B)+\sin(A-B)
\]

\[
\text{ii.}\ 2\cos A\sin B=\sin(A+B)-\sin(A-B)
\]

\[
\text{iii.}\ 2\cos A\cos B=\cos(A+B)+\cos(A-B)
\]

\[
\text{iv.}\ 2\sin A\sin B=\cos(A-B)-\cos(A+B)
\]
