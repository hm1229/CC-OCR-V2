\section*{2.5 \quad Finance formulas -- Practice exercises}

\bigskip

\noindent \textsc{Compound Interest Formula:} \quad $a = p\left(1 + \dfrac{r}{12}\right)^{12y}$

\bigskip

\noindent \textsc{Equivalent APR Formula:} \quad $\text{APR} = \left(1 + \dfrac{r}{12}\right)^{12} - 1$

\bigskip

\noindent \textsc{Future Value Annuity Formula:} \quad $a = p * \dfrac{\left(1 + \frac{r}{12}\right)^{12y} - 1}{\frac{r}{12}}$

\bigskip

\noindent \textsc{Loan Payment Formula:} \quad $p = \dfrac{a * \frac{r}{12}}{1 - \left(1 + \frac{r}{12}\right)^{-12y}}$

\bigskip

\noindent where

\begin{itemize}
    \item[] $a =$ account balance or loan amount (\$)
    \item[] $p =$ initial deposit (principal), regular deposit, or regular payment (\$)
    \item[] $y =$ time invested (years)
    \item[] $r =$ interest rate compounded monthly (as a decimal)
\end{itemize}
