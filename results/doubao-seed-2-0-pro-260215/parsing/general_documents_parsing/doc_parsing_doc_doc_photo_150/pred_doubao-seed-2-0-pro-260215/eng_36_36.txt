\subsection*{2.5 Finance formulas -- Practice exercises}
\hrule
\vspace{1em}

\noindent COMPOUND INTEREST FORMULA: \quad $\displaystyle a = p\left(1+\frac{r}{12}\right)^{12y}$

\vspace{1em}
\noindent EQUIVALENT APR FORMULA: \quad $\displaystyle \text{APR} = \left(1+\frac{r}{12}\right)^{12} - 1$

\vspace{1em}
\noindent FUTURE VALUE ANNUITY FORMULA: \quad $\displaystyle a = p * \frac{\left(1+\frac{r}{12}\right)^{12y} - 1}{\frac{r}{12}}$

\vspace{1em}
\noindent LOAN PAYMENT FORMULA: \quad $\displaystyle p = \frac{a * \frac{r}{12}}{1 - \left(1+\frac{r}{12}\right)^{-12y}}$

\vspace{1em}
\noindent where
\begin{flalign*}
\quad a &= \text{account balance or loan amount (\$)} & \\
\quad p &= \text{initial deposit (principal), regular deposit, or regular payment (\$)} & \\
\quad y &= \text{time invested (years)} & \\
\quad r &= \text{interest rate compounded monthly (as a decimal)} &
\end{flalign*}
