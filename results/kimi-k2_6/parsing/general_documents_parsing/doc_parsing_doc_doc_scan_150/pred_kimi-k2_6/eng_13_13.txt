\textbf{Statistical Analysis and Probability}
\end{center}

\noindent
\begin{minipage}[t]{0.48\textwidth}
\textbf{Mean}
\[
\bar{x} = \frac{1}{n}\sum_{i=1}^{n} x_i
\]

\vspace{0.5cm}

\textbf{Standard Error of the Mean*}
\[
SE_{\bar{x}} = \frac{s}{\sqrt{n}}
\]

\vspace{0.5cm}

\textbf{Laws of Probability}

If A and B are mutually exclusive, then:
\[
P(\text{A or B}) = P(A) + P(B)
\]

If A and B are independent, then:
\[
P(\text{A and B}) = P(A) \times P(B)
\]

\vspace{0.5cm}

\textbf{Hardy-Weinberg Equations}
\[
p^2 + 2pq + q^2 = 1
\]
\[
p + q = 1
\]

$p =$ frequency of the dominant allele in a population

$q =$ frequency of the recessive allele in a population

\end{minipage}%
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Standard Deviation*}
\[
S = \sqrt{\frac{\sum(x_i - \bar{x})^2}{n-1}}
\]

\vspace{0.5cm}

\textbf{Chi-Square}
\[
\chi^2 = \sum \frac{(o-e)^2}{e}
\]

\vspace{0.5cm}

\noindent
\begin{minipage}[t]{0.48\textwidth}
$\bar{x} =$ sample mean

$n =$ size of the sample

$s =$ sample standard deviation (i.e., the sample-based estimate of the standard deviation of the population)

$o =$ observed results

$e =$ expected results

Degrees of freedom are equal to the number of distinct possible outcomes minus one.
\end{minipage}%

\textbf{Mode} = value that occurs most frequently in a data set

\vspace{0.2cm}
\noindent
\textbf{Median} = middle value that separates the greater and lesser halves of a data set

\vspace{0.2cm}
\noindent
\textbf{Mean} = sum of all data points divided by number of data points

\vspace{0.2cm}
\noindent
\textbf{Range} = value obtained by subtracting the smallest observation (sample minimum) from the greatest (sample maximum)
