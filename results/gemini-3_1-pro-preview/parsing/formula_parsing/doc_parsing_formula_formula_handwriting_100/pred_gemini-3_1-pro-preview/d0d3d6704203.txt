\begin{aligned}
&= \sum_{t=1}^{T} r_t \\
&\leq \underbrace{\sum_{t=1}^{T} \mathbb{E}_{r_{t,i}, i \in [n]} \left[ \left| f_2 \left( \phi(x_t, i^*); \theta_{t-1}^{2,*} \right) - \left( r_{t,i^*} - f_1(x_t, i^*; \theta_{t-1}^{1,*}) \right) \right| \right]}_{I_2} \\
