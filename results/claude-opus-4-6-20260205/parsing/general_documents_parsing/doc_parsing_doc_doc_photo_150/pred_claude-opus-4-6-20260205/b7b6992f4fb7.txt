\documentclass[12pt]{article}

\begin{document}

\begin{center}
\textbf{SEQUENCE AND SERIES} \hfill \textbf{6.3}
\end{center}

\begin{center}
\underline{\textbf{OBJECTIVE QUESTIONS}}
\end{center}

\begin{enumerate}

\item If the sum of the first 2n terms of the A.P. 2, 5, 8, \ldots is equal to the sum of the first n terms of the A.P. 57, 59, 61, \ldots , then n equals

\begin{tabular}{ll}
$(a)$ 10 & $(b)$ 12 \\
$(c)$ 11 & $(d)$ 13
\end{tabular}

\item If x, y, z are three positive numbers in G.P. then
$\dfrac{1}{1 + \ln x}, \dfrac{1}{1 + \ln y}, \dfrac{1}{1 + \ln z}$ are in

\begin{tabular}{l}
$(a)$ A.P. \\
$(b)$ G.P. \\
$(c)$ H.P. \\
$(d)$ none of these
\end{tabular}

\item Let a, b, c are in A.P. and $a^3, b^3, c^3$ are in G.P. If a

$< b < c$ and $a + b + c = \dfrac{3}{2}$, then value of a is

\begin{tabular}{ll}
$(a)$ $\dfrac{1}{2\sqrt{2}}$ & $(b)$ $\dfrac{1}{2\sqrt{3}}$ \\[10pt]
$(c)$ $\dfrac{1}{2} - \dfrac{1}{\sqrt{3}}$ & $(d)$ $\dfrac{1}{2} - \dfrac{1}{\sqrt{2}}$
\end{tabular}

\item If 1, $\log_y x$, $\log_z y$, $-15 \log_x z$ are in A.P., then

\begin{tabular}{ll}
$(a)$ $z^3 = x$ & $(b)$ $x = y^{-1}$ \\
$(c)$ $z^{-3} = y$ & $(d)$ all of these
\end{tabular}

\item The sum to n terms of the series
$\dfrac{2}{3} + \dfrac{8}{9} + \dfrac{26}{27} + \dfrac{80}{81} + \ldots$ is equal to

\begin{tabular}{ll}
$(a)$ $1 - \left(\dfrac{1}{3}\right)^n$ & $(b)$ $2 - \dfrac{1}{2}\left(\dfrac{2}{3}\right)^n$ \\[10pt]
$(c)$ $n - n\left(\dfrac{1}{3}\right)^n$ & $(d)$ none of these
\end{tabular}

\item Let $a_n$ be the nth term of a G.P. of positive integers.

Let $\displaystyle\sum_{n=1}^{100} a_{2n} = \alpha$ and $\displaystyle\sum_{n=1}^{100} a_{2n+1} = \beta$ such that $\alpha \neq \beta$.

Thus common ratio of the G.P. is

\begin{tabular}{ll}
$(a)$ $\dfrac{\alpha}{\beta}$ & $(b)$ $\dfrac{\beta}{\alpha}$ \\[10pt]
$(c)$ $\left(\dfrac{\alpha}{\beta}\right)^{1/2}$ & $(d)$ $\left(\dfrac{\beta}{\alpha}\right)^{1/2}$
\end{tabular}

\item Let $T_r$ be the rth term of an A.P. for $r = 1, 2, 3 \ldots$ If for some positive integers m, n

$T_m = \dfrac{1}{n}$, $T_n = \dfrac{1}{m}$, then $T_{mn}$ equals

\begin{tabular}{ll}
$(a)$ $\dfrac{1}{mn}$ & $(b)$ $\dfrac{1}{m} + \dfrac{1}{n}$ \\[10pt]
$(c)$ 1 & $(d)$ 0
\end{tabular}

\item If $a_1, a_2, \ldots, a_n$ are positive real numbers whose product is a fixed number c, then the minimum value of $a_1 + a_2 + \ldots a_{n-1} + 2a_n$ is

\begin{tabular}{l}
$(a)$ $n(2c)^{1/n}$ \\
$(b)$ $(n + 1)c^{1/n}$ \\
$(c)$ $2nc^{1/n}$ \\
$(d)$ $(n + 1)(2c)^{1/n}$
\end{tabular}

\item If the first term of an A.P. of consecutive integers is $p^2 + 1$, then sum of $(2p + 1)$ terms of this series can be expressed as

\begin{tabular}{ll}
$(a)$ $(p + 1)^2$ & $(b)$ $(2p + 1)(p + 1)^2$ \\
$(c)$ $(p + 1)^3$ & $(d)$ $p^3 + (p + 1)^3$
\end{tabular}

\item If $x > 1$ and $\left(\dfrac{1}{x}\right)^a, \left(\dfrac{1}{x}\right)^b, \left(\dfrac{1}{x}\right)^c$ are in G.P., then

a, b, c are in

\begin{tabular}{ll}
$(a)$ A.P. & $(b)$ G.P. \\
$(c)$ H.P. & $(d)$ none of these
\end{tabular}

\item If the positive numbers a, b, c, d be in A.P, then abc, abd, bcd are

\begin{tabular}{l}
$(a)$ not in A.P., G.P. and H.P. \\
$(b)$ in A.P. \\
$(c)$ in G.P. \\
$(d)$ in H.P.
\end{tabular}

\item The sixth term of an A.P. is 2 and its common difference is greater than one. The value of the common difference of the progression so that the product of the first, fourth and fifth terms is greatest, is

\begin{tabular}{ll}
$(a)$ $\dfrac{8}{5}$ & $(b)$ $\dfrac{2}{3}$ \\[10pt]
$(c)$ $\dfrac{5}{8}$ & $(d)$ $\dfrac{3}{2}$
\end{tabular}

\end{enumerate}
