\begin{tabular}{ll}
$\sin = \frac{\text{Opp}}{\text{Hyp}}$ & $\cos = \frac{\text{Adj}}{\text{Hyp}}$ \\
$\tan = \frac{\text{Opp}}{\text{Adj}}$ & $\cot = \frac{\text{Adj}}{\text{Opp}}$ \\
$\sec = \frac{\text{Hyp}}{\text{Adj}}$ & $\cosec = \frac{\text{Hyp}}{\text{Opp}}$
\end{tabular}

\subsection*{2) Imp. Results:}

\begin{enumerate}
    \item $\sin \theta \cdot \cosec \theta = 1$
    \item $\cos \theta \cdot \sec \theta = 1$
    \item $\tan \theta \cdot \cot \theta = 1$
    \item $\sin \theta = \frac{\text{Opp}}{\text{Hyp}}$
    \item $\cot \theta = \frac{\cos \theta}{\sin \theta}$
    \item $\sin^2 \theta + \cos^2 \theta = 1$
    \item $\sec^2 \theta = 1 + \tan^2 \theta$
    \item $\cosec^2 \theta = 1 + \cot^2 \theta$
    \item $\pi^\circ = 180^\circ$
    \item length of arc $= r \theta$
    \item Area of sector $= \frac{1}{2} r \theta$
\end{enumerate}

\subsection*{4) Allied Angles:}

\begin{enumerate}
    \item $\sin(-\theta) = -\sin \theta$
    \item $\cos(-\theta) = \cos \theta$
    \item $\tan(-\theta) = -\tan \theta$
    \item $\cot(-\theta) = -\cot \theta$
    \item $\sec(-\theta) = \sec \theta$
    \item $\cosec(-\theta) = -\cosec \theta$
\end{enumerate}

\begin{enumerate}
    \item $\sin(90^\circ - \theta) = \cos \theta$
    \item $\cos(90^\circ - \theta) = \sin \theta$
    \item $\tan(90^\circ - \theta) = \cot \theta$
    \item $\cot(90^\circ - \theta) = \tan \theta$
    \item $\sec(90^\circ - \theta) = \cosec \theta$
    \item $\cosec(90^\circ - \theta) = \sec \theta$
\end{enumerate}

\subsection*{5) Compound Angles:}

\begin{enumerate}
    \item $\sin(A+B) = \sin A \cos B + \cos A \sin B$
    \item $\sin(A-B) = \sin A \cos B - \cos A \sin B$
    \item $\cos(A+B) = \cos A \cos B - \sin A \sin B$
    \item $\cos(A-B) = \cos A \cos B + \sin A \sin B$
    \item $\tan(A+B) = \frac{\tan A + \tan B}{1 - \tan A \tan B}$
    \item $\tan(A-B) = \frac{\tan A - \tan B}{1 + \tan A \tan B}$
\end{enumerate}

\subsection*{6) Multiple Angle and Sub Multiple Angle:}

\begin{enumerate}
    \item $\sin 2A = 2 \sin A \cos A$
    \item $\cos 2A = \cos^2 A - \sin^2 A$
    \item $\sin A = 2 \sin\left(\frac{A}{2}\right) \cos\left(\frac{A}{2}\right)$
    \item $\cos A = \cos^2\left(\frac{A}{2}\right) - \sin^2\left(\frac{A}{2}\right)$
    \item $\cos A = 1 - 2 \sin^2\left(\frac{A}{2}\right)$
    \item $\cos A = 2 \cos^2\left(\frac{A}{2}\right) - 1$
    \item $\cos A = \frac{1 - \tan^2\left(\frac{A}{2}\right)}{1 + \tan^2\left(\frac{A}{2}\right)}$
    \item $\sin A = \frac{2 \tan\left(\frac{A}{2}\right)}{1 + \tan^2\left(\frac{A}{2}\right)}$
    \item $\tan A = \frac{2 \tan\left(\frac{A}{2}\right)}{1 - \tan^2\left(\frac{A}{2}\right)}$
    \item $\cos^2\left(\frac{A}{2}\right) = \frac{1 + \cos A}{2}$
    \item $1 = \cos A - 2 \sin^2\left(\frac{A}{2}\right)$
\end{enumerate}

\subsection*{7) Triple Angle Formulae:}

\begin{enumerate}
    \item $\sin 3\theta = 3 \sin \theta - 4 \sin^3 \theta$
    \item $\cos 3\theta = 4 \cos^3 \theta - 3 \cos \theta$
    \item $\tan 3\theta = \frac{3 \tan \theta - \tan^3 \theta}{1 - 3 \tan^2 \theta}$
    \item $\cos^3 \theta = \frac{\cos 3\theta + 3 \cos \theta}{4}$
    \item $\sin^3 \theta = \frac{3 \sin \theta - \sin 3\theta}{4}$
\end{enumerate}

\subsection*{8) Factorization Formulae:}

\begin{enumerate}
    \item $\sin C + \sin D = 2 \sin\left(\frac{C+D}{2}\right) \cos\left(\frac{C-D}{2}\right)$
    \item $\sin C - \sin D = 2 \sin\left(\frac{C-D}{2}\right) \cos\left(\frac{C+D}{2}\right)$
    \item $\cos C + \cos D = 2 \cos\left(\frac{C+D}{2}\right) \cos\left(\frac{C-D}{2}\right)$
    \item $\cos C - \cos D = 2 \sin\left(\frac{C+D}{2}\right) \sin\left(\frac{D-C}{2}\right)$ if $C > D$
    \item $= 2 \sin\left(\frac{D+C}{2}\right) \sin\left(\frac{D-C}{2}\right)$ if $D > C$
\end{enumerate}

\subsection*{9) Defactorization Formulae:}

\begin{enumerate}
    \item $2 \sin A \cos B = \sin(A+B) + \sin(A-B)$
    \item $2 \cos A \sin B = \sin(A+B) - \sin(A-B)$
    \item $2 \cos A \cos B = \cos(A+B) + \cos(A-B)$
    \item $2 \sin A \sin B = \cos(A-B) - \cos(A+B)$
\end{enumerate}

\section*{Co-Ordinate - Geometry (2D)}

\begin{enumerate}
    \item Distance Formula: $\sqrt{(x_2-x_1)^2 + (y_2-y_1)^2}$
    \item Section Formula: $P(x,y) = \left(\frac{mx_2+nx_1}{m+n}, \frac{my_2+ny_1}{m+n}\right)$
    \item Mid point Formula: $\left(\frac{x_1+x_2}{2}, \frac{y_1+y_2}{2}\right)$
    \item Centroid of triangle: $G\left(\frac{x_1+x_2+x_3}{3}, \frac{y_1+y_2+y_3}{3}\right)$
    \item Area of triangle $\Delta = \frac{1}{2} |x_1(y_2-y_3) + x_2(y_3-y_1) + x_3(y_1-y_2)|$
    \item Slope of straight line: $m = \tan \theta$
    \item One point Form: $y-y_1 = m(x-x_1)$
    \item Slope Intercept from: $y = mx + C$
    \item Two point Form: $y-y_1 = \frac{y_2-y_1}{x_2-x_1}(x-x_1)$
    \item Intercept Form: $\frac{x}{a} + \frac{y}{b} = 1$
    \item Normal Form: $x \cos \alpha + y \sin \alpha = p$
    \item Slope of line $ax + by + c = 0$ is $-\frac{a}{b}$
    \item Intersection of two lines: $L_1 + \lambda L_2 = 0$
    \item Angle between two s.t. lines: $\tan \theta = \left|\frac{m_1-m_2}{1+m_1m_2}\right|$
    \item If the lines are parallel: $m_1 = m_2$
    \item If the lines are $L_1': m_1x + c_1 = 0$ and $L_2': m_2x + c_2 = 0$
    \item Length of $L'$ form: $\frac{|ax_1+by_1+c|}{\sqrt{a^2+b^2}}$
    \item Bisector Equations: $\frac{ax_1+by_1+c_1}{\sqrt{a_1^2+b_1^2}} = \pm \frac{ax_2+by_2+c_2}{\sqrt{a_2^2+b_2^2}}$
    \item Distance betw. parallel lines: $\frac{|c_1-c_2|}{\sqrt{a^2+b^2}}$
    \item Homogeneous Equation: $ax^2 + 2hxy + by^2 = 0$
    \item General Equation: $ax^2 + 2hxy + by^2 + 2gx + 2fy + c = 0$ represents a pair of lines
    \item If $ab - 2gh - af^2 - bg^2 - cf^2 = 0$
    \item Point of intersection $(x,y) = \left(\frac{hf-bg}{ab-h^2}, \frac{hg-af}{ab-h^2}\right)$
\end{enumerate}

\section*{Co-Ordinate - Geometry (3D)}

\begin{enumerate}
    \item Distance Formula: $\sqrt{(x_2-x_1)^2 + (y_2-y_1)^2 + (z_2-z_1)^2}$
    \item Section Formula: $\left(\frac{mx_2+nx_1}{m+n}, \frac{my_2+ny_1}{m+n}, \frac{mz_2+nz_1}{m+n}\right)$
\end{enumerate}
