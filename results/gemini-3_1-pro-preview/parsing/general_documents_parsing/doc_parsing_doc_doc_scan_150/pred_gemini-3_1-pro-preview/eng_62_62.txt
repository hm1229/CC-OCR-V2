\section*{Mechanics}
\begin{multicols}{3}
    \noindent
    $x = x_0 + v_{x0}t + \frac{1}{2}a_xt^2$ \\[1ex]
    $v = v_0 + at$ \\[1ex]
    $v^2 - v_0^2 = 2a(x - x_0)$ \\[1ex]
    $\bar{a} = \frac{\sum \vec{F}}{m} = \frac{\vec{F}_{net}}{m}$ \\[1ex]
    $|\vec{F}_{friction}| \le \mu |\vec{F}_{Normal}|$ \\[1ex]
    $\vec{p} = m\vec{v}$ \\[1ex]
    $\Delta \vec{p} = \vec{F} \Delta t$ \\[1ex]
    $KE = \frac{1}{2}mv^2$ \\[1ex]
    $\Delta PE = mg\Delta y$ \\[1ex]
    $\Delta E = W = Fd\cos\theta$

    \columnbreak

    \noindent
    $a_c = \frac{v^2}{r}$ \\[1ex]
    $\theta = \theta_0 + \omega_0t + \frac{1}{2}\alpha t^2$ \\[1ex]
    $\omega = \omega_0 + \alpha t$ \\[1ex]
    $T = \frac{2\pi}{\omega} = \frac{1}{f}$ \\[1ex]
    $v = f\lambda$ \\[1ex]
    $x = A\cos(2\pi ft)$ \\[1ex]
    $\vec{\alpha} = \frac{\sum \vec{\tau}}{I} = \frac{\vec{\tau}_{net}}{I}$ \\[1ex]
    $\vec{\tau} = r \times F$ \\[1ex]
    $L = I\omega$ \\[1ex]
    $\Delta L = \tau \Delta t$

    \columnbreak

    \noindent
    $|\vec{F}_{spring}| = k|\vec{x}|$ \\[1ex]
    $PE_{spring} = \frac{1}{2}kx^2$ \\[1ex]
    $T_{spring} = 2\pi \sqrt{\frac{m}{k}}$ \\[1ex]
    $T_{pendulum} = 2\pi \sqrt{\frac{\ell}{g}}$ \\[1ex]
    $|\vec{F}_{gravity}| = G \frac{m_1m_2}{r^2}$ \\[1ex]
    $|\vec{F}_{gravity}| = m\vec{g}$ \\[1ex]
    $PE_{gravity} = -G \frac{m_1m_2}{r}$ \\[1ex]
    $\rho = \frac{m}{V}$ \\[1ex]
    $KE = \frac{1}{2}I\omega^2$
\end{multicols}

\section*{Electricity}

    \columnbreak

    \noindent

    $P = I\Delta V$ \\[2ex]
    $\frac{1}{R_{parallel}} = \frac{1}{R_1} + \frac{1}{R_2} + \dots + \frac{1}{R_n}$
\end{multicols}
