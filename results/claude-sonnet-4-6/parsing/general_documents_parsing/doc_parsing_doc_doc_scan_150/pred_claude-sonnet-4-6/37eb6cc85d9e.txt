\noindent\underline{\textbf{Mean}} \hfill \underline{\textbf{Standard Deviation*}}

\vspace{0.5em}

\[
\bar{x} = \frac{1}{n}\sum_{i=1}^{n} x_i
\hspace{3em}
S = \sqrt{\frac{\sum(x_i - \bar{x})^2}{n-1}}
\]

\vspace{0.5em}

\noindent\underline{\textbf{Standard Error of the Mean*}} \hspace{1em} \underline{\textbf{Chi-Square}}

\vspace{0.5em}

\[
SE_{\bar{x}} = \frac{s}{\sqrt{n}}
\hspace{4em}
\chi^2 = \sum \frac{(o - e)^2}{e}
\]

\vspace{0.5em}

\noindent\underline{\textbf{Laws of Probability}}

\noindent If A and B are mutually exclusive, then:
\[
P(\text{A or B}) = P(\text{A}) + P(\text{B})
\]
If A and B are independent, then:
\[
P(\text{A and B}) = P(\text{A}) \times P(\text{B})
\]

\vspace{0.3em}
\noindent $n =$ size of the sample

\vspace{0.3em}
\noindent $s =$ sample standard deviation (i.e., the sample-based estimate of the standard deviation of the population)

\vspace{0.3em}
\noindent $o =$ observed results

\vspace{0.3em}
\noindent $e =$ expected results

\vspace{0.3em}
\noindent Degrees of freedom are equal to the number of distinct possible outcomes minus one.
