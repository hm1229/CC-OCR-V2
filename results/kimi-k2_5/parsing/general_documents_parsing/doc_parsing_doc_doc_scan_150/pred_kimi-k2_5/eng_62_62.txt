\section*{Mechanics}

\begin{minipage}[t]{0.32\textwidth}
\begin{align*}
x &= x_0 + v_{x0}t + \tfrac{1}{2}a_xt^2 \\
v &= v_0 + at \\
v_x^2 - v_{x0}^2 &= 2a(x-x_0) \\
\vec{a} &= \frac{\Sigma\vec{F}}{m} = \frac{\vec{F}_{\text{net}}}{m} \\
|\vec{F}_{\text{friction}}| &\leq \mu|\vec{F}_{\text{Normal}}| \\
\vec{p} &= m\vec{v} \\
\Delta\vec{p} &= \vec{F}\Delta t \\
KE &= \tfrac{1}{2}mv^2 \\
\Delta PE &= mg\Delta y \\
\Delta E &= W = Fd\cos\theta
\end{align*}
\end{minipage}
\begin{minipage}[t]{0.32\textwidth}
\begin{align*}
a_c &= \frac{v^2}{r} \\
\theta &= \theta_0 + \omega_0t + \tfrac{1}{2}\alpha t^2 \\
\omega &= \omega_0 + \alpha t \\
T &= \frac{2\pi}{\omega} = \frac{1}{f} \\
v &= f\lambda \\
x &= A\cos(2\pi ft) \\
\vec{\alpha} &= \frac{\Sigma\vec{\tau}}{I} = \frac{\vec{\tau}_{\text{net}}}{I} \\
\vec{\tau} &= \vec{r} \times \vec{F} \\
L &= I\omega \\
\Delta L &= \tau\Delta t
\end{align*}
\end{minipage}
\begin{minipage}[t]{0.32\textwidth}
\begin{align*}
|\vec{F}_{\text{spring}}| &= k|\vec{x}| \\
PE_{\text{spring}} &= \tfrac{1}{2}kx^2 \\
T_{\text{spring}} &= 2\pi\sqrt{\frac{m}{k}} \\
T_{\text{pendulum}} &= 2\pi\sqrt{\frac{\ell}{g}} \\
|\vec{F}_{\text{gravity}}| &= G\frac{m_1m_2}{r^2} \\
|\vec{F}_{\text{gravity}}| &= m\vec{g} \\
PE_{\text{gravity}} &= -G\frac{m_1m_2}{r} \\
\rho &= \frac{m}{V} \\
KE &= \tfrac{1}{2}I\omega^2
\end{align*}
\end{minipage}

\section*{Electricity}

\begin{align*}
|\vec{F}_E| &= k\left|\frac{q_1q_2}{r^2}\right| & \Delta V &= IR & R &= \frac{\rho\ell}{A} \\
I &= \frac{\Delta q}{\Delta t} & P &= I\Delta V \\
R_{\text{series}} &= R_1 + R_2 + \ldots + R_n & \frac{1}{R_{\text{Parallel}}} &= \frac{1}{R_1} + \frac{1}{R_2} + \ldots + \frac{1}{R_n}
\end{align*}
