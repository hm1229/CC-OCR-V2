\section*{Statistical Analysis and Probability}

\subsection*{Mean}
\begin{equation}
\bar{x} = \frac{1}{n} \sum_{i=1}^{n} x_i
\end{equation}

\subsection*{Standard Deviation*}
\begin{equation}
S = \sqrt{\frac{\sum (x_i - \bar{x})^2}{n-1}}
\end{equation}

\subsection*{Standard Error of the Mean*}
\begin{equation}
SE_{\bar{x}} = \frac{S}{\sqrt{n}}
\end{equation}

\subsection*{Chi-Square}
\begin{equation}
\chi^2 = \sum \frac{(o-e)^2}{e}
\end{equation}

\subsection*{Laws of Probability}
If A and B are mutually exclusive, then:
\begin{equation}
P(A \text{ or } B) = P(A) + P(B)
\end{equation}

If A and B are independent, then:
\begin{equation}
P(A \text{ and } B) = P(A) \times P(B)
\end{equation}

\subsection*{Hardy-Weinberg Equations}
\begin{equation}
p^2 + 2pq + q^2 = 1
\end{equation}
\begin{equation}
p + q = 1
\end{equation}
where:
\begin{itemize}
    \item $p$ = frequency of the dominant allele in a population
    \item $q$ = frequency of the recessive allele in a population
\end{itemize}

\subsection*{Definitions}
\begin{itemize}
    \item $\bar{x}$ = sample mean
    \item $n$ = size of the sample
    \item $s$ = sample standard deviation (i.e., the sample-based estimate of the standard deviation of the population)
    \item $o$ = observed results
    \item $e$ = expected results
    \item Degrees of freedom are equal to the number of distinct possible outcomes minus one.
\end{itemize}

\subsection*{Additional Definitions}
\begin{itemize}
    \item Mode = value that occurs most frequently in a data set
    \item Median = middle value that separates the greater and lesser halves of a data set
    \item Mean = sum of all data points divided by number of data points
    \item Range = value obtained by subtracting the smallest observation (sample minimum) from the greatest (sample maximum)
\end{itemize}
