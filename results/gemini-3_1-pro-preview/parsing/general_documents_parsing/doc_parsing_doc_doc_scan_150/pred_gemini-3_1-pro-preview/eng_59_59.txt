```latex
\begin{center}
\textbf{Formula Sheet for Physics 251}
\end{center}

\begin{minipage}[t]{0.24\textwidth}
\textbf{\underline{Constants}} \\
$R = 8.3145 \frac{\text{J}}{\text{mol} \cdot \text{K}}$ \\
$N_A = 6.02 \times 10^{23}$ \\
$k_B = \frac{R}{N_A} = 1.38 \times 10^{-23} \frac{\text{J}}{\text{m} \cdot \text{K}}$ \\
$e = 1.602 \times 10^{-19} \text{ C}$ \\
$k = \frac{1}{4\pi\epsilon_0} = 9 \times 10^9 \frac{\text{N} \cdot \text{m}^2}{\text{C}^2}$ \\
$\mu_0 = 4\pi \times 10^{-7} \frac{\text{T} \cdot \text{m}}{\text{A}}$ \\
$c = 3 \times 10^8 \frac{\text{m}}{\text{s}}$ \\

\textbf{\underline{Thermal Expansion}} \\
$\Delta L = \alpha L_0 \Delta T$ \\
$\beta = 3\alpha$ \\

\textbf{\underline{Heat}} \\
$Q = mc\Delta T$ \\
$Q = \pm mL$ \\
$Q = nC_V\Delta T$ \\
$Q = nC_P\Delta T$ \\

\textbf{\underline{Heat Current}} \\
$H = k \frac{A}{L}(T_H - T_C)$ \\
$H = Ae\sigma T^4$ \\

\textbf{\underline{Quantities}} \\

\textbf{\underline{Work}} \\
$W = \int_{V_1}^{V_2} P dV$ \\

\textbf{\underline{Isochoric Processes}} \\
$W = 0$ \\
$Q = nC_V\Delta T$ \\
$* \Delta U = nC_V\Delta T$
\end{minipage}
\hfill
\begin{minipage}[t]{0.24\textwidth}
$m_{tot} = nM$ \\
$M = N_A m$ \\
$N = N_A n$ \\

\textbf{\underline{Ideal gases}} \\
$* PV = nRT$ \\
$* PV = NkT$ \\
$* v_{RMS} = \sqrt{\frac{3RT}{M}}$ \\
$* C_V = \frac{3}{2}R \text{ (monatomic)}$ \\
$* C_V = \frac{5}{2}R \text{ (diatomic)}$ \\
$* C_P = C_V + R$ \\

\textbf{\underline{Isobaric Processes}} \\
$W = P\Delta V$ \\
$Q = nC_P\Delta T$ \\
$* \Delta U = nC_V\Delta T$ \\

\textbf{\underline{Isothermal Processes}} \\
$* W = nRT \ln\left(\frac{V_f}{V_i}\right)$ \\
$* Q = nRT \ln\left(\frac{V_f}{V_i}\right)$ \\
$* \Delta U = 0$ \\

\textbf{\underline{Adiabatic Processes}} \\
$* W = -nC_V\Delta T$ \\
$Q = 0$ \\
$* \Delta U = nC_V\Delta T$ \\
$* P_1 V_1^\gamma = P_2 V_2^\gamma$ \\
$* T_1 V_1^{\gamma-1} = T_2 V_2^{\gamma-1}$ \\
$\gamma = \frac{C_P}{C_V}$ \\

\textbf{\underline{Cyclic Processes}} \\
$W = Q$ \\
$\Delta U = 0$
\end{minipage}
\hfill
\begin{minipage}[t]{0.24\textwidth}
\textbf{\underline{Heat Engines}} \\
$W = Q_H + Q_C$ \\
$e = \frac{W}{Q_H}$ \\
$e_c = 1 - \frac{T_C}{T_H}$ \\

\textbf{\underline{Refrigerators}} \\
$W = Q_H + Q_C$ \\
$K = \frac{|Q_C|}{|W|}$ \\
$K_c = \frac{T_C}{T_H - T_C}$ \\

\textbf{\underline{Force, Field, Energy}} \\
\textbf{\underline{and Potential}} \\
$\vec{F} = q\vec{E}$ \\
$\Delta U = q\Delta V$ \\
$V_{ab} = \int_a^b \vec{E} \cdot d\vec{l}$ \\
$\vec{E} = -\nabla V$ \\

\textbf{\underline{1 Point Charge}} \\
$\vec{E} = k \frac{q}{r^2} \hat{r}$ \\
$V = k \frac{q}{r}$ \\

\textbf{\underline{2 Point Charges}} \\
$\vec{F} = k \frac{q_1 q_2}{r^2} \hat{r}$ \\
$U = k \frac{q_1 q_2}{r}$ \\

\textbf{\underline{Distributed Charge}} \\
$\vec{E} = \int k \frac{dq}{r^2} \hat{r}$ \\
$V = \int k \frac{dq}{r}$
\end{minipage}
\hfill
\begin{minipage}[t]{0.24\textwidth}
\textbf{\underline{Many Sources}} \\
$\vec{E}_T = \sum_i \vec{E}_i$ \\
$U_T = \sum_{pairs} U_{ij}$ \\

\textbf{\underline{"Elementary" E Fields}} \\
$\vec{E} = k \frac{Q}{r^2} \hat{r} \text{ sphere}$ \\
$\vec{E} = \frac{\lambda}{2\pi\epsilon_0 r} \hat{r} \text{ line}$ \\
$\vec{E} = \frac{\sigma}{2\epsilon_0} \hat{n} \text{ plane}$ \\
$\vec{E} = \frac{\sigma}{\epsilon_0} \hat{n} \text{ conducting surface}$ \\

\textbf{\underline{Capacitors}} \\
$C = \frac{Q}{V}$ \\
$C_T = \sum_i C_i \text{ parallel}$ \\
$\frac{1}{C_T} = \sum_i \frac{1}{C_i} \text{ series}$ \\
$U = \frac{1}{2}CV^2 = \frac{Q^2}{2C} = \frac{1}{2}QV$ \\
$\dagger C = \epsilon_0 \frac{A}{d}$ \\
$\dagger |E| = \frac{V}{d} = \frac{\sigma}{\epsilon_0}$ \\

\textbf{\underline{Flux and Gauss's Law}} \\
$\Phi = \iint \vec{E} \cdot d\vec{a}$ \\
$\Phi_{closed} = \frac{Q_{enclosed}}{\epsilon_0}$ \\

\textbf{\underline{Current and Current}} \\
\textbf{\underline{Density}} \\
$I = \frac{dq}{dt}$ \\
$I = \iint \vec{J} \cdot d\vec{a}$ \\
$\vec{J} = nqv_d$
\end{minipage}

\vspace{1em}
\noindent
* indicates formulas that are specific to ideal gases \\
$\dagger$ indicates formulas that are specific to parallel-plate capacitors
```
