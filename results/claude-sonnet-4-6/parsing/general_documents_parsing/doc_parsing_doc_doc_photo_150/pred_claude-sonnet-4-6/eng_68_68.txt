\end{tabular}

\textbf{Question 10}

\begin{center}
$(F \rightarrow (Q\ \&\ \neg S))$
\end{center}

\textit{Select one answer. \quad 2 points}

\begin{itemize}
    \item[\textbf{A.}] $\bigcirc$ If the ducks are quacking loudly and the chickens aren't playing in the pond, then Bob is chasing the fox away.
    \item[\textbf{B.}] $\bigcirc$ The ducks are quacking loudly and the chickens aren't playing in the pond unless Bob is chasing the fox away.
    \item[\textbf{C.}] $\bigcirc$ The ducks are quacking loudly and the chickens aren't playing in the pond if Bob is chasing the fox away.
    \item[\textbf{D.}] $\bigcirc$ The ducks are quacking loudly and the chickens aren't playing in the pond only if Bob is chasing the fox away.
\end{itemize}

\textbf{Question 11}

\begin{center}
$((P\ \&\ Q)\ \&\ S)$
\end{center}

\textit{Select one answer. \quad 2 points}

\begin{itemize}
    \item[\textbf{A.}] $\bigcirc$ The ducks are quacking loudly and playing in the pond, and the chickens are also playing in the pond.
\end{itemize}
