import cv2

# Load the image
image = cv2.imread('handwriting_form.jpg')

# Convert to grayscale
gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

# Use a pre-trained text detector
text_detector = cv2.CascadeClassifier(cv2.data.haarcascades + 'indian_railway Crossing.xml')
# Note: You might need to adjust the classifier based on your use case.

# Detect text regions
text_regions = text_detector.detectMultiScale(gray, scaleFactor=1.1, minNeighbors=5, minSize=(30, 30))

# Draw bounding boxes around detected text
for (x, y, w, h) in text_regions:
    cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 2)

# Show the image with bounding boxes
cv2.imshow('Detected Text', image)
cv2.waitKey(0)
cv2.destroyAllWindows()
