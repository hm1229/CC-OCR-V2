\section*{Geometry}
Circle: $\text{circumference}=2\pi R$, $\text{area}=\pi R^2$\\

\section*{Trigonometry}
\begin{align*}
\sin\alpha &= \frac{A}{C}, \quad \cos\alpha = \frac{B}{C} \\
\tan\alpha &= \frac{A}{B} \\[1em]
\frac{\sin\alpha}{A} &= \frac{\sin\beta}{B} = \frac{\sin\gamma}{C} \\
A^2 + B^2 - 2AB\cos\gamma &= C^2
\end{align*}

\section*{Polar Coordinates}
\begin{align*}
x &= r\cos\theta, \quad y = r\sin\theta \\
r &= \sqrt{x^2 + y^2}, \quad \tan\theta = y/x
\end{align*}

\section*{1-d motion, constant $a$}
\begin{align*}
a &= \frac{v_f - v_0}{t} \\
v_{\text{av}} &= \frac{v_0 + v_f}{2},\quad (x_f - x_0) = v_{\text{av}} t \\
(x_f - x_0) &= v_0 t + \frac{1}{2} a t^2 = v_f t - \frac{1}{2} a t^2 \\
\frac{1}{2}v_f^2 - \frac{1}{2}v_0^2 &= a(x_f - x_0)
\end{align*}

\section*{Projectile Motion}
\begin{align*}
\text{Range} &= \frac{v^2}{g}\sin2\theta, \quad \text{Max. height} = \frac{v_0^2}{2g}\sin^2\theta
\end{align*}

\section*{Momentum, Force and Impulse}
\begin{align*}
p = mv,\ F = \frac{\Delta p}{\Delta t} = ma,\ I = F\Delta t = \Delta p
\end{align*}

\section*{Work, Energy and Power}
\begin{align*}
W = \vec{F}\cdot(\vec{r}-\vec{r}_0),\ KE = \frac{1}{2}mv^2,\ P = \frac{\Delta E}{\Delta t}
\end{align*}

\section*{Gravity and circular orbits}
\begin{align*}
PE &= -G\frac{Mm}{r},\ \Delta PE = mgh\ (\text{small } h),\ F = G\frac{Mm}{r^2} = mg \\
a &= \frac{v^2}{r},\ \frac{GM}{4\pi^2} = \frac{R^3}{T^2}
\end{align*}

\section*{Rotational Motion \& Gravity}
\begin{align*}
v &= \omega r = \frac{2\pi r}{T},\quad \omega = 2\pi f = \frac{2\pi}{T},\ f = 1/T \\
\alpha &= \frac{\omega_f - \omega_0}{t} = \frac{a}{r} \\
L &= I\omega = mvr\sin\theta,\ (\theta=\text{angle between v and r}) \\
KE &= \frac{L^2}{2I} = \frac{1}{2}I\omega^2 \\
\tau &= rF\sin\theta,\ I\alpha = \tau,\ I_{\text{point}} = mR^2 \\
I_{\text{cyl.shell}} &= mR^2,\ I_{\text{sphere}} = \frac{2}{5}mR^2\ I_{\text{solid cyl.}} = \frac{1}{2}mR^2.
\end{align*}

\section*{Gases, liquids and solids}
\begin{align*}
PV &= NRT,\ \Delta P = \rho gh,\langle \frac{1}{2}mv^2\rangle = \frac{3}{2}k_B T \\
F_{\text{bouyant}} &= \rho_{\text{displaced liq.}} V_{\text{displaced liq.}} g \\
\text{Stress} &= F/A,\ \text{Strain} = \Delta L/L,\ Y = \text{Stress}/\text{Strain} \\
\frac{\Delta L}{L} &= \frac{F/A}{B} = \frac{-\Delta P}{B}. \\
P_a + \frac{1}{2}\rho_a v_a^2 + \rho_a g h_a &= P_b + \frac{1}{2}\rho_b v_b^2 + \rho_b g h_b \\
\Delta L/L &= \alpha\Delta T,\ \Delta V/V = 3\alpha\Delta T \\
Q &= mC_v\Delta T + mL\ (\text{if phase trans.}),\ \frac{\Delta Q}{\Delta t} = \frac{kA(T_b - T_a)}{x_b - x_a}
\end{align*}

\section*{Radiation}
\begin{align*}
P = e\sigma A T^4
\end{align*}

\section*{Thermodynamics}
\begin{align*}
\Delta U &= Q - W,\ W = P\Delta V,\ \text{ideal gas: } \Delta U = nC_V\Delta T \\
\text{Adiabatic exp: } &pV^\gamma = \text{const},\ TV^{\gamma-1}=\text{const} \\
\gamma &= C_p/C_V=5/3 \text{ for monotonic gas } =7/5 \text{ for diatomic gas} \\
Q &= T\Delta S,\ \Delta S > 0 \\
\text{Engines: } &\epsilon = W/Q_H < (T_H - T_L)/T_H < 1 \\
\text{Refrigerators and heat pumps: } &\epsilon = Q_L/W < T_L/(T_H - T_L)
\end{align*}

\section*{Simple Harmonic Motion and Waves}
\begin{align*}
\text{Spring: } &F = -kx,\ PE = (1/2)kx^2,\ \omega = \sqrt{k/m} \\
f &= \omega/(2\pi),\ x(t) = A\cos(\omega t) + B\sin(\omega t) \\
\text{Pendulum: } &T = 2\pi\sqrt{L/g} \\
\text{Waves: } &y(x,t) = A\sin[2\pi(ft - x/\lambda + \delta)],\ v = f\lambda \\
I &= \text{const }A^2f^2,\ I_2/I_1 = R_1^2/R_2^2 \\
\text{Standing waves: } &\lambda_n = 2L/n \\
\text{Strings: } v &= \sqrt{T/\mu},\ \text{Solid/Liquid: } v = \sqrt{B/\rho} \\
\text{Sound: } &I = E/(A\cdot\Delta t) = \text{Power}/A \\
I_0 &= 10^{-12}\ \text{W/m}^2,\ \text{Intensity in decibels}=10\log_{10}(I/I_0) \\
\text{Beat freq.: } &= |f_1 - f_2|,\ \text{Doppler:} \\
f_{\text{obs}} &= f_{\text{source}}(V_{\text{sound}} \pm v_{\text{obs}})/(V_{\text{sound}} \pm v_{\text{source}}) \\
\text{Pipes: same at both ends: } &L = \lambda/2, \lambda, 3\lambda/2 \\
\text{Pipes: open at only one end: } &L = \lambda/4, 3\lambda/4\cdots
\end{align*}
