\section*{1. Force between two masses (Newton's Law of Gravitation).}

From $F \propto \frac{1}{r^2}$ and $F \propto m_1 m_2$

\begin{equation}
F = G \frac{Mm}{r^2}, \quad G = 6.67 \times 10^{-11}
\end{equation}

\section*{2. Gravitational Field Strength "g" due to mass M.}

\begin{equation}
g = \frac{F}{m} = \frac{GMm}{r^2m}
\end{equation}

Gives

\begin{equation}
g = \frac{-GM}{r^2}
\end{equation}

Where M is source mass and m is test mass.

\section*{3. Gravitational Potential "φ".}

\begin{equation}
\text{work done} = \frac{F \times d}{\text{mass}}
\end{equation}

Gives

\begin{equation}
\varphi = \frac{-GM}{r}
\end{equation}

\section*{4. Gravitational Potential Energy.}

\begin{equation}
\varphi = \frac{-GMm}{r}
\end{equation}

\section*{5. Speed of Satellite in Orbit.}

\begin{equation}
F_g = F_c
\end{equation}

\begin{equation}
\frac{GMm}{r^2} = \frac{mv^2}{r}
\end{equation}

Gives

\begin{equation}
v^2 = \frac{GM}{r}
\end{equation}

\section*{6. Period or Distance of Satellite.}

\begin{equation}
v = \frac{d}{t} = \frac{2\pi r}{T} \quad \text{squaring} \quad v^2 = \frac{4\pi^2 r^2}{T^2}
\end{equation}

\begin{equation}
v^2 = \frac{4\pi^2 r^2}{T^2} = \frac{GM}{r}
\end{equation}

Rearranging gives where r is distance and T is period

\begin{equation}
T^2 = \left(\frac{4\pi^2}{Gm}\right)r^3
\end{equation}
