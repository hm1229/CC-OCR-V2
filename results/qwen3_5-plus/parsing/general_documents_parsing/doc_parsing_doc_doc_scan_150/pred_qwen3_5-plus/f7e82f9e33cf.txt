\section*{formulas for discrete-time LTI signals and systems}
\begin{tabular}{ll}
\toprule
\textbf{name} & \textbf{formula} \\
\midrule
area under impulse & $\displaystyle \sum_{n} \delta(n) = 1$ \\
multiplication by impulse & $f(n)\,\delta(n) = f(0)\,\delta(n)$ \\
\dots by shifted impulse & $f(n)\,\delta(n-n_o) = f(n_o)\,\delta(n-n_o)$ \\
convolution & $\displaystyle f(n) * g(n) = \sum_{k} f(k)\,g(n-k)$ \\
\dots with an impulse & $f(n) * \delta(n) = f(n)$ \\
\dots with a shifted impulse & $f(n) * \delta(n-n_o) = f(n-n_o)$ \\
transfer function & $\displaystyle H(z) = \sum_{n} h(n)\,z^{-n}$ \\
frequency response & $\displaystyle H^f(\omega) = \sum h(n)\,e^{-j\omega n}$ \\
\dots their connection & $H^f(\omega) = H(e^{j\omega})$ \\
& provided unit circle $\subset$ ROC \\
\bottomrule
\end{tabular}

\section*{formulas for continuous-time LTI signals and systems}
\begin{tabular}{ll}
\toprule
\textbf{name} & \textbf{formula} \\
\midrule
area under impulse & $\displaystyle \int \delta(t)\,dt = 1$ \\
multiplication by impulse & $f(t)\,\delta(t) = f(0)\,\delta(t)$ \\
\dots by shifted impulse & $f(t)\,\delta(t-t_o) = f(t_o)\,\delta(t-t_o)$ \\
convolution & $\displaystyle f(t) * g(t) = \int f(\tau)\,g(t-\tau)\,d\tau$ \\
\dots with an impulse & $f(t) * \delta(t) = f(t)$ \\
\dots with a shifted impulse & $f(t) * \delta(t-t_o) = f(t-t_o)$ \\
transfer function & $\displaystyle H(s) = \int h(t)\,e^{-st}\,dt$ \\
frequency response & $\displaystyle H^f(\omega) = \int h(t)\,e^{-j\omega t}\,dt$ \\
\dots their connection & $H^f(\omega) = H(j\omega)$ \\
& provided $j\omega$-axis $\subset$ ROC \\
\bottomrule
\end{tabular}
