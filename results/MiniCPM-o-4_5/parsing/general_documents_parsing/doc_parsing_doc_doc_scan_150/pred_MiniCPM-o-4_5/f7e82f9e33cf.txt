\begin{tabular}{ll}
\multicolumn{2}{c}{formulas for discrete-time LTI signals and systems} \\
\hline
name & formula \\
\hline
area under impulse & $\sum_n \delta(n)=1$ \\
multiplication by impulse & $f(n)\delta(n)=f(0)\delta(n)$ \\
$\ldots$ by shifted impulse & $f(n)\delta(n-n_0)=f(n_0)\delta(n-n_0)$ \\
convolution & $f(n)*g(n)=\sum_k f(k)g(n-k)$ \\
$\ldots$ with an impulse & $f(n)+\delta(n)=f(n)$ \\
$\ldots$ with a shifted impulse & $f(n)*\delta(n-n_0)=f(n-n_0)$ \\
transfer function & $H(z)=\sum_n h(n)z^{-n}$ \\
frequency response & $H^{\prime}(\omega)=\sum_n h(n)e^{-j\omega n}$ \\
$\ldots$ their connection & $H^{\prime}(\omega)=H(e^{j\omega})$ \\
& provided unit circle $\subset$ ROC \\
\hline
\multicolumn{2}{c}{formulas for continuous-time LTI signals and systems} \\
\hline
name & formula \\
\hline
area under impulse & $\int \delta(t)dt=1$ \\
multiplication by impulse & $f(t)\delta(t)=f(0)\delta(t)$ \\
$\ldots$ by shifted impulse & $f(t)\delta(t-t_0)=f(t_0)\delta(t-t_0)$ \\
convolution & $f(t)*g(t)=\int f(\tau)g(t-\tau)d\tau$ \\
$\ldots$ with an impulse & $f(t)*\delta(t)=f(t)$ \\
$\ldots$ with a shifted impulse & $f(t)*\delta(t-t_0)=f(t-t_0)$ \\
transfer function & $H(s)=\int h(t)e^{-st}dt$ \\
frequency response & $H^{\prime}(\omega)=\int h(t)e^{-j\omega t}dt$ \\
$\ldots$ their connection & $H^{\prime}(\omega)=H(j\omega)$ \\
& provided $\omega$-axis $\subset$ ROC \\
\hline
\end{tabular}
