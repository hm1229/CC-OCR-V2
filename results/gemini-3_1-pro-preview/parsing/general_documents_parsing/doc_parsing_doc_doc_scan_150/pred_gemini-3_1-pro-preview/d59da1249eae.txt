    & trapezoid & $A = \frac{1}{2} (b_1 + b_2)h \quad \text{or} \quad A = \frac{(b_1 + b_2)h}{2}$ \\

    \hline
    \textbf{Volume} & prism or cylinder & $V = Bh^*$ \\
    & pyramid or cone & $V = \frac{1}{3} Bh^*$ \\

    \multicolumn{3}{|l|}{\textit{*B represents the area of the Base of a solid figure.}} \\

    \hline
