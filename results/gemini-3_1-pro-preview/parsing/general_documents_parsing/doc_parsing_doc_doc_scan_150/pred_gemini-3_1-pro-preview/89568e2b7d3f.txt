\textbf{Pythagorean identities:}
\begin{itemize}
    \item $\sin^2 \theta + \cos^2 \theta = 1$
    \item $\tan^2 \theta + 1 = \sec^2 \theta$
    \item $1 + \cot^2 \theta = \csc^2 \theta$
\end{itemize}

\textbf{Reciprocal identities:}
\begin{itemize}
    \item $\csc x = \frac{1}{\sin x}$
    \item $\sec x = \frac{1}{\cos x}$
    \item $\cot x = \frac{1}{\tan x}$
\end{itemize}

\textbf{Even - odd identities:}
\begin{itemize}
    \item $\sin(-x) = -\sin x$
    \item $\cos(-x) = \cos x$
    \item $\tan(-x) = -\tan x$
\end{itemize}

\textbf{Product to sum formulas:}
\begin{itemize}
    \item $\sin x \cdot \sin y = \frac{1}{2} \left[ \cos(x - y) - \cos(x + y) \right]$
    \item $\cos x \cdot \cos y = \frac{1}{2} \left[ \cos(x - y) + \cos(x + y) \right]$
    \item $\sin x \cdot \cos y = \frac{1}{2} \left[ \sin(x + y) + \sin(x - y) \right]$
    \item $\cos x \cdot \sin y = \frac{1}{2} \left[ \sin(x + y) - \sin(x - y) \right]$
\end{itemize}

\textbf{Sum to product:}
\begin{itemize}
    \item $\sin x \pm \sin y = 2 \sin \left( \frac{x \pm y}{2} \right) \cos \left( \frac{x \mp y}{2} \right)$
    \item $\cos x + \cos y = 2 \cos \left( \frac{x + y}{2} \right) \cos \left( \frac{x - y}{2} \right)$
    \item $\cos x - \cos y = -2 \sin \left( \frac{x + y}{2} \right) \sin \left( \frac{x - y}{2} \right)$
\end{itemize}

\textbf{Double - angle formulas:}
\begin{itemize}
    \item $\sin 2\theta = 2 \cdot \sin \theta \cos \theta$
    \item $\cos 2\theta = \cos^2 \theta - \sin^2 \theta = 1 - 2 \sin^2 \theta = 2 \cos^2 \theta - 1$
    \item $\tan 2\theta = \frac{2 \tan \theta}{1 - \tan^2 \theta}$
\end{itemize}

\textbf{Co - function identities:}
\begin{itemize}
    \item $\cos \left( \frac{\pi}{2} - x \right) = \sin x$
    \item $\sin \left( \frac{\pi}{2} - x \right) = \cos x$
    \item $\tan \left( \frac{\pi}{2} - x \right) = \cot x$
    \item $\cot \left( \frac{\pi}{2} - x \right) = \tan x$
    \item $\csc \left( \frac{\pi}{2} - x \right) = \sec x$
    \item $\sec \left( \frac{\pi}{2} - x \right) = \csc x$
\end{itemize}

\textbf{Periodicity identities:}
\begin{itemize}
    \item $\sin(x \pm 2\pi) = \sin x$
    \item $\cos(x \pm 2\pi) = \cos x$
    \item $\tan(x \pm \pi) = \tan x$
    \item $\cot(x \pm \pi) = \cot x$
    \item $\sec(x \pm 2\pi) = \sec x$
    \item $\csc(x \pm 2\pi) = \csc x$
\end{itemize}

\textbf{Sum and difference formulas:}
\begin{itemize}
    \item $\sin(x \pm y) = \sin x \cos y \pm \cos x \sin y$
    \item $\cos(x \pm y) = \cos x \cos y \mp \sin x \sin y$
    \item $\tan(x \pm y) = \frac{\tan x \pm \tan y}{1 \mp \tan x \tan y}$
\end{itemize}

\textbf{Half - angle formulas:}
\begin{itemize}
    \item $\sin \left( \frac{x}{2} \right) = \pm \sqrt{\frac{1 - \cos x}{2}}$
    \item $\cos \left( \frac{x}{2} \right) = \pm \sqrt{\frac{1 + \cos x}{2}}$
    \item $\tan \left( \frac{x}{2} \right) = \frac{(1 - \cos x)}{\sin x}$
\end{itemize}

\textbf{Law of sines:}
\begin{itemize}
    \item $\frac{a}{\sin A} = \frac{b}{\sin B} = \frac{c}{\sin C}$
\end{itemize}

\textbf{Law of cosines:}
\begin{itemize}
    \item $a^2 = b^2 + c^2 - 2bc \cos A$
    \item $A = \cos^{-1} \left( \frac{b^2 + c^2 - a^2}{2bc} \right)$
\end{itemize}

\textbf{Area of triangle:}
\begin{itemize}
    \item $\frac{1}{2} ab \sin C$
    \item $\sqrt{s(s - a)(s - b)(s - c)}$, where $s = \frac{1}{2}(a + b + c)$
\end{itemize}
