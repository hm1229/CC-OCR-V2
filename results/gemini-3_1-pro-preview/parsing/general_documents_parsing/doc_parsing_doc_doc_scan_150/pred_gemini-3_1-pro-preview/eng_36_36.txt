\begin{aligned}
\sin C &= \frac{\text{side opposite to angle C}}{\text{hypotenuse}} = \frac{p}{h} \\[10pt]
\cos C &= \frac{\text{side adjacent to angle C}}{\text{hypotenuse}} = \frac{b}{h} \\[10pt]
\tan C &= \frac{\text{side opposite to angle C}}{\text{side adjacent to angle C}} = \frac{p}{b} \\[15pt]
\text{cosec } \theta &= \frac{1}{\sin \theta}; \sec \theta = \frac{1}{\cos \theta}; \cot \theta = \frac{1}{\tan \theta}; \tan \theta = \frac{\sin \theta}{\cos \theta}
\end{aligned}
