\section*{formulas for continuous-time LTI signals and systems}

\begin{tabular}{ll}
\toprule

\midrule

\ldots by shifted impulse & $f(t)\,\delta(t - t_o) = f(t_o)\,\delta(t - t_o)$ \\[10pt]

\ldots with an impulse & $f(t) * \delta(t) = f(t)$ \\[10pt]

\ldots their connection & $H^f(\omega) = H(j\omega)$ \\
 & provided $j\omega$-axis $\subset$ ROC \\
\bottomrule
\end{tabular}
