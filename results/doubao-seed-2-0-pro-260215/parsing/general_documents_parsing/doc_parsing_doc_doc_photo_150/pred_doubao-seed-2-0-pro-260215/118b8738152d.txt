Factor the difference of two squares.

\[
(16x^4 - 81)
\]
\begin{itemize}
    \item[$\circ$] $(2x + 3)^2(2x - 3)^2$
    \item[$\circ$] $(4x^2 + 9)(4x^2 + 9)$
    \item[$\circ$] $(4x^2 + 9)(2x + 3)(2x - 3)$
    \item[$\circ$] $(4x^2 + 9)(4x^2 - 9)$
\end{itemize}

\section*{Question 8}
Factor using the formula for the sum or difference of two cubes.

\[
27x^3 + 64
\]
\begin{itemize}
    \item[$\circ$] $(3x + 4)(9x^2 + 12x + 16)$
    \item[$\circ$] $(3x + 4)(9x^2 + 16)$
    \item[$\circ$] $(3x + 4)(9x^2 - 12x + 16)$
    \item[$\circ$] $(3x - 4)(9x^2 + 12x + 16)$
\end{itemize}
