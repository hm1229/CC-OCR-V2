\section*{PHY 231C FORMULAS}

\subsection*{Quadratic Formula}
\begin{equation}
ax^2 + bx + c = 0
\end{equation}
\begin{equation}
x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}
\end{equation}

\subsection*{Geometry}
Circle: circumference $= 2\pi R$, area $= \pi R^2$

Sphere: area $= 4\pi R^2$, volume $= \frac{4}{3}\pi R^3$

\subsection*{Trigonometry}
\begin{equation}
\sin \alpha = \frac{A}{C}, \quad \cos \alpha = \frac{B}{C}
\end{equation}
\begin{equation}
\tan \alpha = \frac{A}{B}
\end{equation}
\begin{equation}
\frac{\sin \alpha}{A} = \frac{\sin \beta}{B} = \frac{\sin \gamma}{C}
\end{equation}
\begin{equation}
A^2 + B^2 - 2AB \cos \gamma = C^2
\end{equation}

\subsection*{Polar Coordinates}
\begin{equation}
x = r \cos \theta, \quad y = r \sin \theta
\end{equation}
\begin{equation}
r = \sqrt{x^2 + y^2}, \quad \tan \theta = \frac{y}{x}
\end{equation}

\subsection*{SI Units and Constants}
\begin{tabular}{|l|l|l|}
\hline
quantity & unit & abbreviation \\
\hline
Mass $m$ & kilograms & kg \\
Distance $x$ & meters & m \\
Time $t$ & seconds & s \\
Force $F$ & Newtons & N=kg m/s$^2$ \\
Energy $E$ & Joules & J=N m \\
Power $P$ & Watts & W=J/s \\
Temperature $T$ & $^\circ$C, $^\circ$K or $^\circ$F & $T_{^\circ F} = 32 + (9/5)T_{^\circ C}$ \\
Pressure $P$ & Pascals & Pa=N/m$^2$ \\
\hline
\end{tabular}

$g = 9.8 \text{ m/s}^2$, $G = 6.67 \times 10^{-11} \text{ Nm}^2/\text{kg}^2$, $0^\circ \text{C} = 273.15^\circ \text{K}$

$R = 8.31 \text{ J/(mol K)}$, $N_A = 6.023 \times 10^{23}$

$k_B = R/N_A = 1.38 \times 10^{-23} \text{ J/K}$, $1 \text{ cal} = 4.1868 \text{ J}$

$\sigma = 5.67 \times 10^{-8} \text{ W/(m}^2 \text{K}^4)$, $v_{\text{sound}} = 331 \sqrt{T/273} \text{ m/s}$.

\subsection*{1-d motion, constant $a$}
\begin{equation}
a = \frac{v_f - v_0}{t}
\end{equation}
\begin{equation}
v_{\text{av}} = \frac{v_0 + v_f}{2}, \quad (x_f - x_0) = v_{\text{av}} t
\end{equation}
\begin{equation}
(x_f - x_0) = v_0 t + \frac{1}{2}at^2 = v_f t - \frac{1}{2}at^2
\end{equation}
\begin{equation}
\frac{1}{2}v_f^2 - \frac{1}{2}v_0^2 = a(x_f - x_0)
\end{equation}

\subsection*{Projectile Motion}
\begin{equation}
\text{Range} = \frac{v^2}{g} \sin 2\theta, \quad \text{Max. height} = \frac{v_0^2}{2g} \sin^2 \theta
\end{equation}

\subsection*{Momentum, Force and Impulse}
\begin{equation}
p = mv, \quad F = \frac{\Delta p}{\Delta t} = ma, \quad I = F \Delta t = \Delta p
\end{equation}

\subsection*{Work, Energy and Power}
\begin{equation}
W = \vec{F} \cdot (\vec{r} - \vec{r}_0), \quad KE = \frac{1}{2}mv^2, \quad P = \frac{\Delta E}{\Delta t}
\end{equation}

\subsection*{Gravity and circular orbits}
\begin{equation}
PE = -G \frac{Mm}{r}, \quad \Delta PE = mgh (\text{small } h), \quad F = G \frac{Mm}{r^2} = mg
\end{equation}
\begin{equation}
a = \frac{v^2}{r}, \quad \frac{GM}{r} = \frac{R^3}{T^2}
\end{equation}

\subsection*{Rotational Motion \& Gravity}
\begin{equation}
v = \omega r = \frac{2\pi r}{T}, \quad \omega = 2\pi f = \frac{2\pi}{T}, \quad f = 1/T
\end{equation}
\begin{equation}
\alpha = \frac{\omega_f - \omega_0}{t} = \frac{a}{r}
\end{equation}
\begin{equation}
L = I\omega = mvr \sin \theta, \quad (\theta = \text{angle between } v \text{ and } r)
\end{equation}
\begin{equation}
KE = \frac{L^2}{2I} = \frac{1}{2}I\omega^2
\end{equation}
\begin{equation}
\tau = rF \sin \theta, \quad I\alpha = \tau, \quad I_{\text{point}} = mR^2
\end{equation}
\begin{equation}
I_{\text{cyl. shell}} = mR^2, \quad I_{\text{sphere}} = \frac{2}{5}mR^2, \quad I_{\text{solid cyl.}} = \frac{1}{2}mR^2
\end{equation}

\subsection*{Gases, liquids and solids}
\begin{equation}
PV = NRT, \quad \Delta P = \rho gh, \quad \langle \frac{1}{2}mv^2 \rangle = \frac{3}{2}k_B T
\end{equation}
\begin{equation}
F_{\text{buoyant}} = \rho_{\text{displaced liq.}} V_{\text{displaced liq.}} g
\end{equation}
\begin{equation}
\text{Stress} = F/A, \quad \text{Strain} = \Delta L/L, \quad Y = \text{Stress/Strain}
\end{equation}
\begin{equation}
\frac{\Delta L}{L} = \frac{F/A}{B} = -\frac{\Delta P}{B}
\end{equation}
\begin{equation}
P_a + \frac{1}{2}\rho_a v_a^2 + \rho_a gh_a = P_b + \frac{1}{2}\rho_b v_b^2 + \rho_b gh_b
\end{equation}
\begin{equation}
\Delta L/L = \alpha \Delta T, \quad \Delta V/V = 3\alpha \Delta T
\end{equation}
\begin{equation}
Q = mC_v \Delta T + mL (\text{if phase trans.}), \quad \Delta Q/\Delta t = \frac{kA(T_b - T_a)}{x_b - x_a}
\end{equation}

\subsection*{Radiation}
\begin{equation}
P = \epsilon \sigma AT^4
\end{equation}

\subsection*{Thermodynamics}
\begin{equation}
\Delta U = Q - W, \quad W = P\Delta V, \quad \text{ideal gas: } \Delta U = nC_V \Delta T
\end{equation}
\begin{equation}
\text{Adiabatic exp: } pV^\gamma = \text{const}, \quad TV^{\gamma-1} = \text{const}
\end{equation}
\begin{equation}
\gamma = C_p/C_V = 5/3 \text{ for monatomic gas} = 7/5 \text{ for diatomic gas}
\end{equation}
\begin{equation}
Q = T\Delta S, \quad \Delta S > 0
\end{equation}
\begin{equation}
\text{Engines: } \epsilon = W/Q_H < (T_H - T_L)/T_H < 1
\end{equation}
\begin{equation}
\text{Refrigerators and heat pumps: } \epsilon = Q_L/W < T_L/(T_H - T_L)
\end{equation}

\subsection*{Simple Harmonic Motion and Waves}
\begin{equation}
\text{Spring: } F = -kx, \quad PE = (1/2)kx^2, \quad \omega = \sqrt{k/m}
\end{equation}
\begin{equation}
f = \omega/(2\pi), \quad x(t) = A \cos(\omega t) + B \sin(\omega t)
\end{equation}
\begin{equation}
\text{Pendulum: } T = 2\pi \sqrt{L/g}
\end{equation}
\begin{equation}
\text{Waves: } y(x,t) = A \sin[2\pi(ft - x/\lambda + \delta)], \quad v = f\lambda
\end{equation}
\begin{equation}
I = \text{const} A^2 f^2, \quad I_2/I_1 = R_1^2/R_2^2
\end{equation}
\begin{equation}
\text{Standing waves: } \lambda_n = 2L/n
\end{equation}
\begin{equation}
\text{Strings: } v = \sqrt{T/\mu}, \quad \text{Solid/Liquid: } v = \sqrt{B/\rho}
\end{equation}
\begin{equation}
\text{Sound: } I = E/(A \cdot \Delta t) = \text{Power}/A
\end{equation}
\begin{equation}
I_0 \equiv 10^{-12} \text{ W/m}^2, \quad \text{Intensity in decibels} = 10 \log_{10}(I/I_0)
\end{equation}
\begin{equation}
\text{Beat freq.} = |f_1 - f_2|, \quad \text{Doppler: }
\end{equation}
\begin{equation}
f_{\text{obs}} = f_{\text{source}}(V_{\text{sound}} \pm v_{\text{obs}})/(V_{\text{sound}} \pm v_{\text{source}})
\end{equation}
\begin{equation}
\text{Pipes: same at both ends: } L = \lambda/2, \lambda, 3\lambda/2
\end{equation}
\begin{equation}
\text{Pipes: open at only one end: } L = \lambda/4, 3\lambda/4 \cdots
\end{equation}
