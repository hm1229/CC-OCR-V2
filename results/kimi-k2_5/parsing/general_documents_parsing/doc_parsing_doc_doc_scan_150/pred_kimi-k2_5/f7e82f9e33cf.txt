\section*{formulas for discrete-time LTI signals and systems}

\begin{tabular}{@{}p{1.5in}p{3in}@{}}
name & formula \\
\midrule
area under impulse & $\displaystyle\sum_n \delta(n) = 1$ \\[6pt]
multiplication by impulse & $f(n)\,\delta(n) = f(0)\,\delta(n)$ \\[6pt]
\ldots by shifted impulse & $f(n)\,\delta(n-n_o) = f(n_o)\,\delta(n-n_o)$ \\[6pt]
convolution & $\displaystyle f(n) * g(n) = \sum_k f(k)\,g(n-k)$ \\[6pt]
\ldots with an impulse & $f(n) * \delta(n) = f(n)$ \\[6pt]
\ldots with a shifted impulse & $f(n) * \delta(n-n_o) = f(n-n_o)$ \\[6pt]
transfer function & $\displaystyle H(z) = \sum_n h(n)\,z^{-n}$ \\[6pt]
frequency response & $\displaystyle H^f(\omega) = \sum_n h(n)\,e^{-j\omega n}$ \\[6pt]
\ldots their connection & $H^f(\omega) = H(e^{j\omega})$ \\
& provided unit circle $\subset$ ROC
\end{tabular}

\section*{formulas for continuous-time LTI signals and systems}

\begin{tabular}{@{}p{1.5in}p{3in}@{}}
name & formula \\
\midrule
area under impulse & $\displaystyle\int \delta(t)\,dt = 1$ \\[6pt]
multiplication by impulse & $f(t)\,\delta(t) = f(0)\,\delta(t)$ \\[6pt]
\ldots by shifted impulse & $f(t)\,\delta(t-t_o) = f(t_o)\,\delta(t-t_o)$ \\[6pt]
convolution & $\displaystyle f(t) * g(t) = \int f(\tau)\,g(t-\tau)\,d\tau$ \\[6pt]
\ldots with an impulse & $f(t) * \delta(t) = f(t)$ \\[6pt]
\ldots with a shifted impulse & $f(t) * \delta(t-t_o) = f(t-t_o)$ \\[6pt]
transfer function & $\displaystyle H(s) = \int h(t)\,e^{-st}\,dt$ \\[6pt]
frequency response & $\displaystyle H^f(\omega) = \int h(t)\,e^{-j\omega t}\,dt$ \\[6pt]
\ldots their connection & $H^f(\omega) = H(j\omega)$ \\
& provided $j\omega$-axis $\subset$ ROC
\end{tabular}
