\begin{aligned}
\int_{0}^{\infty} x^{n} \tilde{h}(x) dx &= c \int_{0}^{\infty} dx \, x^{n+1} h(x) + \int_{0}^{\infty} dx \, x^{n} \int_{x}^{\infty} dy \, \Pi(\log(y/x)) h(y) \\
&\quad + q \int_{0}^{\infty} dx \, x^{n} \int_{0}^{\infty} h(xe^{y}) \nu_{q}(dy) \\

&\quad + q \int_{0}^{\infty} \nu_{q}(dy) \int_{0}^{\infty} dx \, x^{n} h(xe^{y}) \\

&\quad + q \int_{0}^{\infty} \nu_{q}(dy) e^{-(n+1)y} \int_{0}^{\infty} dz \, z^{n} h(z) \\
&= \frac{n! (n+1) c}{\prod_{i=1}^{n+1} (q + \phi(i))} + \frac{(n+1)!}{\prod_{i=1}^{n+1} (q + \phi(i))} \frac{\int_{0}^{\infty} (1 - e^{-(n+1)z}) \Pi(dz)}{n+1} \\

&= \frac{n!}{\prod_{i=1}^{n} (q + \phi(i))}
\end{aligned}
