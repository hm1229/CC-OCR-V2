\section*{2.5 \quad Finance formulas - Practice exercises}

\bigskip

\textsc{Compound Interest Formula:}
\[
a = p\left(1 + \frac{r}{12}\right)^{12y}
\]

\textsc{Equivalent APR Formula:}
\[
\mathrm{APR} = \left(1 + \frac{r}{12}\right)^{12} - 1
\]

\textsc{Future Value Annuity Formula:}
\[
a = p \cdot \frac{\left(1 + \frac{r}{12}\right)^{12y} - 1}{\frac{r}{12}}
\]

\textsc{Loan Payment Formula:}
\[
p = \frac{a \cdot \frac{r}{12}}{1 - \left(1 + \frac{r}{12}\right)^{-12y}}
\]

where
\[
\begin{aligned}
a &= \text{account balance or loan amount (\$)} \\
p &= \text{initial deposit (principal), regular deposit, or regular payment (\$)} \\
y &= \text{time invested (years)} \\
r &= \text{interest rate compounded monthly (as a decimal)}
\end{aligned}
\]
