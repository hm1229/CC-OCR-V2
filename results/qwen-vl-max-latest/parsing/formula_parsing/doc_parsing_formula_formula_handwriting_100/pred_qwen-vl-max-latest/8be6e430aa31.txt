\begin{aligned}
\frac{\partial^2 \lambda_k}{\partial W_{pq}\partial W_{rs}} &= \sum_{j+k}[\frac{\partial W_{kj}}{\partial W_{pq}}\frac{\partial \tau^k_{,j}}{\partial W_{rs}}+\frac{\partial W_{kj}}{\partial W_{rs}}\frac{\partial \tau^k_{,j}}{\partial W_{pq}}] \\
&= \left\{
\begin{array}{ll}
\frac{1}{\lambda_k-\lambda_q}, & p=k,\ q\ne k,\ r=q,\ s=k; \\
\frac{1}{\lambda_k-\lambda_s}, & r=k,\ s\ne k,\ p=s,\ q=k; \\
0, & \text{otherwise}.
\end{array}
\right.
\end{aligned}
