\noindent\textbf{Question 10}

\noindent $(F \rightarrow (Q \ \& \ \sim S))$ \hfill \begin{tabular}[t]{|l|} \selectfont Select one answer. \\ \selectfont 2 points \end{tabular}

\begin{enumerate}
    \item[A.] $\bigcirc$ If the ducks are quacking loudly and the chickens aren't playing in the pond, then Bob is chasing the fox away.
    \item[B.] $\bigcirc$ The ducks are quacking loudly and the chickens aren't playing in the pond unless Bob is chasing the fox away.
    \item[C.] $\bigcirc$ The ducks are quacking loudly and the chickens aren't playing in the pond if Bob is chasing the fox away.
    \item[D.] $\bigcirc$ The ducks are quacking loudly and the chickens aren't playing in the pond only if Bob is chasing the fox away.
\end{enumerate}

\vspace{1cm}

\noindent\textbf{Question 11}

\noindent $((P \ \& \ Q) \ \& \ S)$ \hfill \begin{tabular}[t]{|l|} \selectfont Select one answer. \\ \selectfont 2 points \end{tabular}
