\section*{TRIGONOMETRY}

\subsection*{Angle Sum/Difference Identities}
\begin{align*}
\sin(A+B) &= \sin A \cdot \cos B + \cos A \cdot \sin B \\
\sin(A-B) &= \sin A \cdot \cos B - \cos A \cdot \sin B \\
\cos(A+B) &= \cos A \cdot \cos B - \sin A \cdot \sin B \\
\cos(A-B) &= \cos A \cdot \cos B + \sin A \cdot \sin B \\
\tan(A+B) &= \frac{\tan A + \tan B}{1 - \tan A \tan B} \\
\tan(A-B) &= \frac{\tan A - \tan B}{1 + \tan A \cdot \tan B} \\
\cot(A+B) &= \frac{\cot B \cot A - 1}{\cot B + \cot A} \\
\cot(A-B) &= \frac{\cot B \cot A + 1}{\cot B - \cot A}
\end{align*}

\subsection*{Sum-to-Product Identities}
\begin{align*}
\sin C + \sin D &= 2 \cdot \sin\left(\frac{C+D}{2}\right) \cdot \cos\left(\frac{C-D}{2}\right) \\
\sin C - \sin D &= 2 \cos\left(\frac{C+D}{2}\right) \cdot \sin\left(\frac{C-D}{2}\right) \\
\cos C + \cos D &= 2 \cos\left(\frac{C+D}{2}\right) \cdot \cos\left(\frac{C-D}{2}\right) \\
\cos C - \cos D &= -2 \sin\left(\frac{C+D}{2}\right) \cdot \sin\left(\frac{C-D}{2}\right) \\
&= 2 \sin\left(\frac{C+D}{2}\right) \cdot \sin\left(\frac{D-C}{2}\right)
\end{align*}

\subsection*{Multiple Angle Identities}
\begin{align*}
\sin 2x &= 2 \sin x \cdot \cos x = \frac{2 \tan x}{1 + \tan^2 x} \\
\cos 2x &= \cos^2 x - \sin^2 x = 2 \cos^2 x - 1 \\
&= 1 - 2 \sin^2 x = \frac{1 - \tan^2 x}{1 + \tan^2 x} \\
\tan 2x &= \frac{2 \tan x}{1 - \tan^2 x} \\
\sin 3x &= 3 \sin x - 4 \sin^3 x \\
\cos 3x &= 4 \cos^3 x - 3 \cos x \\
\tan 3x &= \frac{3 \tan x - \tan^3 x}{1 - 3 \tan^2 x}
\end{align*}

\subsection*{Linear Combination of Sine and Cosine}
For $a\sin x + b\cos x$:
\begin{align*}
\text{Max. Value} &= +\sqrt{a^2 + b^2} \\
\text{Min. Value} &= -\sqrt{a^2 + b^2}
\end{align*}

\subsection*{Odd/Even Function Properties}
\begin{itemize}
\item $\text{Odd } F(x) \implies f(-x) = -f(x)$
\item $\text{Even } f(x) \implies F(-x) = +F(x)$
\end{itemize}

\renewcommand{\arraystretch}{1.2}
\begin{tabular}{|l|l|}
\hline
$\cos(-x) = \cos x$ & $\sec(-x) = \sec x$ \\
$\sin(-x) = -\sin x$ & $\cosec(-x) = -\cosec x$ \\
$\tan(-x) = -\tan x$ & $\cot(-x) = -\cot x$ \\
\hline
\end{tabular}

\subsection*{Product-to-Sum Identities}
\begin{align*}
2 \sin A \cdot \cos B &= \sin(A+B) + \sin(A-B) \\
2 \cos A \cdot \sin B &= \sin(A+B) - \sin(A-B) \\
2 \cos A \cdot \cos B &= \cos(A+B) + \cos(A-B) \\
-2 \sin A \cdot \sin B &= \cos(A+B) - \cos(A-B) \\
\sin(A+B) \cdot \sin(A-B) &= \sin^2 A - \sin^2 B \\
\cos(A+B) \cdot \cos(A-B) &= \cos^2 A - \sin^2 B
\end{align*}

\subsection*{Pythagorean and Double Angle Rearrangements}
\begin{tabular}{|l|l|}
\hline
$\sin^2 \theta + \cos^2 \theta = 1$ & $1 + \cos 2x = 2\cos^2 x$ \\
$\sec^2 \theta - \tan^2 \theta = 1$ & $1 - \cos 2x = 2\sin^2 x$ \\
$\cosec^2 \theta - \cot^2 \theta = 1$ & $\tan x = \frac{1 - \cos 2x}{\sin x}$ \\
\hline
\end{tabular}

\subsection*{General Solution of Trigonometric Equations}
\begin{align*}
\sin x = 0 &\implies x = n\pi \\
\cos x = 0 &\implies x = (2n+1)\frac{\pi}{2} \\
\tan x = 0 &\implies x = n\pi \\
\sin x = \sin y &\implies x = n\pi + (-1)^n \cdot y \\
\cos x = \cos y &\implies x = 2n\pi \pm y \\
\tan x = \tan y &\implies x = n\pi + y
\end{align*}

\[
\left.
\begin{aligned}
\sin^2 x = \sin^2 y \\
\cos^2 x = \cos^2 y \\
\tan^2 x = \tan^2 y
\end{aligned}
\right\} \implies x = n\pi \pm y
\]

\[
\boxed{n \in \mathbb{Z}}
