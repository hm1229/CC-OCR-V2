$$\mathbb{E} \int_a \int_a |Kg(t,x) - Kg(s,y)|^p \, dt \, dx \, ds \, dy$$

$$= \mathbb{E} \int_a \int_a \left| \int_a \int_{\mathbb{R}^d} K(t-r,z) g(x-z) \, dz \, dW(r) \right.$$

$$\left. - \int_0^s \int_{\mathbb{R}^d} K(s-r,z) g(y-z) \, dz \, dW(r) \right|^p dt \, dx \, ds \, dy$$

$$\leq N(p) \mathbb{E} \int_a \int_a \left| \int_a \int_{\mathbb{R}^d} (K(t-r,z) - K(s-r,z)) g(y-z) dz \, dW(r) \right|^p$$

$$+ N(p) \mathbb{E} \int_a \int_a \left| \int_0^s \int_{\mathbb{R}^d} K(t-r,x) (g(x-z) - g(y-z)) \, dz \, dW(r) \right|^p$$

$$+ N(p) \mathbb{E} \int_a \int_a \left| \int_s^t \int_{\mathbb{R}^d} K(t-r,z) g(x-z) \, dz \, dW(r) \right|^p dt \, dx \, ds \, dy$$

$$\leq N(p) \mathbb{E} \int_a \int_a \left( \int_0^{s_1} \int_{\mathbb{R}^d} (K(t-r,z) - K(s-r,z)) g(y-z) \, dz \right|^2 dr \right)^{\frac{p}{2}}$$

$$+ N(p) \mathbb{E} \int_a \int_a \left( \int_0^s \left| \int_{\mathbb{R}^d} K(t-r,z) (g(x-z) - g(y-z)) \, dz \right|^2 dr \right)^{\frac{p}{2}}$$

$$+ N(p) \mathbb{E} \int_a \int_a \left( \int_s^t \left| \int_{\mathbb{R}^d} K(t-r,z) g(x-z) \, dz \right|^2 dr \right)^{\frac{p}{2}}$$

$$=: I_1 + I_2 + I_3.$$
