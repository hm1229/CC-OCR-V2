\section*{formulas for discrete-time LTI signals and systems}
\begin{tabular}{ll}
\hline
name & formula \\
\hline
area under impulse & $\sum_{n} \delta(n) = 1$ \\
multiplication by impulse & $f(n) \delta(n) = f(0) \delta(n)$ \\
\ldots by shifted impulse & $f(n) \delta(n-n_{o}) = f(n_{o}) \delta(n-n_{o})$ \\
convolution & $f(n) * g(n) = \sum_{k} f(k) g(n-k)$ \\
\ldots with an impulse & $f(n) * \delta(n) = f(n)$ \\
\ldots with a shifted impulse & $f(n) * \delta(n-n_{o}) = f(n-n_{o})$ \\
transfer function & $H(z) = \sum_{n} h(n) z^{-n}$ \\
frequency response & $H^{I}(\omega) = \sum_{n} h(n) e^{-j\omega n}$ \\
\ldots their connection & $H^{I}(\omega) = H(e^{j\omega})$ \\
& provided unit circle $\subset$ ROC \\
\hline
\end{tabular}

\section*{formulas for continuous-time LTI signals and systems}
\begin{tabular}{ll}
\hline
name & formula \\
\hline
area under impulse & $\int \delta(t) dt = 1$ \\
multiplication by impulse & $f(t) \delta(t) = f(0) \delta(t)$ \\
\ldots by shifted impulse & $f(t) \delta(t-t_{o}) = f(t_{o}) \delta(t-t_{o})$ \\
convolution & $f(t) * g(t) = \int f(\tau) g(t-\tau) d\tau$ \\
\ldots with an impulse & $f(t) * \delta(t) = f(t)$ \\
\ldots with a shifted impulse & $f(t) * \delta(t-t_{o}) = f(t-t_{o})$ \\
transfer function & $H(s) = \int h(t) e^{-st} dt$ \\
frequency response & $H^{I}(\omega) = \int h(t) e^{-j\omega t} dt$ \\
\ldots their connection & $H^{I}(\omega) = H(j\omega)$ \\
& provided $j\omega$-axis $\subset$ ROC \\
\hline
\end{tabular}
