\begin{center}
\begin{tabular}{|l|l|}
\hline
$B$ & Bob is gathering eggs. \\
\hline
$C$ & Bob has fed the chickens. \\
\hline
$D$ & Bob has fed the ducks. \\
\hline
$F$ & Bob is chasing the fox away. \\
\hline
$P$ & The ducks are playing in the pond. \\
\hline
$Q$ & The ducks are quacking loudly. \\
\hline
$S$ & The chickens are playing in the pond. \\
\hline
\end{tabular}
\end{center}

\section*{Question 10}

\[
(F \rightarrow (Q \,\&\, \sim S))
\]

\begin{itemize}
    \item[A.] If the ducks are quacking loudly and the chickens aren't playing in the pond, then Bob is chasing the fox away.
    \item[B.] The ducks are quacking loudly and the chickens aren't playing in the pond unless Bob is chasing the fox away.
    \item[C.] The ducks are quacking loudly and the chickens aren't playing in the pond if Bob is chasing the fox away.
    \item[D.] The ducks are quacking loudly and the chickens aren't playing in the pond only if Bob is chasing the fox away.
\end{itemize}

\section*{Question 11}

\[
((P \,\&\, Q) \,\&\, S)
\]

\begin{itemize}
    \item[A.] The ducks are quacking loudly and playing in the pond, and the chickens are also playing in the pond.
\end{itemize}
