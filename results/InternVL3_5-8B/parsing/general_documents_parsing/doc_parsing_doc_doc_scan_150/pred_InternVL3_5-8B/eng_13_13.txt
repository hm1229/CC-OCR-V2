\subsection*{Law of Probability}
\begin{itemize}
    \item If A and B are mutually exclusive, then:
    \[
    P(A \text{ or } B) = P(A) + P(B)
    \]
    \item If A and B are independent, then:
    \[
    P(A \text{ and } B) = P(A) \times P(B)
    \]
\end{itemize}

\subsection*{Hartl-Winnebcke Equations}
\begin{itemize}
    \item $p$ = frequency of the dominant allele in a population
    \item $q$ = frequency of the recessive allele in a population
    \item $p + q = 1$
    \item $p^2 + 2pq + q^2 = 1$
\end{itemize}

\begin{itemize}
