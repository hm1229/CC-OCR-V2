\[
\|f(u)-f(v)\|_{X}
\le
\left\|Sp^{-\frac12}\big(|u|^{p-1}ug(|u|)-|v|^{p-1}vg(|v|)\big)\right\|_{W^{1+\theta}_{0}(0,T_0)}
+
\left\|D^{2-\frac1s}\big(|u|^{p-1}ug(|u|)-|v|^{p-1}vg(|v|)\big)\right\|_{W^{\theta}_{0}(0,T_0)}
\]
\[
\le
g(\|u\|_{L^\infty_tL^\infty_x(0,T_0)})\,
\Bigg(
\big(\|u\|_{S(0,T_0)}^{p-1}+\|v\|_{S(0,T_0)}^{p-1}\big)
\Big(
\|D^{sp-\frac12}(u-v)\|_{W(0,T_0)}
+\|D^{2-\frac1s}(u-v)\|_{W(0,T_0)}
\Big)
\]
\[
\qquad\qquad
+
\big(\|u\|_{S(0,T_0)}^{p-2}+\|v\|_{S(0,T_0)}^{p-2}\big)
\Big(
\|D^{sp-\frac12}u\|_{W(0,T_0)}
+\|D^{2-\frac1s}u\|_{W(0,T_0)}
+\|D^{sp-\frac12}v\|_{W(0,T_0)}
+\|D^{2-\frac1s}v\|_{W(0,T_0)}
\Big)
\|u-v\|_{S(0,T_0)}
\Bigg)
\]
\[
\le
\Big(g(2CM)(2\delta)^{p-1}+(2\delta)^{p-2}(2CM)\Big)\,\|u-v\|_{X}.
