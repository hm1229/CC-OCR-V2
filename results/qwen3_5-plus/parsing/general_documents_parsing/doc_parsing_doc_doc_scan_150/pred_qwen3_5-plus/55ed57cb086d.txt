```latex
\documentclass{article}
\usepackage[utf8]{inputenc}
\usepackage{geometry}
\geometry{a4paper, margin=1in}
\usepackage{enumitem}
\usepackage{tcolorbox}
\usepackage{titlesec}

% Set section title format to match the underlined, centered title in the image
\titleformat{\section}{\normalfont\large\bfseries\centering}{}{0em}{}[\titlerule]
\titlespacing*{\section}{0pt}{0pt}{10pt}

\begin{document}

\section*{Argumentative Essay Structure}

\begin{tcolorbox}
\textbf{Paragraph 1: Introduction}
\begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=5pt]
    \item Background
    \item (a) Overview of current situation/ Context based on question
    \item (b) Contextual Definition of key terms (if necessary)
    \item (c) Why this issue is interesting/significant
    \begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=2pt]
        \item Persuasive techniques: Hook (Statistics/ Facts/ Social trends from credible sources)/ Anecdote / Rhetorical question / Relevant Quote/ Astounding statement/ Alliteration/ Addressing reader in second person pronoun/ figurative language)
    \end{itemize}
    \item (d) Comparison of different viewpoints (if necessary)
    \item (e) My Stand: Using synonyms (varied vocab) to rephrase question \& defining a boundary for discussion
    \item (f) Summarised Points (of 3 supporting reasons)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}
\textbf{Paragraph 2: Point 1 (Most significant/strongest point)}
\begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=5pt]
    \item State Point (topic sentence)
    \item Give Explanation and Elaboration of Point
    \item Example 1 to support Point
    \item Explanation and Elaboration of Example 1
    \item Example 2 to support Point (if necessary)
    \item Explanation and Elaboration of Example 2 (if necessary)
    \item Link (use synonyms (varied vocab) to further re-emphasise topic sentence)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}
\textbf{Paragraph 3: Point 2 (Second most significant/ strongest point)}
\begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=5pt]
    \item State Point (topic sentence)
    \item Give Explanation and Elaboration of Point
    \item Example 1 to support Point
    \item Explanation and Elaboration of Example 1
    \item Example 2 to support Point (if necessary)
    \item Explanation and Elaboration of Example 2 (if necessary)
    \item Link (use synonyms (varied vocab) to further re-emphasise topic sentence)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}
\textbf{Paragraph 4: Counterargument and Rebuttal}
\begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=5pt]
    \item An argument, usually a popular or common argument, that someone will use to tear down one of your topic sentence or your stand (addressing the other side of the argument to make it seem 'balanced')
    \item A rebuttal to crush/demolish/defeat the counterargument (as a defence of your stand/supporting points)
    \item Explanation \& Examples to support and show why your supporting argument/stand is still stronger
    \item Link (use synonyms (varied vocab) to re-emphasise stand)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}
\textbf{Paragraph 5: Point 3 (Third most significant/ strongest point)}
\begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=5pt]
    \item State Point (topic sentence)
    \item Give Explanation and Elaboration of Point
    \item Example 1 to support Point
    \item Explanation and Elaboration of Example 1
    \item Example 2 to support Point (if necessary)
    \item Explanation and Elaboration of Example 2 (if necessary)
    \item Link (use synonyms (varied vocab) to further re-emphasise topic sentence)
\end{itemize}
\end{tcolorbox}

\begin{tcolorbox}
\textbf{Paragraph 6: Conclusion}
\begin{itemize}[leftmargin=*, label=-, itemsep=2pt, topsep=5pt]
    \item Restate stand (using synonyms (varied vocab) to rephrase question)
    \item Summarised Points
    \item End with a bang! - Statistics/ Rhetorical question / Relevant Quote/ Astounding statement/ Alliteration/ Addressing reader in second person pronoun/ figurative language
    \item Suggest something out of the box (a solution/future possibilities) related to topic/question (if applicable)
\end{itemize}
\end{tcolorbox}

\end{document}
```
