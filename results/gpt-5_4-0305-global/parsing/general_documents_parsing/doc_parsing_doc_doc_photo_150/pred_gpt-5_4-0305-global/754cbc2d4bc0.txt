\[
\text{\underline{TRIGONOMETRY}}
\]

\begin{itemize}
  \item $\sin(A+B)=\sin A\cdot\cos B+\cos A\cdot\sin B$
  \item $\sin(A-B)=\sin A\cdot\cos B-\cos A\cdot\sin B$
  \item $\cos(A+B)=\cos A\cdot\cos B-\sin A\cdot\sin B$
  \item $\cos(A-B)=\cos A\cdot\cos B+\sin A\cdot\sin B$
  \item $\tan(A+B)=\dfrac{\tan A+\tan B}{1-\tan A\cdot\tan B}$
  \item $\tan(A-B)=\dfrac{\tan A-\tan B}{1+\tan A\cdot\tan B}$
  \item $\cot(A+B)=\dfrac{\cot B\cdot\cot A-1}{\cot B+\cot A}$
  \item $\cot(A-B)=\dfrac{\cot B\cdot\cot A+1}{\cot B-\cot A}$
\end{itemize}

\begin{itemize}
  \item odd $f(x)\Rightarrow f(-x)=-f(x)$
  \item Even $f(x)\Rightarrow f(-x)=f(x)$
\end{itemize}

\end{aligned}

\begin{aligned}

\cosec(-x)&=-\cosec x \\

\begin{itemize}
  \item $2\sin A\cdot\cos B=\sin(A+B)+\sin(A-B)$
  \item $2\cos A\cdot\sin B=\sin(A+B)-\sin(A-B)$
  \item $2\cos A\cdot\cos B=\cos(A+B)+\cos(A-B)$
  \item $-2\sin A\cdot\sin B=\cos(A+B)-\cos(A-B)$
  \item $\sin(A+B)\cdot\sin(A-B)=\sin^2A-\sin^2B$
  \item $\cos(A+B)\cdot\cos(A-B)=\cos^2A-\sin^2B$
\end{itemize}

\begin{itemize}
  \item $\sin C+\sin D=2\sin\left(\dfrac{C+D}{2}\right)\cdot\cos\left(\dfrac{C-D}{2}\right)$
  \item $\sin C-\sin D=2\cos\left(\dfrac{C+D}{2}\right)\cdot\sin\left(\dfrac{C-D}{2}\right)$
  \item $\cos C+\cos D=2\cos\left(\dfrac{C+D}{2}\right)\cdot\cos\left(\dfrac{C-D}{2}\right)$
  \item $\cos C-\cos D=-2\sin\left(\dfrac{C+D}{2}\right)\cdot\sin\left(\dfrac{C-D}{2}\right)$
  \[
  =2\sin\left(\dfrac{C+D}{2}\right)\cdot\sin\left(\dfrac{D-C}{2}\right)
  \]
\end{itemize}

\begin{itemize}
  \item $\sin2x=2\sin x\cdot\cos x=\dfrac{2\tan x}{1+\tan^2x}$
  \item $\cos2x=\cos^2x-\sin^2x=2\cos^2x-1$
  \[
  =1-2\sin^2x=\dfrac{1-\tan^2x}{1+\tan^2x}

  \item $\tan2x=\dfrac{2\tan x}{1-\tan^2x}$
  \item $\sin3x=3\sin x-4\sin^3x$
  \item $\cos3x=4\cos^3x-3\cos x$
  \item $\tan3x=\dfrac{3\tan x-\tan^3x}{1-3\tan^2x}$
\end{itemize}

\[
\text{\underline{*General Solution of trigonometry}}
\]

\begin{itemize}
  \item $\sin x=0 \;\Rightarrow\; x=n\pi$
  \item $\cos x=0 \;\Rightarrow\; x=\dfrac{(2n+1)\pi}{2}$
  \item $\tan x=0 \;\Rightarrow\; x=n\pi$
  \item $\sin x=\sin y \;\Rightarrow\; x=n\pi+(-1)^n\cdot y$
  \item $\cos x=\cos y \;\Rightarrow\; x=2n\pi\pm y$
  \item $\tan x=\tan y \;\Rightarrow\; x=n\pi+y$
  \item $\sin^2x=\sin^2y$
  \item $\cos^2x=\cos^2y$
  \item $\tan^2x=\tan^2y$
\end{itemize}

\[
\left.
\begin{aligned}
\sin^2x&=\sin^2y \\
\cos^2x&=\cos^2y \\
\tan^2x&=\tan^2y
\end{aligned}
\right\}
\Rightarrow x=n\pi\pm y
\qquad
\boxed{n\in\mathbb{Z}}
\]
